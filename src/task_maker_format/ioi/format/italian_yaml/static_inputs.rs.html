9/// The iterator over the static input files. It looks for `input/input{}.txt` starting from zero
10/// till it finds the last input. It uses due functions to get the input validator and the output
11/// generator.
12struct StaticInputIter<V, O>
13where
14    V: Fn(Option<SubtaskId>) -> InputValidator,
15    O: Fn(TestcaseId) -> OutputGenerator,
16{
17    /// The path to the input files directory.
18    path: PathBuf,
19    /// The index of the next input file.
20    index: u32,
21    /// The function to use to get the input validator of an input file.
22    get_validator: V,
23    /// The function to use to get the output generator of an input file.
24    get_output_gen: O,
25}
26
27impl<V, O> Iterator for StaticInputIter<V, O>
28where
29    V: Fn(Option<SubtaskId>) -> InputValidator,
30    O: Fn(TestcaseId) -> OutputGenerator,
31{
32    type Item = TaskInputEntry;
33
34    fn next(&mut self) -> Option<Self::Item> {
35        // the first iteration will emit the subtask entry
36        if self.index == 0 {
37            self.index = 1;
38            return Some(TaskInputEntry::Subtask(SubtaskInfo {
39                id: 0,
40                name: Some("static-testcases".into()),
41                max_score: 100.0,
42                is_default: true,
43                input_validator: (self.get_validator)(Some(0)),
44                ..Default::default()
45            }));
46        }
47        let id = self.index - 1; // offset caused by the first iteration
48        let path = self.path.join(format!("input{}.txt", id));
49        if path.exists() {
50            self.index += 1;
51            Some(TaskInputEntry::Testcase(TestcaseInfo::new(
52                id,
53                InputGenerator::StaticFile(path),
54                (self.get_output_gen)(id),
55            )))
56        } else {
57            None
58        }
59    }
60}

62/// Make a new iterator of the static input files inside the `input/` directory relative to the task
63/// root.
64///
65/// - `get_validator` is a function that, given the id of a subtask, returns the input validator
66/// - `get_output_get` is a function that, given the id of a testcase, returns the output generator
67pub(crate) fn static_inputs<P: Into<PathBuf>, V, O>(
68    task_dir: P,
69    get_validator: V,
70    get_output_gen: O,
71) -> Box<dyn Iterator<Item = TaskInputEntry>>
72where
73    V: Fn(Option<SubtaskId>) -> InputValidator + 'static,
74    O: Fn(TestcaseId) -> OutputGenerator + 'static,
75{
76    Box::new(StaticInputIter {
77        path: task_dir.into().join("input"),
78        index: 0,
79        get_validator,
80        get_output_gen,
81    })
82}

105    fn make_task<I: IntoIterator<Item = N>, N: std::fmt::Display>(iter: I) -> TempDir {
106        let dir = TempDir::new().unwrap();
107        fs::create_dir(dir.path().join("input")).unwrap();
108        for tc in iter.into_iter() {
109            fs::write(
110                dir.path().join("input").join(format!("input{}.txt", tc)),
111                "",
112            )
113            .unwrap();
114        }
115        dir
116    }

118    #[test]
119    fn test_some_inputs() {
120        let task = make_task([0, 1, 2]);
121        let entries: Vec<_> =
122            static_inputs(task.path(), get_validator, get_output_generator).collect();
123        if let [Subtask(subtask), Testcase(testcase0), Testcase(testcase1), Testcase(testcase2)] =
124            entries.as_slice()
125        {
126            assert_eq!(subtask.id, 0);
127            assert_eq!(subtask.max_score as u32, 100);
128            match &testcase0.input_generator {
129                InputGenerator::StaticFile(path) => {
130                    assert_eq!(path, &task.path().join("input/input0.txt"))
131                }
132                InputGenerator::Custom(_, _) => panic!("Invalid generator"),
133            }
134            match &testcase1.input_generator {
135                InputGenerator::StaticFile(path) => {
136                    assert_eq!(path, &task.path().join("input/input1.txt"))
137                }
138                InputGenerator::Custom(_, _) => panic!("Invalid generator"),
139            }
140            match &testcase2.input_generator {
141                InputGenerator::StaticFile(path) => {
142                    assert_eq!(path, &task.path().join("input/input2.txt"))
143                }
144                InputGenerator::Custom(_, _) => panic!("Invalid generator"),

146        } else {
147            panic!("Wrong entries returned: {:?}", entries);
