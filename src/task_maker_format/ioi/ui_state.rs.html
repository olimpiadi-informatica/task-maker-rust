36/// Status of the evaluation of a solution on a testcase.
37#[derive(Debug, Clone, Eq, PartialEq)]
38pub enum TestcaseEvaluationStatus {
39    /// The solution has not started yet.
40    Pending,
41    /// The solution is running.
42    Solving,
43    /// The solution exited successfully, waiting for the checker.
44    Solved,
45    /// Checker is running.
46    Checking,
47    /// The solution scored 100% of the testcase.
48    Accepted(String),
49    /// The output is wrong.
50    WrongAnswer(String),
51    /// The solution is partially correct.
52    Partial(String),
53    /// The solution timed out.
54    TimeLimitExceeded,
55    /// The solution exceeded the wall time limit.
56    WallTimeLimitExceeded,
57    /// The solution exceeded the memory limit.
58    MemoryLimitExceeded,
59    /// The solution crashed.
60    RuntimeError,
61    /// Something went wrong.
62    Failed,
63    /// The evaluation has been skipped.
64    Skipped,
65}

67impl From<&TestcaseEvaluationStatus> for Option<TestcaseEvaluationResult> {
68    fn from(status: &TestcaseEvaluationStatus) -> Self {
69        use TestcaseEvaluationResult as TER;
70        use TestcaseEvaluationStatus as TES;
71        match status {
72            TES::Accepted(_) => Some(TER::Accepted),
73            TES::Partial(_) => Some(TER::Partial),
74            TES::WrongAnswer(_) => Some(TER::WrongAnswer),
75            TES::TimeLimitExceeded => Some(TER::TimeLimitExceeded),
76            TES::WallTimeLimitExceeded => Some(TER::WallTimeLimitExceeded),
77            TES::MemoryLimitExceeded => Some(TER::MemoryLimitExceeded),
78            TES::RuntimeError => Some(TER::RuntimeError),
79            _ => None,
80        }
81    }

117impl SolutionTestcaseEvaluationState {
118    /// Checks whether the resources used by a solution on a testcase are close to the limits of
119    /// time or memory.
120    ///
121    /// The time 't' is close to the limit TL if:
122    ///   TL * threshold <= t <= TL / threshold  &&  t <= ceil(TL + extra time) - 0.1s
123    ///
124    /// The second condition guards against a value of extra time too small, which would mark every
125    /// TLE as "close to the limits".
126    ///
127    /// Memory limit is in MiB.
128    pub fn is_close_to_limits(
129        &self,
130        time_limit: Option<f64>,
131        extra_time: f64,
132        memory_limit: Option<u64>,
133        threshold: f64,
134    ) -> bool {
135        for result in self.results.iter().flatten() {
136            let resources = &result.resources;
137            if let Some(time_limit) = time_limit {
138                let lower_bound = time_limit * threshold;
139                let upper_bound = time_limit / threshold;
140                if lower_bound <= resources.cpu_time
141                    && resources.cpu_time <= upper_bound
142                    && resources.cpu_time <= (time_limit + extra_time).ceil() - 0.1
143                {
144                    return true;
145                }
146            }
147            if let Some(memory_limit) = memory_limit {
148                if resources.memory as f64 >= memory_limit as f64 * 1024.0 * threshold {
149                    return true;
150                }
151            }
152        }
153
154        false
155    }
156}

256impl TestcaseEvaluationStatus {
257    /// Whether the testcase evaluation has completed, either successfully or not.
258    pub fn has_completed(&self) -> bool {
259        !matches!(
260            self,
261            TestcaseEvaluationStatus::Pending
262                | TestcaseEvaluationStatus::Solving
263                | TestcaseEvaluationStatus::Solved
264                | TestcaseEvaluationStatus::Checking
265        )
266    }
267
268    /// Whether the testcase evaluation has completed successfully.
269    pub fn is_success(&self) -> bool {
270        matches!(self, TestcaseEvaluationStatus::Accepted(_))
271    }
272
273    /// Whether the testcase evaluation has completed with a partial score.
274    pub fn is_partial(&self) -> bool {
275        matches!(self, TestcaseEvaluationStatus::Partial(_))
276    }
277
278    /// A message representing this status.
279    pub fn message(&self) -> String {
280        use TestcaseEvaluationStatus::*;
281        match self {
282            Pending => "Not done".into(),
283            Solving => "Solution running".into(),
284            Solved => "Solution completed".into(),
285            Checking => "Checker running".into(),
286            Accepted(s) => {
287                if s.is_empty() {
288                    "Output is correct".into()
289                } else {
290                    s.clone()
291                }
292            }
293            WrongAnswer(s) => {
294                if s.is_empty() {
295                    "Output is not correct".into()
296                } else {
297                    s.clone()
298                }
299            }
300            Partial(s) => {
301                if s.is_empty() {
302                    "Partially correct".into()
303                } else {
304                    s.clone()
305                }
306            }
307            TimeLimitExceeded => "Time limit exceeded".into(),
308            WallTimeLimitExceeded => "Execution took too long".into(),
309            MemoryLimitExceeded => "Memory limit exceeded".into(),
310            RuntimeError => "Runtime error".into(),
311            Failed => "Execution failed".into(),
312            Skipped => "Execution skipped".into(),
313        }
314    }

330impl UIState {
331    /// Make a new `UIState`.
332    pub fn new(task: &IOITask, config: ExecutionDAGConfig) -> UIState {
333        let generations = task
334            .subtasks
335            .iter()
336            .map(|(st_num, subtask)| {
337                (
338                    *st_num,
339                    SubtaskGenerationState {
340                        testcases: subtask
341                            .testcases_owned
342                            .iter()
343                            .map(|&tc_num| {
344                                (
345                                    tc_num,
346                                    TestcaseGenerationState {
347                                        status: TestcaseGenerationStatus::Pending,
348                                        generation: None,
349                                        validation: None,
350                                        solution: None,
351                                    },
352                                )
353                            })
354                            .collect(),
355                    },
356                )
357            })
358            .collect();
359        UIState {
360            config,
361            max_score: task.subtasks.values().map(|s| s.max_score).sum(),
362            task: task.clone(),
363            solutions: HashMap::new(),
364            compilations: HashMap::new(),
365            generations,
366            evaluations: HashMap::new(),
367            executor_status: None,
368            booklets: HashMap::new(),
369            diagnostics: Default::default(),
370        }
371    }
372
373    /// Evaluate the checks of all the solutions.
374    ///
375    /// This function should be called only after all the executions have completed.
376    pub fn run_solution_checks(&self) -> Vec<SolutionCheckOutcome> {
377        let mut result = vec![];
378        for (path, solution) in self.solutions.iter() {
379            for check in solution.checks.iter() {
380                let subtasks = self
381                    .task
382                    .find_subtasks_by_pattern_name(&check.subtask_name_pattern);
383                for subtask in subtasks {
384                    let solution_result = self.evaluations.get(path);
385                    // The solution was not run on this subtask.
386                    let Some(solution_result) = solution_result else {
387                        continue;
388                    };
389                    let testcase_results: Vec<Option<TestcaseEvaluationResult>> = self
390                        .task
391                        .subtasks[&subtask.id]
392                        .testcases
393                        .iter()
394                        .map(|testcase_id| (&solution_result.testcases[testcase_id].status).into())
395                        .collect();
396                    if testcase_results.iter().any(Option::is_none) {
397                        continue;
398                    }
399                    let testcase_results = testcase_results
400                        .into_iter()
401                        .map(Option::unwrap)
402                        .collect_vec();
403                    let success = check.result.check(&testcase_results);
404                    result.push(SolutionCheckOutcome {
405                        solution: path.clone(),
406                        check: check.clone(),
407                        subtask_id: subtask.id,
408                        success,
409                    })
410                }
411            }
412        }
413        result
414    }

417impl UIStateT for UIState {
418    /// Apply a `UIMessage` to this state.
419    fn apply(&mut self, message: UIMessage) {
420        match message {
421            UIMessage::StopUI => {}
422            UIMessage::ServerStatus { status } => self.executor_status = Some(status),
423            UIMessage::Solutions { solutions } => {
424                self.solutions = solutions
425                    .into_iter()
426                    .map(|info| (info.path.clone(), info))
427                    .collect()
428            }
429            UIMessage::Compilation { file, status } => self
430                .compilations
431                .entry(file)
432                .or_insert(CompilationStatus::Pending)
433                .apply_status(status),
434            UIMessage::IOITask { .. } => {}
435            UIMessage::IOIGeneration {
436                subtask,
437                testcase,
438                status,
439            } => {
440                let gen = self
441                    .generations
442                    .get_mut(&subtask)
443                    .expect("Subtask is gone")
444                    .testcases
445                    .get_mut(&testcase)
446                    .expect("Testcase is gone");
447                match status {
448                    UIExecutionStatus::Pending => gen.status = TestcaseGenerationStatus::Pending,
449                    UIExecutionStatus::Started { .. } => {
450                        gen.status = TestcaseGenerationStatus::Generating
451                    }
452                    UIExecutionStatus::Done { result } => {
453                        if let ExecutionStatus::Success = result.status {
454                            gen.status = TestcaseGenerationStatus::Generated;
455                        } else {
456                            gen.status = TestcaseGenerationStatus::Failed;
457                        }
458                        gen.generation = Some(result);
459                    }
460                    UIExecutionStatus::Skipped => gen.status = TestcaseGenerationStatus::Skipped,
461                }
462            }
463            UIMessage::IOIValidation {
464                subtask,
465                testcase,
466                status,
467            } => {
468                let Some(gen) = self
469                    .generations
470                    .get_mut(&subtask)
471                    .expect("Subtask is gone")
472                    .testcases
473                    .get_mut(&testcase)
474                else {
475                    return;
476                };
477
478                match status {
479                    UIExecutionStatus::Pending => gen.status = TestcaseGenerationStatus::Pending,
480                    UIExecutionStatus::Started { .. } => {
481                        gen.status = TestcaseGenerationStatus::Validating
482                    }
483                    UIExecutionStatus::Done { result } => {
484                        if let ExecutionStatus::Success = result.status {
485                            gen.status = TestcaseGenerationStatus::Validated;
486                        } else {
487                            gen.status = TestcaseGenerationStatus::Failed;
488                        }
489                        gen.validation = Some(result);
490                    }
491                    UIExecutionStatus::Skipped => {
492                        if let TestcaseGenerationStatus::Failed = gen.status {
493                        } else {
494                            gen.status = TestcaseGenerationStatus::Skipped;
495                        }
496                    }
497                }
498            }
499            UIMessage::IOISolution {
500                subtask,
501                testcase,
502                status,
503            } => {
504                let gen = self
505                    .generations
506                    .get_mut(&subtask)
507                    .expect("Subtask is gone")
508                    .testcases
509                    .get_mut(&testcase)
510                    .expect("Testcase is gone");
511                match status {
512                    UIExecutionStatus::Pending => gen.status = TestcaseGenerationStatus::Pending,
513                    UIExecutionStatus::Started { .. } => {
514                        gen.status = TestcaseGenerationStatus::Solving
515                    }
516                    UIExecutionStatus::Done { result } => {
517                        if let ExecutionStatus::Success = result.status {
518                            gen.status = TestcaseGenerationStatus::Solved;
519                        } else {
520                            gen.status = TestcaseGenerationStatus::Failed;
521                        }
522                        gen.solution = Some(result);
523                    }
524                    UIExecutionStatus::Skipped => {
525                        if let TestcaseGenerationStatus::Failed = gen.status {
526                        } else {
527                            gen.status = TestcaseGenerationStatus::Skipped;
528                        }
529                    }
530                }
531            }
532            UIMessage::IOIEvaluation {
533                testcase,
534                solution,
535                status,
536                part,
537                num_parts,
538                ..
539            } => {
540                let task = &self.task;
541                let eval = self
542                    .evaluations
543                    .entry(solution)
544                    .or_insert_with(|| SolutionEvaluationState::new(task));
545                let testcase = eval.testcases.get_mut(&testcase).expect("Missing testcase");
546                if testcase.results.len() != num_parts {
547                    testcase.results = vec![None; num_parts];
548                }
549                match status {
550                    UIExecutionStatus::Pending => {}
551                    UIExecutionStatus::Started { .. } => {
552                        testcase.status = TestcaseEvaluationStatus::Solving
553                    }
554                    UIExecutionStatus::Done { result } => {
555                        match result.status {
556                            ExecutionStatus::Success => {
557                                testcase.status = TestcaseEvaluationStatus::Solved
558                            }
559                            ExecutionStatus::ReturnCode(_) => {
560                                testcase.status = TestcaseEvaluationStatus::RuntimeError
561                            }
562                            ExecutionStatus::Signal(_, _) => {
563                                testcase.status = TestcaseEvaluationStatus::RuntimeError
564                            }
565                            ExecutionStatus::TimeLimitExceeded => {
566                                testcase.status = TestcaseEvaluationStatus::TimeLimitExceeded
567                            }
568                            ExecutionStatus::SysTimeLimitExceeded => {
569                                testcase.status = TestcaseEvaluationStatus::TimeLimitExceeded
570                            }
571                            ExecutionStatus::WallTimeLimitExceeded => {
572                                testcase.status = TestcaseEvaluationStatus::WallTimeLimitExceeded
573                            }
574                            ExecutionStatus::MemoryLimitExceeded => {
575                                testcase.status = TestcaseEvaluationStatus::MemoryLimitExceeded
576                            }
577                            ExecutionStatus::InternalError(_) => {
578                                testcase.status = TestcaseEvaluationStatus::Failed
579                            }
580                        }
581                        testcase.results[part] = Some(result);
582                    }
583                    UIExecutionStatus::Skipped => {
584                        testcase.status = TestcaseEvaluationStatus::Skipped
585                    }
586                }
587            }
588            UIMessage::IOIChecker {
589                testcase,
590                solution,
591                status,
592                ..
593            } => {
594                let task = &self.task;
595                let eval = self
596                    .evaluations
597                    .entry(solution)
598                    .or_insert_with(|| SolutionEvaluationState::new(task));
599                let testcase = eval.testcases.get_mut(&testcase).expect("Missing testcase");
600                match status {
601                    UIExecutionStatus::Started { .. } => {
602                        testcase.status = TestcaseEvaluationStatus::Checking;
603                    }
604                    UIExecutionStatus::Done { result } => {
605                        testcase.checker = Some(result);
606                    }
607                    _ => {}
608                }
609            }
610            UIMessage::IOITestcaseScore {
611                testcase,
612                solution,
613                score,
614                message,
615                ..
616            } => {
617                let task = &self.task;
618                let eval = self
619                    .evaluations
620                    .entry(solution)
621                    .or_insert_with(|| SolutionEvaluationState::new(task));
622                let testcase = eval.testcases.get_mut(&testcase).expect("Missing testcase");
623                testcase.score = Some(score);
624                if !testcase.status.has_completed() {
625                    testcase.status = match ScoreStatus::from_score(score, 1.0) {
626                        ScoreStatus::WrongAnswer => TestcaseEvaluationStatus::WrongAnswer(message),
627                        ScoreStatus::Accepted => TestcaseEvaluationStatus::Accepted(message),
628                        ScoreStatus::PartialScore => TestcaseEvaluationStatus::Partial(message),
629                    };
630                }
631            }
632            UIMessage::IOISubtaskScore {
633                subtask,
634                solution,
635                score,
636                normalized_score,
637            } => {
638                let task = &self.task;
639                let eval = self
640                    .evaluations
641                    .entry(solution)
642                    .or_insert_with(|| SolutionEvaluationState::new(task));
643                let subtask = eval.subtasks.get_mut(&subtask).expect("Missing subtask");
644                subtask.score = Some(score);
645                subtask.normalized_score = Some(normalized_score);
646            }
647            UIMessage::IOITaskScore { solution, score } => {
648                let task = &self.task;
649                let eval = self
650                    .evaluations
651                    .entry(solution)
652                    .or_insert_with(|| SolutionEvaluationState::new(task));
653                eval.score = Some(score);
654            }
655            UIMessage::IOIBooklet { name, status } => {
656                self.booklets
657                    .entry(name)
658                    .or_insert_with(|| BookletState {
659                        status: UIExecutionStatus::Pending,
660                        dependencies: HashMap::new(),
661                    })
662                    .status = status;
663            }
664            UIMessage::IOIBookletDependency {
665                booklet,
666                name,
667                step,
668                num_steps,
669                status,
670            } => {
671                self.booklets
672                    .entry(booklet)
673                    .or_insert_with(|| BookletState {
674                        status: UIExecutionStatus::Pending,
675                        dependencies: HashMap::new(),
676                    })
677                    .dependencies
678                    .entry(name)
679                    .or_insert_with(|| {
680                        (0..num_steps)
681                            .map(|_| BookletDependencyState {
682                                status: UIExecutionStatus::Pending,
683                            })
684                            .collect()
685                    })
686                    .get_mut(step)
687                    .expect("Statement dependency step is gone")
688                    .status = status;
689            }
690            UIMessage::Diagnostic { diagnostic } => {
691                self.diagnostics.add_diagnostic(diagnostic);
692            }
693            UIMessage::TerryTask { .. }
694            | UIMessage::TerryGeneration { .. }
695            | UIMessage::TerryValidation { .. }
696            | UIMessage::TerrySolution { .. }
697            | UIMessage::TerryChecker { .. }
698            | UIMessage::TerrySolutionOutcome { .. } => unreachable!("Terry message on IOI UI"),
699        }
700    }
