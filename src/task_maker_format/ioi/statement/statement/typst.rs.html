25    fn create_execution(
26        self: Box<Typst>,
27        booklet: &Booklet,
28        booklet_name: String,
29        eval: &mut EvaluationData,
30    ) -> Result<(), Error> {
31        let mut exec = Execution::new(
32            "Compilation of the booklet",
33            ExecutionCommand::system("typst"),
34        );
35
36        exec.args(vec![
37            "compile",
38            "booklet.typ",
39            "--input",
40            "gen_gen=GEN",
41            "--input",
42            "constraints_yaml=constraints.yaml",
43            "--input",
44            "contest_yaml=../../contest.yaml",
45            "--package-cache-path",
46            "typst-cache",
47        ]);
48
49        exec.limits_mut()
50            .read_only(false)
51            .allow_multiprocess()
52            .mount_tmpfs(true)
53            .mount_proc(true)
54            .add_extra_readable_dir("/etc");
55
56        exec.tag(Tag::Booklet.into());
57        exec.priority(BOOKLET_PRIORITY);
58        let output = exec.output("booklet.pdf");
59
60        let source = File::new("Source of the booklet");
61        let typst = self.build_booklet_source(booklet);
62        exec.input(&source, "booklet.typ", false);
63        eval.dag.provide_content(source, typst.into_bytes());
64
65        let contest_yaml = File::new("Contest yaml");
66        exec.input(&contest_yaml, "contest.yaml", false);
67        eval.dag.provide_content(
68            contest_yaml,
69            serde_yaml::to_string(&booklet.config)?.into_bytes(),
70        );
71
72        for statement in booklet.statements.iter() {
73            let name = &statement.config().name;
74            let typst = File::new(format!("Source of statement of {}", name));
75            exec.input(
76                &typst,
77                Path::new(&name).join("statement/statement.typ"),
78                false,
79            );
80            eval.dag
81                .provide_content(typst, self.build_statement_source(statement).into_bytes());
82
83            let task_yaml = File::new(format!("task.yaml for {}", name));
84            exec.input(&task_yaml, Path::new(&name).join("task.yaml"), false);
85            eval.dag.provide_content(
86                task_yaml,
87                serde_yaml::to_string(statement.config())?.into_bytes(),
88            );
89
90            let base_dir = PathBuf::from(&name).join("statement");
91            let deps = statement
92                .build_deps(eval, &booklet_name, &booklet.config)
93                .context("Failed to build booklet dependencies")?;
94
95            for (path, file) in &deps {
96                let path = if path == Path::new("limiti.yaml") {
97                    Path::new("constraints.yaml")
98                } else {
99                    path
100                };
101
102                exec.input(file, base_dir.join(path), false);
103            }
104        }
105
106        // Copy the intro page if needed
107        if let Some(intro_page) = &booklet.config.intro_page {
108            let intro = File::new("Intro page for booklet");
109            exec.input(&intro, "intro_page.typ", false);
110            eval.dag.provide_file(intro, intro_page)?;
111        }
112
113        // Copy the typst cache to provide packages
114        let cache_dir = match env::var("XDG_CACHE_HOME") {
115            Ok(cache) => Path::new(&cache).join("typst/packages"),
116            Err(_) => Path::new(&env::var("HOME")?).join(".cache/typst/packages"),
117        };
118        let glob_pattern = cache_dir.to_string_lossy().to_string() + "/**/*";
119        for path in glob::glob(&glob_pattern).context("Invalid glob pattern")? {
120            let path = path.context("Failed to iterate with glob")?;
121            if !path.is_file() {
122                continue;
123            }
124            let file = File::new(format!("Typst package file at {:?}", path.display(),));
125            eval.dag
126                .provide_file(file.clone(), &path)
127                .context("Failed to provide typst package file")?;
128            exec.input(
129                file,
130                Path::new("typst-cache").join(path.strip_prefix(&cache_dir)?),
131                false,
132            );
133        }
134
135        bind_exec_callbacks!(
136            eval,
137            exec.uuid,
138            |status, name| UIMessage::IOIBooklet { name, status },
139            booklet_name
140        )?;
141        if eval.dag.data.config.copy_logs {
142            let log_dir = eval.task_root.join("bin/logs/booklets");
143            let stderr_dest = log_dir.join(format!("{}.stderr.log", booklet_name));
144            let stdout_dest = log_dir.join(format!("{}.stdout.log", booklet_name));
145            eval.dag
146                .write_file_to_allow_fail(exec.stderr(), stderr_dest, false);
147            eval.dag
148                .write_file_to_allow_fail(exec.stdout(), stdout_dest, false);
149        }
150        let sender = eval.sender.clone();
151        exec.capture_stderr(1024 * 1024 * 1024);
152
153        let dest = booklet.dest.file_name().unwrap().to_owned();
154        eval.dag.on_execution_done(&exec.uuid, move |res| {
155            if let Some(content) = res.stderr {
156                self.emit_warnings(PathBuf::from(dest), content, sender)?;
157            }
158            Ok(())
159        });
160        eval.dag.add_execution(exec);
161
162        eval.dag.write_file_to(output, &booklet.dest, false);
163
164        Ok(())
165    }

167    fn build_statement_source(&self, statement: &Statement) -> String {
168        statement.content.clone()
169    }
170
171    fn build_booklet_source(&self, booklet: &Booklet) -> String {
172        let statements = booklet
173            .statements
174            .iter()
175            .map(|statement| {
176                format!(
177                    "#include \"{}/statement/statement.typ\"",
178                    statement.config().name
179                )
180            })
181            .join("\n");
182
183        if booklet.config.intro_page.is_some() {
184            format!("#include \"intro_page.typ\"\n{}", statements)
185        } else {
186            statements
187        }
188    }

190    fn emit_warnings(
191        &self,
192        booklet_name: PathBuf,
193        content: Vec<u8>,
194        sender: Arc<Mutex<UIMessageSender>>,
195    ) -> Result<(), Error> {
196        if !content.is_empty() {
197            let mut buf = String::new();
198            content.as_slice().read_to_string(&mut buf)?;
199
200            sender.add_diagnostic(
201                Diagnostic::warning(format!(
202                    "The compilation of the booklet at {} produced the following errors or warnings",
203                    booklet_name.display()
204                )).with_note(&buf).with_help("Use --copy-logs to save the compilation stderr")
205            )?;
206
207            // if it seems like typst tried to download a package we emit an error
208            if buf.contains("error: failed to download package") {
209                sender.add_diagnostic(
210                    Diagnostic::error("Typst tried to download a package, but couldn't")
211                        .with_note("This is because the compilation of the booklet is sandboxed")
212                        .with_help(
213                            "Try compiling manually once to store the package in the offline cache",
214                        ),
215                )?;
216            }
217        }
218        Ok(())
219    }
