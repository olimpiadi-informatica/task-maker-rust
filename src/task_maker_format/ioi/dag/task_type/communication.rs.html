14/// The type of communication for the solution in a communication task.
15#[derive(Debug, Clone, Copy, Serialize, Deserialize, TypeScriptify, Eq, PartialEq)]
16#[serde(rename_all = "snake_case")]
17pub enum UserIo {
18    /// Communication is achieved by using stdin/stdout.
19    StdIo,
20    /// Communication is achieved by using the pipes passed in argv.
21    FifoIo,
22}
23
24impl UserIo {
25    /// Used for deserialization.
26    /// Returns UserIo::StdIo.
27    pub fn std_io() -> Self {
28        UserIo::StdIo
29    }
30
31    /// Used for deserialization.
32    /// Returns UserIo::FifoIo.
33    pub fn fifo_io() -> Self {
34        UserIo::FifoIo
35    }
36}

38/// The internal data of a task of type `Batch`.
39#[derive(Debug, Clone, Serialize, Deserialize, TypeScriptify)]
40pub struct CommunicationTypeData {
41    /// The source file of the manager that communicates with the solutions.
42    pub manager: Arc<SourceFile>,
43    /// Number of solution processes to spawn in parallel in a communication task.
44    pub num_processes: u8,
45    /// The type of communication for the solution in a communication task.
46    pub user_io: UserIo,
47}

49/// Internal data of `ScoreSender`.
50#[derive(Debug, Clone)]
51struct ScoreSenderData {
52    /// The id of the current subtask.
53    subtask_id: SubtaskId,
54    /// The id of the current testcase.
55    testcase_id: TestcaseId,
56    /// The sender to use for with the `ScoreManager`.
57    sender: Arc<Mutex<UIMessageSender>>,
58    /// The score manager to use for sending the score.
59    score_manager: Arc<Mutex<ScoreManager>>,
60    /// The number of missing calls to `send` or to `skip`.
61    missing_answers: usize,
62    /// The score to send to the [`ScoreManager`].
63    ///
64    /// This will be sent only then the `missing_answers` counter reaches zero, and if multiple
65    /// answers are received, the smallest one will be sent.
66    answer: Option<(f64, String)>,
67}
68
69/// Utility structure for sending the score only once. Since there are many points where the score
70/// can be generated (the manager, but also each process if it fails), it's easier to centralize the
71/// control of the sending.
72/// It's important that, in case of a failure, the first process that fails is marked as the cause
73/// because it will stop the entire group, maybe letting the other executions fail.
74#[derive(Debug, Clone)]
75struct ScoreSender {
76    /// Interior mutability allowing this struct to be Clone, Send and Sync.
77    data: Arc<Mutex<ScoreSenderData>>,
78}

80/// Evaluate a solution in a task of Batch type.
81#[allow(clippy::too_many_arguments)]
82pub fn evaluate(
83    task: &IOITask,
84    eval: &mut EvaluationData,
85    subtask_id: SubtaskId,
86    testcase_id: TestcaseId,
87    source_file: &SourceFile,
88    input: FileUuid,
89    validation_handle: Option<FileUuid>,
90    _correct_output: Option<FileUuid>,
91    score_manager: Arc<Mutex<ScoreManager>>,
92    data: &CommunicationTypeData,
93) -> Result<(), Error> {
94    let mut group = ExecutionGroup::new(format!(
95        "Evaluation of {} on testcase {}, subtask {}",
96        source_file.name(),
97        testcase_id,
98        subtask_id
99    ));
100
101    let mut fifo_man2sol = Vec::new();
102    let mut fifo_sol2man = Vec::new();
103    for _ in 0..data.num_processes {
104        let fifo1 = group.new_fifo().sandbox_path();
105        fifo_man2sol.push(
106            fifo1
107                .to_str()
108                .ok_or_else(|| anyhow!("Non-UTF8 fifo path"))?
109                .to_string(),
110        );
111        let fifo2 = group.new_fifo().sandbox_path();
112        fifo_sol2man.push(
113            fifo2
114                .to_str()
115                .ok_or_else(|| anyhow!("Non-UTF8 fifo path"))?
116                .to_string(),
117        );
118    }
119
120    let path = source_file.path.clone();
121    let num_processes = data.num_processes as usize;
122    let score_sender = ScoreSender::new(
123        subtask_id,
124        testcase_id,
125        eval.sender.clone(),
126        score_manager,
127        num_processes + 1, // num_processes + the manager
128    );
129    for process_index in 0..num_processes {
130        let mut args = match data.user_io {
131            UserIo::FifoIo => vec![
132                fifo_man2sol[process_index].clone(),
133                fifo_sol2man[process_index].clone(),
134            ],
135            UserIo::StdIo => vec![],
136        };
137        if num_processes > 1 {
138            args.push(process_index.to_string());
139        }
140        let mut sol_exec = source_file
141            .execute(
142                eval,
143                format!(
144                    "Evaluation of {} (process {}/{}) on testcase {}, subtask {}",
145                    source_file.name(),
146                    process_index + 1,
147                    num_processes,
148                    testcase_id,
149                    subtask_id
150                ),
151                args,
152            )
153            .context("Failed to execute solution source file")?;
154        if data.user_io == UserIo::StdIo {
155            sol_exec.stdin_redirect_path(&fifo_man2sol[process_index]);
156            sol_exec.stdout_redirect_path(&fifo_sol2man[process_index]);
157        }
158        sol_exec.tag(Tag::Evaluation.into());
159        sol_exec.priority(EVALUATION_PRIORITY - testcase_id as Priority);
160        let limits = sol_exec.limits_mut();
161        if let Some(time_limit) = task.time_limit {
162            limits.cpu_time(time_limit);
163            limits.wall_time(time_limit * 1.5 + 1.0); // some margin
164        }
165        if let Some(memory_limit) = task.memory_limit {
166            limits.memory(memory_limit * 1024); // MiB -> KiB
167        }
168        bind_exec_callbacks!(
169            eval,
170            sol_exec.uuid,
171            |status, solution| UIMessage::IOIEvaluation {
172                subtask: subtask_id,
173                testcase: testcase_id,
174                solution,
175                status,
176                part: process_index,
177                num_parts: num_processes,
178            },
179            path
180        )?;
181        let score_sender = score_sender.clone();
182        eval.dag.on_execution_done(&sol_exec.uuid, move |result| {
183            if !result.status.is_success() {
184                score_sender.send(0.0, format!("{:?}", result.status))?;
185            } else {
186                // We cannot compute the score here, we should wait for the manager.
187                score_sender.skip()?;
188            }
189            Ok(())
190        });
191        group.add_execution(sol_exec);
192    }
193
194    let mut args = Vec::new();
195    for process_index in 0..num_processes {
196        args.push(&fifo_sol2man[process_index]);
197        args.push(&fifo_man2sol[process_index]);
198    }
199    let mut manager_exec = data
200        .manager
201        .execute(
202            eval,
203            format!(
204                "Manager of {} on testcase {}, subtask {}",
205                source_file.name(),
206                testcase_id,
207                subtask_id
208            ),
209            args,
210        )
211        .context("Failed to execute manager source file")?;
212    manager_exec
213        .tag(Tag::Evaluation.into())
214        .priority(EVALUATION_PRIORITY - testcase_id as Priority)
215        .capture_stdout(128)
216        .capture_stderr(1024);
217    bind_exec_io!(manager_exec, task, input, validation_handle);
218    let limits = manager_exec.limits_mut();
219    if let Some(time_limit) = task.time_limit {
220        let cpu_time = (time_limit + 1.0) * num_processes as f64;
221        let wall_time = cpu_time * 1.5 + 1.0; // some margin
222        limits.cpu_time(cpu_time);
223        limits.wall_time(wall_time);
224    }
225    if let Some(memory_limit) = task.memory_limit {
226        limits.memory(memory_limit * 1024); // MiB -> KiB
227    }
228    bind_exec_callbacks!(
229        eval,
230        manager_exec.uuid,
231        |status, solution| UIMessage::IOIChecker {
232            subtask: subtask_id,
233            testcase: testcase_id,
234            solution,
235            status,
236        },
237        path
238    )?;
239    eval.dag
240        .on_execution_done(&manager_exec.uuid, move |result| {
241            if !result.status.is_success() {
242                score_sender.send(0.0, "Checker failed".to_string())?;
243                return Ok(());
244            }
245            let stdout = result
246                .stdout
247                .ok_or_else(|| anyhow!("Checker stdout not captured"))?;
248            let stderr = result
249                .stderr
250                .ok_or_else(|| anyhow!("Checker stderr not captured"))?;
251            let score = String::from_utf8_lossy(&stdout);
252            let score: f64 = score.trim().parse().context("Invalid score from checker")?;
253            let message = String::from_utf8_lossy(&stderr).trim().to_string();
254            let message = Checker::translate_checker_message(message);
255            score_sender.send(score, message)?;
256            Ok(())
257        });
258    group.add_execution(manager_exec);
259    eval.dag.add_execution_group(group);
260    Ok(())
261}

263impl ScoreSender {
264    /// Make a new `ScoreSender` for a testcase of a solution.
265    fn new(
266        subtask_id: SubtaskId,
267        testcase_id: TestcaseId,
268        sender: Arc<Mutex<UIMessageSender>>,
269        score_manager: Arc<Mutex<ScoreManager>>,
270        num_answers: usize,
271    ) -> ScoreSender {
272        ScoreSender {
273            data: Arc::new(Mutex::new(ScoreSenderData {
274                subtask_id,
275                testcase_id,
276                sender,
277                score_manager,
278                missing_answers: num_answers,
279                answer: None,
280            })),
281        }
282    }
283
284    /// Set the score and message for a testcase. Note that this may be overridden by a call with a
285    /// smaller score.
286    ///
287    /// The score will be sent to the [`ScoreManager`] only if this is the last missing call.
288    fn send(&self, score: f64, message: String) -> Result<(), Error> {
289        let mut data = self.data.lock().unwrap();
290        assert!(
291            data.missing_answers > 0,
292            "send() called with missing_answers = 0"
293        );
294        data.missing_answers -= 1;
295
296        let answer = (score, message);
297        if data.answer.is_none() || data.answer.as_ref().unwrap().0 > score {
298            data.answer = Some(answer);
299        }
300
301        Self::maybe_flush(&data)
302    }
303
304    /// Decrease the number of missing calls by one, but without setting a score/message.
305    ///
306    /// The score will be sent to the [`ScoreManager`] only if this is the last missing call.
307    fn skip(&self) -> Result<(), Error> {
308        let mut data = self.data.lock().unwrap();
309        assert!(
310            data.missing_answers > 0,
311            "skip() called with missing_answers = 0"
312        );
313        data.missing_answers -= 1;
314
315        Self::maybe_flush(&data)
316    }
317
318    /// Check if all the answers have been received, and if so send to the [`ScoreManager`] the score
319    /// and message.
320    fn maybe_flush(data: &ScoreSenderData) -> Result<(), Error> {
321        if data.missing_answers > 0 {
322            return Ok(());
323        }
324        if let Some((score, message)) = &data.answer {
325            data.score_manager
326                .lock()
327                .unwrap()
328                .score(
329                    data.subtask_id,
330                    data.testcase_id,
331                    *score,
332                    message.clone(),
333                    data.sender.clone(),
334                )
335                .with_context(|| {
336                    format!(
337                        "Failed to store testcase score (score: {}, message: {})",
338                        score, message
339                    )
340                })?;
341        }
342        Ok(())
343    }
344}
