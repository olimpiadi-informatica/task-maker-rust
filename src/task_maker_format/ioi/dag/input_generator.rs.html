16/// The source of the input files. It can either be a statically provided input file or a custom
17/// command that will generate an input file.
18#[derive(Debug, Clone, Serialize, Deserialize, TypeScriptify)]
19pub enum InputGenerator {
20    /// Use the static file as input. The file will be copied without transformations.
21    StaticFile(PathBuf),
22    /// Use a custom command to generate the input file. The file has to be printed to stdout.
23    Custom(Arc<SourceFile>, Vec<String>),
24}
25
26impl InputGenerator {
27    /// Build the execution for the generation of the input file. Return the handle to the input
28    /// file and the `Execution` if any. The execution does not send UI messages yet and it's not
29    /// added to the DAG.
30    pub(crate) fn generate(
31        &self,
32        eval: &mut EvaluationData,
33        task_path: &Path,
34        description: String,
35        subtask_id: SubtaskId,
36        testcase_id: TestcaseId,
37    ) -> Result<(FileUuid, Option<Execution>), Error> {
38        match self {
39            InputGenerator::StaticFile(path) => {
40                if !path.exists() {
41                    bail!("COPY from not existing file: {:?}", path);
42                }
43                let file = File::new(format!(
44                    "Static input file of testcase {}, subtask {} from {:?}",
45                    subtask_id, testcase_id, path
46                ));
47                let uuid = file.uuid;
48                eval.dag.provide_file(file, path).with_context(|| {
49                    format!(
50                        "Failed to provide static input file from {}",
51                        path.display()
52                    )
53                })?;
54                Ok((uuid, None))
55            }
56            InputGenerator::Custom(source_file, args) => {
57                let mut exec = source_file
58                    .execute(eval, description, args.clone())
59                    .context("Failed to execute generator source file")?;
60
61                exec.limits_mut().allow_multiprocess();
62                exec.tag(Tag::Generation.into());
63                exec.priority(GENERATION_PRIORITY - testcase_id as Priority);
64
65                // Add limiti.yaml and constraints.yaml file to the sandbox of the generator
66                for filename in &["limiti.yaml", "constraints.yaml"] {
67                    let path = task_path.join("gen").join(filename);
68
69                    if !path.is_file() {
70                        continue;
71                    }
72
73                    let file = File::new(format!("Constraints file at {}", path.display()));
74                    exec.input(&file, filename, false);
75                    eval.dag.provide_file(file, path)?;
76                }
77
78                let stdout = exec.stdout();
79                Ok((stdout.uuid, Some(exec)))
80            }
81        }
82    }
83
84    /// Add the generation of the input file to the DAG and the callbacks to the UI, returning the
85    /// handle to the input file.
86    pub(crate) fn generate_and_bind(
87        &self,
88        eval: &mut EvaluationData,
89        task_path: &Path,
90        subtask_id: SubtaskId,
91        testcase_id: TestcaseId,
92    ) -> Result<FileUuid, Error> {
93        let (input, gen) = self.generate(
94            eval,
95            task_path,
96            format!(
97                "Generation of input file of testcase {}, subtask {}",
98                testcase_id, subtask_id
99            ),
100            subtask_id,
101            testcase_id,
102        )?;
103        eval.dag.write_file_to(
104            input,
105            eval.task_root
106                .join("input")
107                .join(format!("input{}.txt", testcase_id)),
108            false,
109        );
110        // If there is an execution, bind its callbacks and store the input file.
111        if let Some(mut gen) = gen {
112            gen.capture_stderr(STDERR_CONTENT_LENGTH);
113            bind_exec_callbacks!(eval, gen.uuid, |status| UIMessage::IOIGeneration {
114                subtask: subtask_id,
115                testcase: testcase_id,
116                status
117            })?;
118            let sender = eval.sender.clone();
119            let args = gen.args.join(" ");
120            eval.dag.on_execution_done(&gen.uuid, move |result| {
121                if !result.status.is_success() {
122                    let mut diagnostic =
123                        Diagnostic::error(format!("Failed to generate input {}", testcase_id))
124                            .with_note(format!("Generator arguments are: {}", args));
125                    if let Some(stderr) = result.stderr {
126                        diagnostic = diagnostic.with_help_attachment(stderr);
127                    }
128                    sender.add_diagnostic(diagnostic)?;
129                }
130                Ok(())
131            });
132            eval.dag.add_execution(gen);
133        }
134        Ok(input)
135    }
136}
