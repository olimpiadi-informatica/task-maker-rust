12/// Check that the input and output files end with `\n`.
13#[derive(Debug, Default)]
14pub struct IOEndWithNewLine {
15    /// The list of input files that triggered the warning.
16    inputs: Arc<Mutex<Vec<String>>>,
17    /// The list of output files that triggered the warning.
18    outputs: Arc<Mutex<Vec<String>>>,
19}
20make_sanity_check!(IOEndWithNewLine);

22/// Check that a file ends with `\n` and emit a warning if it doesn't. An empty file is considered
23/// valid.
24#[derive(Debug)]
25pub struct CheckEndWithNewLine {
26    /// Whether the last chunk ends with a new line.
27    last_chunk_ends_with_new_line: bool,
28    /// Whether the file is binary, if so, do not emit the warning.
29    is_binary: bool,
30    /// The path of the file that is being checked.
31    path: String,
32    /// Where to insert the warning.
33    list: Arc<Mutex<Vec<String>>>,
34}
35
36impl CheckEndWithNewLine {
37    pub fn new(path: String, list: Arc<Mutex<Vec<String>>>) -> Self {
38        Self {
39            last_chunk_ends_with_new_line: true,
40            is_binary: false,
41            path,
42            list,
43        }
44    }
45
46    pub fn bind(
47        eval: &mut EvaluationData,
48        file: FileUuid,
49        path: String,
50        list: Arc<Mutex<Vec<String>>>,
51    ) {
52        let mut checker = Self::new(path, list);
53        eval.dag
54            .get_file_content_chunked(file, move |chunk| checker.add_chunk(chunk));
55    }
56
57    pub fn add_chunk(&mut self, chunk: &[u8]) -> Result<(), Error> {
58        self.is_binary |= chunk.contains(&0); // UTF-8 never contains NULL bytes.
59        if chunk.is_empty() {
60            if !self.last_chunk_ends_with_new_line && !self.is_binary {
61                self.list.lock().unwrap().push(self.path.clone());
62            }
63        } else {
64            self.last_chunk_ends_with_new_line = chunk.last().map(|&c| c == b'\n').unwrap_or(false);
65        }
66        Ok(())
67    }
68
69    pub fn emit_warning(
70        eval: &mut EvaluationData,
71        files: &[String],
72        kind: &str,
73    ) -> Result<(), Error> {
74        if !files.is_empty() {
75            let files: HashSet<_> = files.iter().collect();
76            let message = format!(
77                "These {} files don't end with a new line: {}",
78                kind,
79                files.iter().sorted().join(", ")
80            );
81            eval.add_diagnostic(
82                Diagnostic::warning(message)
83                    .with_note("It's bad practice to have files that do not end with new-line"),
84            )?;
85        }
86        Ok(())
87    }
88}
89
90impl SanityCheck for IOEndWithNewLine {
91    type Task = IOITask;
92
93    fn name(&self) -> &'static str {
94        "IOEndWithNewLine"
95    }
96
97    fn category(&self) -> SanityCheckCategory {
98        SanityCheckCategory::Io
99    }
100
101    fn pre_hook(&self, task: &IOITask, eval: &mut EvaluationData) -> Result<(), Error> {
102        for (&testcase_id, testcase) in &task.testcases {
103            if let Some(input_file) = testcase.input_file {
104                CheckEndWithNewLine::bind(
105                    eval,
106                    input_file,
107                    format!("input/input{}.txt", testcase_id),
108                    self.inputs.clone(),
109                );
110            }
111            if let Some(output_file) = testcase.official_output_file {
112                CheckEndWithNewLine::bind(
113                    eval,
114                    output_file,
115                    format!("output/output{}.txt", testcase_id),
116                    self.outputs.clone(),
117                );
118            }
119        }
120        Ok(())
121    }
122
123    fn post_hook(&self, _task: &Self::Task, eval: &mut EvaluationData) -> Result<(), Error> {
124        let inputs = self.inputs.lock().unwrap();
125        CheckEndWithNewLine::emit_warning(eval, &inputs, "input")?;
126        let outputs = self.outputs.lock().unwrap();
127        CheckEndWithNewLine::emit_warning(eval, &outputs, "official output")?;
128
129        Ok(())
130    }
131}
