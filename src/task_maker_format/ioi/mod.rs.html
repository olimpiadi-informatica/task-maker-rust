1//! The IOI task format.
2//!
3//! In IOI-like tasks there is the concept of _subtask_ and of _testcase_: a testcase is a single
4//! instance of the evaluation of a solution on a given input file, producing a single output file
5//! which will be used for the scoring. For each solution every testcase is worth from 0.0 to 1.0
6//! points.
7//!
8//! A subtask is a group of testcases, it has a `max_score` parameter which scales its value from
9//! 0.0 to `max_score` points. For computing the score of the subtask a `TestcaseScoreAggregator` is
10//! used. The score of the task for a solution is the sum of all the subtask scores.
11//!
12//! There are many different valid task types, the most common is `Batch` where the solution is
13//! simply executed once per testcase, feeding in the input file (either via stdin or normal file)
14//! and getting the output file (either via stdout or normal file). The output is then checked using
15//! a `Checker`, a program that computes the score of the testcase given the input file, the output
16//! file and the _correct_ output file (the one produced by the jury).
17

56/// In IOI tasks the subtask numbers are non-negative 0-based integers.
57pub type SubtaskId = u32;
58/// In IOI tasks the testcase numbers are non-negative 0-based integers.

108#[derive(Debug, Clone, Serialize, Deserialize, TypeScriptify)]

117    pub title: String,
118    /// The time limit for the execution of the solutions, if `None` it's unlimited.
119    pub time_limit: Option<f64>,
120    /// The memory limit in MiB of the execution of the solution, if `None` it's unlimited.
121    pub memory_limit: Option<u64>,
122    /// The input file for the solutions, usually `Some("input.txt")` or `None` (stdin).
123    pub infile: Option<PathBuf>,
124    /// The output file for the solutions, usually `Some("output.txt")` or `None` (stdout).

133    /// The aggregator to use to compute the score of the subtask based on the score of the

136    /// The number of decimal digits when displaying the scores.
137    #[serde(default)]
138    pub score_precision: usize,
139    /// The graders registered for this task.
140    pub grader_map: Arc<GraderMap>,
141    /// The booklets to compile for this task.
142    pub booklets: Vec<Booklet>,
143    /// An integer that defines the difficulty of the task. Used only in booklet compilations.
144    pub difficulty: Option<u8>,
145    /// An integer that defines the level inside a _syllabus_ (for example for the Olympiads in
146    /// Teams). Used only in booklet compilations.
147    pub syllabus_level: Option<u8>,
148    /// The sanity checks attached to this task. Wrapped in Arc since `SanityChecks` is not Clone.
149    /// It's also not `Serialize` nor `Deserialize`, all the sanity checks will be lost on

153}

263    /// Add the executions required for evaluating this task to the execution DAG.
264    pub fn build_dag(
265        &mut self,
266        eval: &mut EvaluationData,
267        config: &EvaluationConfig,
268    ) -> Result<(), Error> {
269        eval.sender.send(UIMessage::IOITask {
270            task: Box::new(self.clone()),
271        })?;
272        eval.solutions = config.find_solutions(
273            &self.path,
274            vec!["sol/*"],
275            Some(self.grader_map.clone()),
276            eval,
277        );
278
279        let solutions: Vec<_> = eval
280            .solutions
281            .clone()
282            .into_iter()
283            .map(|source| {
284                let path = source.source_file.path.clone();
285                Ok((
286                    source,
287                    Arc::new(Mutex::new(ScoreManager::new(
288                        self,
289                        path,
290                        eval.sender.clone(),
291                    )?)),
292                ))
293            })
294            .collect::<Result<_, Error>>()?;
295
296        let solution_info = solutions
297            .iter()
298            .map(|(solution, _)| SolutionInfo::from(solution))
299            .collect_vec();
300        eval.sender.send(UIMessage::Solutions {
301            solutions: solution_info,
302        })?;
303
304        self.task_type
305            .prepare_dag(eval)
306            .context("Failed to prepare DAG")?;
307
308        let mut generated_io: HashMap<_, _> = HashMap::new();
309
310        for subtask in self.subtasks.values() {
311            trace!("Executing the generation of subtask {}", subtask.id);
312
313            for &testcase_id in subtask.testcases_owned.iter() {
314                trace!(
315                    "Executing the generation of testcase {} of subtask {}",
316                    testcase_id,
317                    subtask.id
318                );
319
320                let testcase = self
321                    .testcases
322                    .get(&testcase_id)
323                    .expect("Testcase not found in the task");
324                let input = testcase
325                    .input_generator
326                    .generate_and_bind(eval, &self.path, subtask.id, testcase.id)
327                    .context("Failed to bind input generator")?;
328                let val_handle = subtask
329                    .input_validator
330                    .validate_and_bind(
331                        eval,
332                        &self.path,
333                        subtask.id,
334                        subtask.name.as_deref(),
335                        testcase.id,
336                        input,
337                    )
338                    .context("Failed to bind validator")?;
339                let output = testcase
340                    .output_generator
341                    .generate_and_bind(self, eval, subtask.id, testcase.id, input, val_handle)
342                    .context("Failed to bind output generator")?;
343                // Store the generated input and output files for setting them into the task
344                // outside the loop.
345                generated_io.insert(testcase.id, (input, output));
346
347                for (solution, score_manager) in solutions.iter() {
348                    trace!(
349                        "Evaluation of the solution {:?} against subtask {} / testcase {}",
350                        solution.source_file.name(),
351                        subtask.id,
352                        testcase.id
353                    );
354
355                    self.task_type
356                        .evaluate(
357                            self,
358                            eval,
359                            subtask.id,
360                            testcase.id,
361                            &solution.source_file,
362                            input,
363                            val_handle,
364                            output,
365                            score_manager.clone(),
366                        )
367                        .context("Failed to bind evaluation")?;
368                }
369            }
370        }
371        // Store inside the task the FileUuid of the input and official output files. This cannot
372        // be done while generating because task cannot be borrowed mutably in the loop.
373        for (testcase_id, (input, output)) in generated_io {
374            let testcase = self.testcases.get_mut(&testcase_id).unwrap();
375            testcase.input_file = Some(input);
376            testcase.official_output_file = output;
377        }
378
379        for subtask in self.subtasks.values() {
380            for &testcase_id in subtask.testcases.iter() {
381                if !subtask.testcases_owned.contains(&testcase_id) {
382                    let testcase = self
383                        .testcases
384                        .get(&testcase_id)
385                        .expect("Testcase not found in the task");
386
387                    let _val_handle = subtask
388                        .input_validator
389                        .validate_and_bind(
390                            eval,
391                            &self.path,
392                            subtask.id,
393                            subtask.name.as_deref(),
394                            testcase.id,
395                            testcase.input_file.unwrap(),
396                        )
397                        .context("Failed to bind validator")?;
398                }
399            }
400        }
401        for booklet in self.booklets.iter() {
402            booklet
403                .build(eval)
404                .context("Failed to bind booklet compilation")?;
405        }
406        self.sanity_checks
407            .pre_hook(self, eval)
408            .context("Sanity check pre-hooks failed")?;
409        Ok(())
410    }

418    /// Clean the task folder removing the files that can be generated automatically.
419    pub fn clean(&self) -> Result<(), Error> {
420        for dir in &["input", "output"] {
421            let dir = self.path.join(dir);
422            if !dir.exists() {
423                continue;
424            }
425            for file in glob::glob(dir.join("*.txt").to_string_lossy().as_ref())
426                .context("Invalid glob pattern")?
427            {
428                let file = match file {
429                    Ok(file) => file,
430                    Err(e) => {
431                        warn!("Glob error: {:?}", e);
432                        continue;
433                    }
434                };
435                // check if the file is used by a static generator
436                if self.testcases.values().any(|tc| {
437                    match (&tc.input_generator, &tc.output_generator) {
438                        (InputGenerator::StaticFile(path), _)
439                        | (_, OutputGenerator::StaticFile(path)) => path == &file,
440                        _ => false,
441                    }
442                }) {
443                    continue;
444                }
445                info!("Removing {}", file.display());
446                std::fs::remove_file(&file)
447                    .with_context(|| format!("Failed to clean file {}", file.display()))?;
448            }
449            info!("Removing {}", dir.display());
450            if let Err(e) = std::fs::remove_dir(&dir) {
451                // FIXME: this should be `e.kind() == ErrorKind::DirectoryNotEmpty`, but it is not
452                //        stable yet.
453                if e.to_string().contains("Directory not empty") {
454                    warn!("Directory {} not empty!", dir.display());
455                } else {
456                    Err(e).with_context(|| format!("Cannot remove {}", dir.display()))?;
457                }
458            }
459        }
460        // remove the bin/ folder
461        let bin_path = self.path.join("bin");
462        if bin_path.exists() {
463            info!("Removing {}", bin_path.display());
464            std::fs::remove_dir_all(&bin_path).with_context(|| {
465                format!("Failed to remove bin/ directory at {}", bin_path.display())
466            })?;
467        }
468        // remove the compiled checkers
469        if let TaskType::Batch(data) = &self.task_type {
470            if let Checker::Custom(_) = data.checker {
471                for checker in &["check/checker", "cor/correttore"] {
472                    let path = self.path.join(checker);
473                    if path.exists() {
474                        info!("Removing {}", path.display());
475                        std::fs::remove_file(&path).with_context(|| {
476                            format!("Failed to remove compiled checker at {}", path.display())
477                        })?;
478                    }
479                }
480            }
481        }
482        // remove the gen/GEN if there is cases.gen
483        let gen_gen_path = self.path.join("gen/GEN");
484        let cases_gen_path = self.path.join("gen/cases.gen");
485        if cases_gen_path.exists() && gen_gen_path.exists() {
486            if is_tm_deletable(&gen_gen_path)? {
487                info!("Removing {}", gen_gen_path.display());
488                std::fs::remove_file(&gen_gen_path).with_context(|| {
489                    format!("Failed to remove gen/GEN at {}", gen_gen_path.display())
490                })?;
491            } else {
492                warn!(
493                    "Won't remove gen/GEN since it doesn't contain {}",
494                    TM_ALLOW_DELETE_COOKIE
495                );
496            }
497        }
498        // remove task.yaml if there is task.yaml.orig
499        let task_yaml_path = self.path.join("task.yaml");
500        let task_yaml_orig_path = self.path.join("task.yaml.orig");
501        if task_yaml_orig_path.exists() && task_yaml_path.exists() {
502            if is_tm_deletable(&task_yaml_path)? {
503                info!("Removing {}", task_yaml_path.display());
504                std::fs::remove_file(&task_yaml_path).with_context(|| {
505                    format!("Failed to remove task.yaml at {}", task_yaml_path.display())
506                })?;
507            } else {
508                warn!(
509                    "Won't remove task.yaml since it doesn't contain {}",
510                    TM_ALLOW_DELETE_COOKIE
511                );
512            }
513        }
514        Ok(())
515    }

524    /// Find the list of all the subtasks that match the given pattern.
525    fn find_subtasks_by_pattern_name(&self, pattern: impl AsRef<str>) -> Vec<&SubtaskInfo> {
526        // Normalize the pattern; the subtask names are already normalized.
527        let pattern = pattern.as_ref().nfkc().collect::<String>();
528        let pattern = WildMatch::new(&pattern);
529        let mut result = vec![];
530        for subtask in self.subtasks.values() {
531            if subtask.name_matches(&pattern) {
532                result.push(subtask);
533            }
534        }
535        result
536    }
537}

567impl ScoreManager {
568    /// Make a new `ScoreManager` based on the subtasks and testcases of the specified task.
569    pub fn new(
570        task: &IOITask,
571        solution: PathBuf,
572        sender: Arc<Mutex<UIMessageSender>>,
573    ) -> Result<ScoreManager, Error> {
574        let mut ret = ScoreManager {
575            solution,
576            subtask_scores: task.subtasks.keys().map(|st_num| (*st_num, None)).collect(),
577            max_subtask_scores: task
578                .subtasks
579                .values()
580                .map(|st| (st.id, st.max_score))
581                .collect(),
582            testcase_scores: task.testcases.keys().map(|tc| (*tc, None)).collect(),
583            subtask_testcases: task
584                .subtasks
585                .iter()
586                .map(|(st_num, st)| (*st_num, st.testcases.clone()))
587                .collect(),
588            aggregator: task.testcase_score_aggregator,
589        };
590
591        for (st_num, st) in &task.subtasks {
592            if st.testcases.is_empty() {
593                ret.score_subtask(*st_num, sender.clone())?;
594            }
595        }
596        Ok(ret)
597    }
598
599    /// Store the score of the testcase and eventually compute the score of the subtask and of the
600    /// task.
601    pub fn score(
602        &mut self,
603        subtask_id: SubtaskId,
604        testcase_id: TestcaseId,
605        score: f64,
606        message: String,
607        sender: Arc<Mutex<UIMessageSender>>,
608    ) -> Result<(), Error> {
609        self.testcase_scores.insert(testcase_id, Some(score));
610        sender.send(UIMessage::IOITestcaseScore {
611            subtask: subtask_id,
612            testcase: testcase_id,
613            solution: self.solution.clone(),
614            score,
615            message,
616        })?;
617
618        for (subtask_id, subtask) in self
619            .subtask_scores
620            .iter()
621            .map(|(&a, &b)| (a, b))
622            .collect_vec()
623        {
624            if subtask.is_none() {
625                self.score_subtask(subtask_id, sender.clone())?;
626            }
627        }
628        Ok(())
629    }
630
631    fn score_subtask(
632        &mut self,
633        subtask_id: SubtaskId,
634        sender: Arc<Mutex<UIMessageSender>>,
635    ) -> Result<(), Error> {
636        if self.subtask_testcases[&subtask_id]
637            .iter()
638            .all(|tc| self.testcase_scores[tc].is_some())
639        {
640            let normalized_score = self.aggregator.aggregate(
641                self.subtask_testcases[&subtask_id]
642                    .iter()
643                    .map(|tc| self.testcase_scores[tc].unwrap()),
644            );
645            let subtask_score = self.max_subtask_scores[&subtask_id] * normalized_score;
646            self.subtask_scores.insert(subtask_id, Some(subtask_score));
647            sender.send(UIMessage::IOISubtaskScore {
648                subtask: subtask_id,
649                solution: self.solution.clone(),
650                score: subtask_score,
651                normalized_score,
652            })?;
653            if self.subtask_scores.values().all(Option::is_some) {
654                let task_score: f64 = self
655                    .subtask_scores
656                    .values()
657                    .map(|score| score.unwrap())
658                    .sum();
659                sender.send(UIMessage::IOITaskScore {
660                    solution: self.solution.clone(),
661                    score: task_score,
662                })?;
663            }
664        }
665        Ok(())
666    }
