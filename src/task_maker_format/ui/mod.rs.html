37lazy_static! {
38    /// Whether the terminal supports ANSI 256 colors.
39    static ref HAS_ANSI256: bool = {
40        if std::env::var("TM_ANSI256").as_deref() == Ok("true") {
41            if let Some(support) = supports_color::on(supports_color::Stream::Stdout) {
42                support.has_256
43            } else {
44                false
45            }
46        } else {
47            false
48        }
49    };
50    /// Whetner the terminal supports 24-bit Truecolor.
51    static ref HAS_TRUECOLOR: bool = {
52        if let Some(support) = supports_color::on(supports_color::Stream::Stdout) {
53            support.has_16m
54        } else {
55            false
56        }
57    };
58}

60macro_rules! define_color_inner {
61    ($color:expr,) => {};
62    ($color:expr, ansi($ansi:expr), $($tt:tt)*) => {
63        if *HAS_ANSI256 {
64            $color.set_fg(Some(Color::Ansi256($ansi)));
65        }
66        define_color_inner!($color, $($tt)*)
67    };
68    ($color:expr, rgb($r:expr, $g:expr, $b:expr), $($tt:tt)*) => {
69        if *HAS_TRUECOLOR {
70            $color.set_fg(Some(Color::Rgb($r, $g, $b)));
71        }
72        define_color_inner!($color, $($tt)*)
73    };
74    ($color:expr, basic($basic:ident), $($tt:tt)*) => {
75        $color.set_fg(Some(Color::$basic));
76        define_color_inner!($color, $($tt)*)
77    };
78    ($color:expr, intense, $($tt:tt)*) => {
79        $color.set_intense(true);
80        define_color_inner!($color, $($tt)*)
81    };
82    ($color:expr, bold, $($tt:tt)*) => {
83        $color.set_bold(true);
84        define_color_inner!($color, $($tt)*)
85    };
86}
87macro_rules! define_color {
88    ($($tt:tt)*) => {{
89        let mut color = ColorSpec::new();
90        define_color_inner!(color, $($tt)*,);
91        color
92    }};
93}

112/// The status of an execution.
113#[derive(Debug, Serialize, Deserialize, Clone, PartialEq, TypeScriptify)]
114pub enum UIExecutionStatus {
115    /// The `Execution` is known to the DAG and when all its dependencies are ready it will
116    /// started.
117    Pending,
118    /// The `Execution` has been started on a worker.
119    Started {
120        /// The UUID of the worker.
121        worker: WorkerUuid,
122    },
123    /// The `Execution` has been completed.
124    Done {
125        /// The result of the execution.
126        result: ExecutionResult,
127    },
128    /// At least one of its dependencies have failed, the `Execution` has been skipped.
129    Skipped,
130}

132/// The status of the compilation of a file.
133#[derive(Debug, Clone, PartialEq)]
134pub enum CompilationStatus {
135    /// The compilation is known but it has not started yet.
136    Pending,
137    /// The compilation is running on a worker.
138    Running,
139    /// The compilation has completed.
140    Done {
141        /// The result of the compilation.
142        result: ExecutionResult,
143        /// The standard output of the compilation.
144        stdout: Option<String>,
145        /// The standard error of the compilation.
146        stderr: Option<String>,
147    },
148    /// The compilation has failed.
149    Failed {
150        /// The result of the compilation.
151        result: ExecutionResult,
152        /// The standard output of the compilation.
153        stdout: Option<String>,
154        /// The standard error of the compilation.
155        stderr: Option<String>,
156    },
157    /// The compilation has been skipped.
158    Skipped,
159}
160
161impl CompilationStatus {
162    /// Apply to this `CompilationStatus` a new `UIExecutionStatus`.
163    pub fn apply_status(&mut self, status: UIExecutionStatus) {
164        match status {
165            UIExecutionStatus::Pending => *self = CompilationStatus::Pending,
166            UIExecutionStatus::Started { .. } => *self = CompilationStatus::Running,
167            UIExecutionStatus::Done { result } => {
168                let stdout = result
169                    .stdout
170                    .as_ref()
171                    .map(|s| String::from_utf8_lossy(s).into());
172                let stderr = result
173                    .stderr
174                    .as_ref()
175                    .map(|s| String::from_utf8_lossy(s).into());
176                if let ExecutionStatus::Success = result.status {
177                    *self = CompilationStatus::Done {
178                        result,
179                        stdout,
180                        stderr,
181                    };
182                } else {
183                    *self = CompilationStatus::Failed {
184                        result,
185                        stdout,
186                        stderr,
187                    };
188                }
189            }
190            UIExecutionStatus::Skipped => *self = CompilationStatus::Skipped,
191        }
192    }
193}

212/// Collection of utilities for drawing the finish UI.
213pub struct FinishUIUtils<'a> {
214    /// Stream where to print to.
215    stream: &'a mut StandardStream,
216}
217
218impl<'a> FinishUIUtils<'a> {
219    /// Make a new `FinishUIUtils` borrowing a `StandardStream`.
220    pub fn new(stream: &'a mut StandardStream) -> FinishUIUtils<'a> {
221        FinishUIUtils { stream }
222    }
223
224    /// Print all the compilation statuses.
225    pub fn print_compilations(&mut self, compilations: &HashMap<PathBuf, CompilationStatus>) {
226        cwriteln!(self, BLUE, "Compilations");
227        let max_len = compilations
228            .keys()
229            .map(|p| p.file_name().expect("Invalid file name").len())
230            .max()
231            .unwrap_or(0);
232        for (path, status) in compilations {
233            print!(
234                "{:width$}  ",
235                path.file_name()
236                    .expect("Invalid file name")
237                    .to_string_lossy(),
238                width = max_len
239            );
240            match status {
241                CompilationStatus::Done { result, .. } => {
242                    cwrite!(self, GREEN, " OK  ");
243                    FinishUIUtils::print_time_memory(&result.resources);
244                }
245                CompilationStatus::Failed {
246                    result,
247                    stdout,
248                    stderr,
249                } => {
250                    cwrite!(self, RED, "FAIL ");
251                    FinishUIUtils::print_time_memory(&result.resources);
252                    if let Some(stdout) = stdout {
253                        if !stdout.trim().is_empty() {
254                            println!();
255                            cwriteln!(self, BOLD, "stdout:");
256                            println!("{}", stdout.trim());
257                        }
258                    }
259                    if let Some(stderr) = stderr {
260                        if !stderr.trim().is_empty() {
261                            println!();
262                            cwriteln!(self, BOLD, "stderr:");
263                            println!("{}", stderr.trim());
264                        }
265                    }
266                }
267                _ => {
268                    cwrite!(self, YELLOW, "{:?}", status);
269                }
270            }
271            println!();
272        }
273    }
274
275    /// Print the time and memory usage of an execution.
276    pub fn print_time_memory(resources: &ExecutionResourcesUsage) {
277        print!(
278            "{:2.3}s | {:3.1}MiB",
279            resources.cpu_time,
280            (resources.memory as f64) / 1024.0
281        );
282    }
283
284    /// Print a message for the non-successful variants of the provided status.
285    pub fn print_fail_execution_status(status: &ExecutionStatus) {
286        match status {
287            ExecutionStatus::Success => {}
288            ExecutionStatus::ReturnCode(code) => print!("Exited with {}", code),
289            ExecutionStatus::Signal(sig, name) => print!("Signal {} ({})", sig, name),
290            ExecutionStatus::TimeLimitExceeded => print!("Time limit exceeded"),
291            ExecutionStatus::SysTimeLimitExceeded => print!("Kernel time limit exceeded"),
292            ExecutionStatus::WallTimeLimitExceeded => print!("Wall time limit exceeded"),
293            ExecutionStatus::MemoryLimitExceeded => print!("Memory limit exceeded"),
294            ExecutionStatus::InternalError(err) => print!("Internal error: {}", err),
295        }
296    }
297
298    /// Find the maximum length of the solutions name from the keys of the given structure.
299    pub fn get_max_len<T>(solutions: &HashMap<PathBuf, T>) -> usize {
300        solutions
301            .keys()
302            .map(|p| p.file_name().expect("Invalid file name").len())
303            .max()
304            .unwrap_or(0)
305    }
306
307    /// Print the diagnostics.
308    pub fn print_diagnostic_messages(&mut self, diagnostics: &DiagnosticContext) {
309        let diagnostics = diagnostics.diagnostics();
310        if !diagnostics.is_empty() {
311            for diagnostic in diagnostics.iter().sorted_by_key(|d| d.level()) {
312                println!("{}", diagnostic);
313            }
314        }
315    }

318/// The sender of the UIMessage
319#[derive(Debug)]
320pub struct UIMessageSender {
321    sender: UIChannelSender,
322}
323
324impl UIMessageSender {
325    /// Make a new pair of UIMessageSender and ChannelReceiver.
326    pub fn new() -> (UIMessageSender, UIChannelReceiver) {
327        let (sender, receiver) = channel();
328        (UIMessageSender { sender }, receiver)
329    }
330
331    /// Send a message to the channel.
332    pub fn send(&self, message: UIMessage) -> Result<(), Error> {
333        self.sender.send(message).map_err(|e| e.into())
334    }

345/// The type of the UI to use, it enumerates all the known UI interfaces.
346#[derive(Debug, Clone)]
347pub enum UIType {
348    /// The `PrintUI`.
349    Print,
350    /// The `RawUI`.
351    Raw,
352    /// The `CursesUI`.
353    Curses,
354    /// The `JsonUI`.
355    Json,
356    /// The `SilentUI`.
357    Silent,
358}
359
360impl std::str::FromStr for UIType {
361    type Err = String;
362
363    fn from_str(s: &str) -> Result<UIType, Self::Err> {
364        match s.to_ascii_lowercase().as_str() {
365            "print" => Ok(UIType::Print),
366            "raw" => Ok(UIType::Raw),
367            "curses" => Ok(UIType::Curses),
368            "json" => Ok(UIType::Json),
369            "silent" => Ok(UIType::Silent),
370            _ => Err(format!("Unknown ui: {}", s)),
371        }
372    }
373}

390/// Write to `$self.stream`, in the color specified as second parameter. The arguments that follow
391/// will be passed to `write!`.
392///
393/// ```
394/// #[macro_use]
395/// extern crate task_maker_format;
396///
397/// use termcolor::{ColorSpec, ColorChoice};
398/// use task_maker_format::ui::StdoutPrinter;
399/// use task_maker_format::cwrite;
400///
401/// # fn main() {
402/// let mut color = ColorSpec::new();
403/// color.set_bold(true);
404///
405/// let mut printer = StdoutPrinter::default();
406/// cwrite!(printer, color, "The output is {}", 42);
407/// # }
408/// ```
409#[macro_export]
410macro_rules! cwrite {
411    ($self:expr, $color:expr, $($arg:tt)*) => {{
412        use std::io::Write;
413        use $crate::ui::WriteColor;
414        $self.stream.set_color(&$color).unwrap();
415        write!(&mut $self.stream, $($arg)*).unwrap();
416        $self.stream.reset().unwrap();
417    }};
418}
419
420/// Write to `$self.stream`, in the color specified as second parameter. The arguments that follow
421/// will be passed to `writeln!`.
422///
423/// ```
424/// #[macro_use]
425/// extern crate task_maker_format;
426/// use termcolor::{ColorSpec, ColorChoice};
427/// use task_maker_format::ui::StdoutPrinter;
428/// use task_maker_format::cwrite;
429///
430/// # fn main() {
431/// let mut color = ColorSpec::new();
432/// color.set_bold(true);
433///
434/// let mut printer = StdoutPrinter::default();
435/// cwriteln!(printer, color, "The output is {}", 42);
436/// # }
437/// ```
438#[macro_export]
439macro_rules! cwriteln {
440    ($self:expr, $color:expr, $($arg:tt)*) => {{
441        use std::io::Write;
442        use $crate::ui::WriteColor;
443        $self.stream.set_color(&$color).unwrap();
444        writeln!(&mut $self.stream, $($arg)*).unwrap();
445        $self.stream.reset().unwrap();
446    }};
