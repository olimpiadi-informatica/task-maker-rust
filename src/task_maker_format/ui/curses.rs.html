35macro_rules! define_color_inner {
36    ($color:expr,) => {
37        $color
38    };
39    ($color:expr, basic($basic:ident), $($tt:tt)*) => {
40        define_color_inner!($color.fg(Color::$basic), $($tt)*)
41    };
42    ($color:expr, rgb($r:expr, $g:expr, $b:expr), $($tt:tt)*) => {
43        define_color_inner!(if *$crate::ui::HAS_TRUECOLOR {
44            $color.fg(Color::Rgb($r, $g, $b))
45        } else {
46            $color
47        }, $($tt)*)
48    };
49    ($color:expr, bold, $($tt:tt)*) => {
50        define_color_inner!($color.add_modifier(Modifier::BOLD), $($tt)*)
51    };

74/// A generic animated UI for tasks, dynamically refreshing using curses as a backend.
75///
76/// - `State` is the type of `UIState` for this UI.
77/// - `Drawer` is the drawer for the frames of the UI.
78/// - `FinishUI` is the UI that prints the final results.
79pub struct CursesUI<State, Drawer, Finish>
80where
81    State: UIStateT + Send + Sync + 'static,
82    Drawer: CursesDrawer<State> + Send + Sync + 'static,
83    Finish: FinishUI<State> + Send + Sync + 'static,
84{
85    /// The thread where the UI lives.
86    ui_thread: Option<JoinHandle<()>>,
87    /// The state of the task for the UI.
88    state: Arc<RwLock<State>>,
89    /// When it becomes true the UI will stop.
90    stop: Arc<AtomicBool>,
91
92    drawer: PhantomData<Drawer>,
93    finish_ui: PhantomData<Finish>,
94}
95
96/// A drawer for the frames of the UI.
97pub trait CursesDrawer<State> {
98    /// Draw a frame of the UI using the provided state, onto the frame, using the loading
99    /// character. Frame index is a counter of the number of frames encountered so far.
100    fn draw(state: &State, frame: &mut Frame, loading: char, frame_index: usize);
101}
102
103impl<State, Drawer, Finish> CursesUI<State, Drawer, Finish>
104where
105    State: UIStateT + Send + Sync + 'static,
106    Drawer: CursesDrawer<State> + Send + Sync + 'static,
107    Finish: FinishUI<State> + Send + Sync + 'static,
108{
109    /// Make a new generic `CursesUI`.
110    pub fn new(state: State) -> Result<CursesUI<State, Drawer, Finish>, Error> {
111        let state = Arc::new(RwLock::new(state));
112        let stop = Arc::new(AtomicBool::new(false));
113        let mut ui = CursesUI {
114            ui_thread: None,
115            state: state.clone(),
116            stop: stop.clone(),
117            drawer: Default::default(),
118            finish_ui: Default::default(),
119        };
120        let handle = ui.start(state, stop)?;
121        ui.ui_thread = Some(handle);
122        Ok(ui)
123    }
124
125    /// Start the drawing thread of the UI, returning the `JoinHandle` of it.
126    fn start(
127        &mut self,
128        state: Arc<RwLock<State>>,
129        stop: Arc<AtomicBool>,
130    ) -> Result<JoinHandle<()>, Error> {
131        let stdout = io::stdout().into_raw_mode()?;
132        let stdout = MouseTerminal::from(stdout)
133            .into_alternate_screen()
134            .context("Failed to enter alternate screen mode, the terminal may not support it")?;
135        let backend = TermionBackend::new(stdout);
136        let mut terminal = Terminal::new(backend)?;
137        terminal.hide_cursor()?;
138        Ok(std::thread::Builder::new()
139            .name("CursesUI thread".to_owned())
140            .spawn(move || {
141                let loading = ['◐', '◓', '◑', '◒'];
142                let mut loading_index = 0;
143                let stdin = termion::async_stdin();
144                let mut events = stdin.events();
145                while !stop.load(Ordering::Relaxed) {
146                    // FIXME: handling the ^C this way inhibits the real ^C handler. Doing so the workers may
147                    //        not be killed properly (locally and remotely).
148                    if let Some(Ok(Event::Key(Key::Ctrl('c') | Key::Ctrl('\\')))) = events.next() {
149                        drop(terminal);
150                        send_ctrl_c();
151                        return;
152                    }
153                    let loading = loading[loading_index % loading.len()];
154                    terminal
155                        .draw(|f| {
156                            let state = state.read().expect("UI state lock is poisoned");
157                            Drawer::draw(&state, f, loading, loading_index);
158                        })
159                        .expect("Failed to draw to the screen");
160                    // reduce the framerate to at most `FPS`
161                    std::thread::sleep(std::time::Duration::from_micros(1_000_000 / FPS));
162                    loading_index += 1;
163                }
164            })?)
165    }
166}

168impl<State, Drawer, Finish> UI for CursesUI<State, Drawer, Finish>
169where
170    State: UIStateT + Send + Sync + 'static,
171    Drawer: CursesDrawer<State> + Send + Sync + 'static,
172    Finish: FinishUI<State> + Send + Sync + 'static,
173{
174    fn on_message(&mut self, message: UIMessage) {
175        self.state
176            .write()
177            .expect("UI state lock is poisoned")
178            .apply(message);
179    }
180
181    fn finish(&mut self) {
182        self.stop.store(true, Ordering::Relaxed);
183        self.ui_thread
184            .take()
185            .expect("UI finished more than once")
186            .join()
187            .expect("UI thread failed");
188        // at this point the terminal should be restored
189        let state = self.state.read().expect("State lock is poisoned");
190        Finish::print(&state);
191    }
192}

194impl<State, Drawer, Finish> Drop for CursesUI<State, Drawer, Finish>
195where
196    State: UIStateT + Send + Sync + 'static,
197    Drawer: CursesDrawer<State> + Send + Sync + 'static,
198    Finish: FinishUI<State> + Send + Sync + 'static,
199{
200    fn drop(&mut self) {
201        // tell the ui to stop and wait for it, the terminal will be released.
202        self.stop.store(true, Ordering::Relaxed);
203        if self.ui_thread.is_some() {
204            // try not to panic during unwind
205            let _ = self.ui_thread.take().unwrap().join();
206        }
207    }
208}

210/// Get the rect of the inner rect of a block with the borders.
211pub fn inner_block(rect: Rect) -> Rect {
212    if rect.width < 2 || rect.height < 2 {
213        return Rect::new(rect.x + 1, rect.y + 1, 0, 0);
214    }
215    Rect::new(rect.x + 1, rect.y + 1, rect.width - 2, rect.height - 2)
216}

218/// Draw the compilation block.
219pub(crate) fn draw_compilations<'a, I>(
220    frame: &mut Frame,
221    rect: Rect,
222    compilations: I,
223    loading: char,
224) where
225    I: Iterator<Item = (&'a PathBuf, &'a CompilationStatus)>,
226{
227    let compilations: Vec<_> = compilations.collect();
228    let max_len = compilations
229        .iter()
230        .map(|(k, _)| k.file_name().expect("Invalid file name").len())
231        .max()
232        .unwrap_or(0)
233        + 4;
234    let text: Vec<Line> = compilations
235        .iter()
236        .sorted_by_key(|(k, _)| *k)
237        .map(|(file, status)| {
238            vec![
239                Span::raw(format!(
240                    "{:<max_len$}",
241                    file.file_name()
242                        .expect("Invalid file name")
243                        .to_string_lossy(),
244                    max_len = max_len
245                )),
246                compilation_status_text(status, loading),
247            ]
248            .into()
249        })
250        .collect();
251    let paragraph = Paragraph::new(text);
252    frame.render_widget(paragraph, rect);
253}

255/// Get the `Text` relative to the compilation status of a file.
256pub(crate) fn compilation_status_text(status: &CompilationStatus, loading: char) -> Span<'static> {
257    match status {
258        CompilationStatus::Pending => Span::raw("... "),
259        CompilationStatus::Running => Span::raw(format!("{}   ", loading)),
260        CompilationStatus::Done { .. } => Span::styled("OK  ", *GREEN),
261        CompilationStatus::Failed { .. } => Span::styled("FAIL", *RED),
262        CompilationStatus::Skipped => Span::styled("skip", *YELLOW),
263    }
264}

274/// Draw the server status block.
275pub fn render_server_status(
276    frame: &mut Frame,
277    rect: Rect,
278    status: Option<&ExecutorStatus<SystemTime>>,
279    loading: char,
280    frame_index: usize,
281) {
282    let title = " Server status ";
283    render_block(frame, rect, title);
284    draw_server_status_summary(
285        frame,
286        Rect::new(
287            rect.x + title.len() as u16 + 2,
288            rect.y,
289            rect.width.saturating_sub(title.len() as u16 + 2),
290            1,
291        ),
292        status,
293    );
294    draw_server_status(
295        frame,
296        inner_block(rect),
297        status,
298        loading,
299        frame_index / FPS as usize / ROTATION_DELAY as usize,
300    );
301}

303/// Draw the summary of the server status on the border of the block.
304fn draw_server_status_summary(
305    frame: &mut Frame,
306    rect: Rect,
307    status: Option<&ExecutorStatus<SystemTime>>,
308) {
309    let status = if let Some(status) = status {
310        status
311    } else {
312        return;
313    };
314    let paragraph = Paragraph::new(Line::from(vec![
315        Span::styled(" Ready ", Style::default().add_modifier(Modifier::BOLD)),
316        Span::raw(format!("{} ─", status.ready_execs)),
317        Span::styled(" Waiting ", Style::default().add_modifier(Modifier::BOLD)),
318        Span::raw(format!("{} ", status.waiting_execs)),
319    ]));
320    frame.render_widget(paragraph, rect);
321}

323/// Draw the content of the server status box, splitting the workers in 2 groups if they don't fit,
324/// and rotating them if they still don't fit.
325fn draw_server_status(
326    frame: &mut Frame,
327    rect: Rect,
328    status: Option<&ExecutorStatus<SystemTime>>,
329    loading: char,
330    mut rotation_index: usize,
331) {
332    let status = if let Some(status) = status {
333        status
334    } else {
335        return;
336    };
337    let rects = if status.connected_workers.len() as u16 > rect.height {
338        vec![
339            Rect::new(rect.x, rect.y, rect.width / 2, rect.height),
340            Rect::new(
341                rect.x + rect.width / 2,
342                rect.y,
343                rect.width - rect.width / 2,
344                rect.height,
345            ),
346        ]
347    } else {
348        vec![rect]
349    };
350    let workers: Vec<_> = status
351        .connected_workers
352        .iter()
353        .sorted_by_key(|worker| &worker.name)
354        .collect();
355    // if the workers fit in the screen there is no need to rotate them
356    if rect.height as usize * rects.len() >= workers.len() {
357        rotation_index = 0;
358    }
359    // avoid drawing if the screen is too small
360    if rect.height == 0 {
361        return;
362    }
363    let num_workers = workers.len();
364    let chunks = workers
365        .into_iter()
366        .cycle()
367        .skip(rotation_index)
368        .take(num_workers)
369        .chunks(rect.height as usize);
370    for (rect, chunk) in rects.into_iter().zip(&chunks) {
371        draw_workers_chunk(frame, rect, &chunk.collect_vec(), loading);
372    }
373}

375/// Draw a chunk of workers in the specified rectangle.
376fn draw_workers_chunk(
377    frame: &mut Frame,
378    rect: Rect,
379    workers: &[&ExecutorWorkerStatus<SystemTime>],
380    loading: char,
381) {
382    let max_len = workers
383        .iter()
384        .map(|worker| worker.name.len())
385        .max()
386        .unwrap_or(0);
387    let text: Vec<Line> = workers
388        .iter()
389        .map(|worker| {
390            let worker_name = format!("- {:<max_len$} ", worker.name, max_len = max_len);
391            let worker_name_len = worker_name.chars().count();
392            let mut spans = vec![Span::raw(worker_name)];
393
394            if let Some(job) = &worker.current_job {
395                let duration = job.duration.elapsed().unwrap_or_default().as_secs_f32();
396                let mut line = format!("{} {} ({:.2}s)", loading, job.job, duration);
397                let line_len = line.chars().count();
398                // Line is too long.
399                if worker_name_len + line_len > rect.width as usize {
400                    // We need to remove "extra_len" characters.
401                    let extra_len = worker_name_len + line_len - rect.width as usize;
402                    // There is not enough space for the job name, even with the ellipsis.
403                    if extra_len + 3 > job.job.len() {
404                        line = line.chars().take(rect.width as usize).collect();
405                    } else {
406                        let job_name: String =
407                            line.chars().take(job.job.len() - extra_len - 3).collect();
408                        line = format!("{} {}... ({:.2}s)", loading, job_name, duration);
409                    }
410                }
411                spans.push(Span::raw(line));
412            }
413            spans.into()
414        })
415        .collect();
416    let paragraph = Paragraph::new(text);
417    frame.render_widget(paragraph, rect);
418}

420/// Send to the current process `SIGINT`, letting it exit gracefully.
421fn send_ctrl_c() {
422    let pid = std::process::id();
423    if let Err(e) = signal::kill(Pid::from_raw(pid as i32), Signal::SIGINT) {
424        error!("Failed to send SIGINT to {}: {}", pid, e);
425    }
