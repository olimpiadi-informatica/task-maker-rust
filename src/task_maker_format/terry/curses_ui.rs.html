29/// Draw a frame of interface to the provided `Frame`.
30fn draw_frame(state: &UIState, f: &mut Frame, loading: char, frame_index: usize) {
31    let header: Line = vec![
32        Span::styled(
33            state.task.description.clone(),
34            Style::default().add_modifier(Modifier::BOLD),
35        ),
36        Span::raw(" ("),
37        Span::raw(state.task.name.clone()),
38        Span::raw(")"),
39    ]
40    .into();
41    let header_len = 2;
42    let num_compilations = state
43        .compilations
44        .iter()
45        .filter(|(k, _)| !state.solutions.contains_key(*k))
46        .count();
47    let compilations_len = if num_compilations > 0 {
48        num_compilations as u16 + 2
49    } else {
50        0
51    };
52    let evaluations_len = state.solutions.len() as u16 + 2;
53    let mut workers_len = state
54        .executor_status
55        .as_ref()
56        .map(|s| s.connected_workers.len())
57        .unwrap_or(0) as u16
58        + 2;
59    let total_height = f.area().height;
60    // fixed size section heights
61    let top_height = header_len + compilations_len;
62    // if the sections don't just fit, reduce the size of the workers until they fit but
63    // without shortening it more than 3 lines (aka box + 1 worker).
64    if top_height + evaluations_len + workers_len > total_height {
65        workers_len = std::cmp::max(
66            3,
67            total_height as i16 - top_height as i16 - evaluations_len as i16,
68        ) as u16;
69    }
70    let chunks = Layout::default()
71        .direction(Direction::Vertical)
72        .constraints(
73            [
74                Constraint::Length(header_len),
75                Constraint::Length(compilations_len),
76                Constraint::Min(0),
77                Constraint::Length(workers_len),
78            ]
79            .as_ref(),
80        )
81        .split(f.area());
82    let paragraph = Paragraph::new(header).block(Block::default().borders(Borders::NONE));
83    f.render_widget(paragraph, chunks[0]);
84    if compilations_len > 0 {
85        render_block(f, chunks[1], " Compilations ");
86        draw_compilations(
87            f,
88            inner_block(chunks[1]),
89            state
90                .compilations
91                .iter()
92                .filter(|(p, _)| !state.solutions.contains_key(*p)),
93            loading,
94        );
95    }
96    render_block(f, chunks[2], " Evaluations ");
97    draw_evaluations(f, inner_block(chunks[2]), state, loading);
98    render_server_status(
99        f,
100        chunks[3],
101        state.executor_status.as_ref(),
102        loading,
103        frame_index,
104    );
105}

107/// Draw the evaluations of the solutions.
108fn draw_evaluations(frame: &mut Frame, rect: Rect, state: &UIState, loading: char) {
109    let max_len = FinishUIUtils::get_max_len(&state.solutions);
110    let text: Vec<Line> = state
111        .solutions
112        .keys()
113        .sorted()
114        .map(|path| {
115            let solution_state = &state.solutions[path];
116            let mut spans = vec![Span::raw(format!(
117                "{:<max_len$}  ",
118                path.file_name()
119                    .expect("Invalid file name")
120                    .to_string_lossy(),
121                max_len = max_len
122            ))];
123            if let Some(comp_status) = state.compilations.get(path) {
124                spans.push(compilation_status_text(comp_status, loading));
125            } else {
126                spans.push(Span::raw("    "));
127            }
128            spans.push(Span::raw(" "));
129            spans.push(evaluation_score(
130                state.task.max_score,
131                solution_state,
132                loading,
133            ));
134            spans.push(Span::raw("  "));
135            spans.append(&mut evaluation_line(solution_state));
136            spans.into()
137        })
138        .collect();
139    let paragraph = Paragraph::new(text);
140    frame.render_widget(paragraph, rect);
141}

143/// Return the line with the status of the evaluation of a solution.
144fn evaluation_line<'a>(state: &SolutionState) -> Vec<Span<'a>> {
145    match &state.status {
146        SolutionStatus::Pending => vec![],
147        SolutionStatus::Generating => vec![Span::raw("Generating")],
148        SolutionStatus::Generated => vec![Span::raw("Generated")],
149        SolutionStatus::Validating => vec![Span::raw("Validating")],
150        SolutionStatus::Validated => vec![Span::raw("Validated")],
151        SolutionStatus::Solving => vec![Span::raw("Solving")],
152        SolutionStatus::Solved => vec![Span::raw("Solved")],
153        SolutionStatus::Checking => vec![Span::raw("Checking")],
154        SolutionStatus::Done => evaluation_outcome(state.outcome.as_ref()),
155        SolutionStatus::Failed(e) => vec![Span::raw(format!("Failed: {}", e))],
156        SolutionStatus::Skipped => vec![Span::raw("Skipped")],
157    }
158}

160/// Return the line with the outcome of the evaluation of a solution.
161fn evaluation_outcome<'a>(outcome: Option<&Result<SolutionOutcome, String>>) -> Vec<Span<'a>> {
162    match outcome {
163        Some(Ok(outcome)) => {
164            let mut res = Vec::new();
165            for (val, feed) in outcome
166                .validation
167                .cases
168                .iter()
169                .zip(outcome.feedback.cases.iter())
170            {
171                match val.status {
172                    CaseStatus::Missing => res.push(Span::styled("m ", *YELLOW)),
173                    CaseStatus::Parsed => {
174                        if feed.correct {
175                            res.push(Span::styled("c ", *GREEN))
176                        } else {
177                            res.push(Span::styled("w ", *RED))
178                        }
179                    }
180                    CaseStatus::Invalid => res.push(Span::styled("i ", *RED)),
181                }
182            }
183            res
184        }
185        Some(Err(e)) => vec![Span::raw(format!("Checker failed: {}", e))],
186        None => vec![Span::raw("unknown")],
187    }
188}

190/// Return the score column of a solution.
191fn evaluation_score<'a>(max_score: f64, state: &SolutionState, loading: char) -> Span<'a> {
192    match state.status {
193        SolutionStatus::Pending => Span::raw("..."),
194        SolutionStatus::Generating
195        | SolutionStatus::Generated
196        | SolutionStatus::Validating
197        | SolutionStatus::Validated
198        | SolutionStatus::Solving
199        | SolutionStatus::Solved
200        | SolutionStatus::Checking => Span::raw(format!(" {} ", loading)),
201        SolutionStatus::Done => {
202            if let Some(Ok(outcome)) = &state.outcome {
203                let score = format!("{:>3.0}", outcome.score * max_score);
204                if abs_diff_eq!(outcome.score, 0.0) {
205                    Span::styled(score, *RED)
206                } else if abs_diff_eq!(outcome.score, 1.0) {
207                    Span::styled(score, *GREEN)
208                } else {
209                    Span::styled(score, *YELLOW)
210                }
211            } else {
212                Span::raw(format!(" {} ", loading))
213            }
214        }
215        SolutionStatus::Failed(_) | SolutionStatus::Skipped => Span::raw(" X "),
216    }
