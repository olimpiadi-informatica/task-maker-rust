13/// The state of a Terry task, all the information for the UI are stored here.
14#[derive(Debug, Clone)]
15pub struct UIState {
16    /// The task.
17    pub task: TerryTask,
18    /// The status of the compilations.
19    pub compilations: HashMap<PathBuf, CompilationStatus>,
20    /// The state of the solutions known.
21    pub solutions: HashMap<PathBuf, SolutionState>,
22    /// The status of the executor.
23    pub executor_status: Option<ExecutorStatus<SystemTime>>,
24    /// Diagnostics context.
25    pub diagnostics: DiagnosticContext,
26}

28/// The state of the evaluation of a solution.
29#[derive(Debug, Clone)]
30pub struct SolutionState {
31    /// The status of the evaluation.
32    pub status: SolutionStatus,
33    /// The checker's outcome or its error.
34    pub outcome: Option<Result<SolutionOutcome, String>>,
35    /// The seed used for the generation.
36    pub seed: Option<Seed>,
37    /// The result of the generator.
38    pub generator_result: Option<ExecutionResult>,
39    /// The result of the validator.
40    pub validator_result: Option<ExecutionResult>,
41    /// The result of the solution.
42    pub solution_result: Option<ExecutionResult>,
43    /// The result of the checker.
44    pub checker_result: Option<ExecutionResult>,
45}
46
47/// The status of the evaluation of a solution.
48#[derive(Debug, Clone, Default)]
49pub enum SolutionStatus {
50    /// The solution has not started yet.
51    #[default]
52    Pending,
53    /// The generator is preparing the input file.
54    Generating,
55    /// The generator prepared the input file.
56    Generated,
57    /// The validator is running.
58    Validating,
59    /// The validator checked the input file.
60    Validated,
61    /// The solution is running.
62    Solving,
63    /// The solution produced the output file.
64    Solved,
65    /// The checker is running.
66    Checking,
67    /// The checker completed successfully.
68    Done,
69    /// Something failed during the evaluation.
70    Failed(String),
71    /// The evaluation has been skipped.
72    Skipped,
73}

102impl UIStateT for UIState {
103    /// Apply a `UIMessage` to this state.
104    fn apply(&mut self, message: UIMessage) {
105        macro_rules! process_step {
106            ($self:expr, $solution:expr, $status:expr, $step_result:tt, $start_status:tt, $ok_status:tt, $name:literal) => {{
107                let sol = $self
108                    .solutions
109                    .get_mut(&$solution)
110                    .expect("Outcome of an unknown solution");
111                match $status {
112                    UIExecutionStatus::Pending => sol.status = SolutionStatus::Pending,
113                    UIExecutionStatus::Started { .. } => sol.status = SolutionStatus::$start_status,
114                    UIExecutionStatus::Done { result } => {
115                        if let ExecutionStatus::Success = result.status {
116                            sol.status = SolutionStatus::$ok_status;
117                        } else {
118                            sol.status = SolutionStatus::Failed(format!("{} failed", $name));
119                        }
120                        sol.$step_result = Some(result);
121                    }
122                    UIExecutionStatus::Skipped => {
123                        if let SolutionStatus::Failed(_) = sol.status {
124                        } else {
125                            sol.status = SolutionStatus::Skipped;
126                        }
127                    }
128                }
129                sol
130            }};
131        }
132
133        match message {
134            UIMessage::StopUI => {}
135            UIMessage::ServerStatus { status } => self.executor_status = Some(status),
136            UIMessage::Solutions { solutions } => {
137                self.solutions = solutions
138                    .into_iter()
139                    .map(|info| (info.path, SolutionState::new()))
140                    .collect();
141            }
142            UIMessage::Compilation { file, status } => self
143                .compilations
144                .entry(file)
145                .or_insert(CompilationStatus::Pending)
146                .apply_status(status),
147            UIMessage::TerryTask { .. } => {}
148            UIMessage::TerryGeneration {
149                solution,
150                seed,
151                status,
152            } => {
153                let sol = process_step!(
154                    self,
155                    solution,
156                    status,
157                    generator_result,
158                    Generating,
159                    Generated,
160                    "Generator"
161                );
162                sol.seed = Some(seed);
163            }
164            UIMessage::TerryValidation { solution, status } => {
165                process_step!(
166                    self,
167                    solution,
168                    status,
169                    validator_result,
170                    Validating,
171                    Validated,
172                    "Validator"
173                );
174            }
175            UIMessage::TerrySolution { solution, status } => {
176                process_step!(
177                    self,
178                    solution,
179                    status,
180                    solution_result,
181                    Solving,
182                    Solved,
183                    "Solution"
184                );
185            }
186            UIMessage::TerryChecker { solution, status } => {
187                process_step!(
188                    self,
189                    solution,
190                    status,
191                    checker_result,
192                    Checking,
193                    Done,
194                    "Checker"
195                );
196            }
197            UIMessage::TerrySolutionOutcome { solution, outcome } => {
198                let sol = self
199                    .solutions
200                    .get_mut(&solution)
201                    .expect("Outcome of an unknown solution");
202                sol.outcome = Some(outcome);
203            }
204            UIMessage::Diagnostic { diagnostic } => {
205                self.diagnostics.add_diagnostic(diagnostic);
206            }
207            UIMessage::IOITask { .. }
208            | UIMessage::IOIGeneration { .. }
209            | UIMessage::IOIValidation { .. }
210            | UIMessage::IOISolution { .. }
211            | UIMessage::IOIEvaluation { .. }
212            | UIMessage::IOIChecker { .. }
213            | UIMessage::IOITestcaseScore { .. }
214            | UIMessage::IOISubtaskScore { .. }
215            | UIMessage::IOITaskScore { .. }
216            | UIMessage::IOIBooklet { .. }
217            | UIMessage::IOIBookletDependency { .. } => unreachable!("IOI message on Terry UI"),
218        }
219    }
220
221    fn finish(&mut self) {
222        finish_ui::FinishUI::print(self)
