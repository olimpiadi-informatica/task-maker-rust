36/// Information about a generic Terry task.
37#[derive(Debug, Clone, Serialize, Deserialize, TypeScriptify)]
38pub struct TerryTask {
39    /// Path of the directory of the task.
40    pub path: PathBuf,
41    /// The name of the task (the short one).
42    pub name: String,
43    /// The title of the task (the long one).
44    pub description: String,
45    /// The maximum score for this task.
46    pub max_score: f64,
47
48    /// The statement for this task
49    #[serde(skip_serializing)]
50    pub statement: Option<Statement>,
51    /// The generator of input files of this task.
52    #[serde(skip_serializing)]
53    pub generator: InputGenerator,
54    /// The validator of input files of this task.
55    #[serde(skip_serializing)]
56    pub validator: Option<InputValidator>,
57    /// The checker of input/output files of this task.
58    #[serde(skip_serializing)]
59    pub checker: Checker,
60    /// The official solution of this task, if any. Will be compiled and placed in the sandbox of
61    /// the generation/validation/checking.
62    #[serde(skip_serializing)]
63    pub official_solution: Option<Arc<SourceFile>>,
64    /// The sanity checks attached to this task. Wrapped in Arc since `SanityChecks` is not Clone.
65    /// It's also not `Serialize` nor `Deserialize`, all the sanity checks will be lost on
66    /// serialization.
67    #[serde(skip_serializing, skip_deserializing)]
68    pub sanity_checks: Arc<SanityChecks<TerryTask>>,
69}

169    /// Get an appropriate `UI` for this task.
170    pub fn ui(&self, ui_type: &UIType, _config: ExecutionDAGConfig) -> Result<Box<dyn UI>, Error> {
171        match ui_type {
172            UIType::Raw => Ok(Box::new(RawUI::new())),
173            UIType::Json => Ok(Box::new(JsonUI::new())),
174            UIType::Silent => Ok(Box::new(SilentUI::new())),
175            UIType::Print => Ok(Box::new(PrintUI::new(UIState::new(self)))),
176            UIType::Curses => Ok(Box::new(CursesUI::new(UIState::new(self))?)),
177        }
178    }

180    /// Add the executions required for evaluating this task to the execution DAG.
181    pub fn build_dag(
182        &self,
183        eval: &mut EvaluationData,
184        config: &EvaluationConfig,
185    ) -> Result<(), Error> {
186        eval.sender.send(UIMessage::TerryTask {
187            task: Box::new(self.clone()),
188        })?;
189        eval.solutions = config.find_solutions(&self.path, vec!["solutions/*"], None, eval);
190
191        let solution_info = eval.solutions.iter().map(SolutionInfo::from).collect_vec();
192        eval.sender.send(UIMessage::Solutions {
193            solutions: solution_info,
194        })?;
195
196        let solutions = eval.solutions.clone();
197        for solution in solutions {
198            let seed = if let Some(seed) = config.seed {
199                seed
200            } else {
201                fastrand::u64(0..(i32::MAX as u64))
202            };
203            let input_file = self.generator.generate_and_bind(
204                eval,
205                &solution.source_file,
206                seed,
207                self.official_solution.clone(),
208            )?;
209            let validation_file = if let Some(validator) = self.validator.as_ref() {
210                Some(validator.validate_and_bind(
211                    eval,
212                    &solution.source_file,
213                    input_file,
214                    self.official_solution.clone(),
215                )?)
216            } else {
217                None
218            };
219            let output_file =
220                Solution::solve_and_bind(eval, &solution.source_file, input_file, validation_file)?;
221            let sender = eval.sender.clone();
222            let solution_path = solution.source_file.path.clone();
223            self.checker.check_and_bind(
224                eval,
225                &solution.source_file,
226                input_file,
227                output_file,
228                self.official_solution.clone(),
229                move |outcome| {
230                    sender.send(UIMessage::TerrySolutionOutcome {
231                        solution: solution_path,
232                        outcome: outcome.map_err(|e| format!("Invalid checker outcome: {}", e)),
233                    })
234                },
235            )?;
236        }
237
238        if let Some(statement) = &self.statement {
239            statement.generate_and_bind(eval)?;
240        }
241
242        self.sanity_checks.pre_hook(self, eval)?;
243        Ok(())
244    }

252    /// Clean the task folder removing the files that can be generated automatically.
253    pub fn clean(&self) -> Result<(), Error> {
254        let all_managers: HashSet<PathBuf> = list_files(&self.path, vec!["managers/*.*"])
255            .iter()
256            .map(|f| f.file_stem().unwrap().into())
257            .collect();
258        for maybe_generated in list_files(&self.path, vec!["managers/*.*.*"]) {
259            let name = Path::new(maybe_generated.file_stem().unwrap());
260            let name = Path::new(name.file_stem().unwrap());
261            // if there is a file.X associated to file.Y.Z, remove file.Y.Z, e.g.:
262            // in managers/ there is validator.py => all_managers includes "validator"
263            //   maybe_generated == "validator.linux.x86_64"
264            //   name == "validator"
265            if all_managers.contains(name) {
266                info!("Removing {}", maybe_generated.display());
267                std::fs::remove_file(maybe_generated)?;
268            }
269        }
270        // remove the bin/ folder
271        let bin_path = self.path.join("bin");
272        if bin_path.exists() {
273            info!("Removing {}", bin_path.display());
274            std::fs::remove_dir_all(bin_path)?;
275        }
276        Ok(())
277    }
