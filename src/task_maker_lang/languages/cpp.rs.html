13/// Configuration of the C++ language to use.
14#[derive(Clone, Debug)]
15pub struct LanguageCppConfiguration {
16    /// Compiler to use (e.g. ExecutionCommand::system("g++") ).
17    pub compiler: ExecutionCommand,
18    /// Version of the C++ standard library to use (e.g. c++11).
19    pub std_version: String,
20    /// Extra flags to pass to the compiler.
21    pub extra_flags: Vec<String>,
22}

24/// The C++ language.
25#[derive(Debug)]
26pub struct LanguageCpp {
27    pub config: LanguageCppConfiguration,
28}
29
30impl LanguageCppConfiguration {
31    /// Get the configuration of C++ from the environment variables.
32    pub fn from_env() -> LanguageCppConfiguration {
33        let compiler = std::env::var_os("TM_CXX")
34            .or_else(|| std::env::var_os("CXX"))
35            .unwrap_or_else(|| "g++".into());
36        let std_version = std::env::var("TM_CXX_STD_VERSION").unwrap_or_else(|_| "c++17".into());
37        let extra_flags = std::env::var("TM_CXXFLAGS")
38            .or_else(|_| std::env::var("CXXFLAGS"))
39            .unwrap_or_else(|_| String::new());
40        let extra_flags = shell_words::split(&extra_flags).expect("Invalid $TM_CXXFLAGS");
41        LanguageCppConfiguration {
42            compiler: ExecutionCommand::System(compiler.into()),
43            std_version,
44            extra_flags,
45        }
46    }
47}
48
49impl LanguageCpp {
50    /// Make a new LanguageCpp using the specified version.
51    pub fn new(config: LanguageCppConfiguration) -> LanguageCpp {
52        LanguageCpp { config }
53    }
54}

56impl Language for LanguageCpp {
57    fn name(&self) -> &'static str {
58        "C++"
59    }
60
61    fn extensions(&self) -> Vec<&'static str> {
62        vec!["cpp", "cc", "c++"]
63    }
64
65    fn need_compilation(&self) -> bool {
66        true
67    }
68
69    fn inline_comment_prefix(&self) -> Option<&'static str> {
70        Some("//")
71    }
72
73    fn compilation_builder(
74        &self,
75        source: &Path,
76        settings: CompilationSettings,
77    ) -> Option<Box<dyn CompiledLanguageBuilder + '_>> {
78        let mut metadata = SimpleCompiledLanguageBuilder::new(
79            self,
80            source,
81            settings,
82            self.config.compiler.clone(),
83        );
84        let binary_name = metadata.binary_name.clone();
85        metadata
86            .add_arg("-O2")
87            .add_arg("-Wall")
88            .add_arg("-ggdb3")
89            .add_arg("-DEVAL")
90            .add_arg("-fdiagnostics-color=always")
91            .add_arg(format!("-std={}", self.config.std_version))
92            .add_arg("-o")
93            .add_arg(binary_name);
94        for arg in &self.config.extra_flags {
95            metadata.add_arg(arg);
96        }
97        if metadata.settings.list_static {
98            metadata.add_arg("-static");
99        }
100
101        find_cpp_deps(source)
102            .into_iter()
103            .for_each(|d| metadata.add_dependency(d));
104        Some(Box::new(metadata))
105    }
106}

108/// Extract all the dependencies of a C/C++ source file.
109pub(crate) fn find_cpp_deps(path: &Path) -> Vec<Dependency> {
110    find_dependencies(path, extract_includes)
111}
112
113/// Extracts all the #include in the file. The supported includes are the ones in the form:
114/// * `#include <file>`
115/// * `#include "file"`
116///
117/// The space after `include` is optional.
118///
119/// The returned values are in the form (local_path, sandbox_path). Those paths are equal, and are
120/// just the include itself.
121fn extract_includes(path: &Path) -> Vec<(PathBuf, PathBuf)> {
122    lazy_static! {
123        static ref RE: Regex = Regex::new(r#"#include\s*[<"]([^">]+)[>"]"#).expect("Invalid regex");
124    }
125    let content = match std::fs::read_to_string(path) {
126        Ok(content) => content,
127        _ => return vec![],
128    };
129    let mut res = Vec::new();
130    let mut add_file = |include: PathBuf| {
131        let local = path.with_file_name(&include);
132        res.push((local, include));
133    };
134    for cap in RE.captures_iter(&content) {
135        if let Some(path) = cap.get(1) {
136            add_file(PathBuf::from(path.as_str()));
137        }
138    }
139    res
140}
