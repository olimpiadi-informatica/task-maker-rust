11/// Version of the Python interpreter to use.
12#[allow(dead_code)]
13#[derive(Debug)]
14pub enum LanguagePythonVersion {
15    /// Use the shebang written as the first line of the source.
16    Autodetect,
17    /// Force `python2`
18    Python2,
19    /// Force `python3`
20    Python3,
21}
22
23/// The Python language
24#[derive(Debug)]
25pub struct LanguagePython {
26    version: LanguagePythonVersion,
27}
28
29impl LanguagePython {
30    /// Make a new LanguagePython using the specified version.
31    pub fn new(version: LanguagePythonVersion) -> LanguagePython {
32        LanguagePython { version }
33    }
34}
35
36impl Language for LanguagePython {
37    fn name(&self) -> &'static str {
38        match self.version {
39            LanguagePythonVersion::Autodetect => "Python / Autodetect",
40            LanguagePythonVersion::Python2 => "Python2",
41            LanguagePythonVersion::Python3 => "Python3",
42        }
43    }
44
45    fn extensions(&self) -> Vec<&'static str> {
46        vec!["py"]
47    }
48
49    fn need_compilation(&self) -> bool {
50        false
51    }
52
53    fn inline_comment_prefix(&self) -> Option<&'static str> {
54        Some("#")
55    }
56
57    fn runtime_command(&self, path: &Path, write_to: Option<&Path>) -> ExecutionCommand {
58        match self.version {
59            LanguagePythonVersion::Autodetect => {
60                ExecutionCommand::local(self.executable_name(path, write_to))
61            }
62            LanguagePythonVersion::Python2 => ExecutionCommand::system("python2"),
63            LanguagePythonVersion::Python3 => ExecutionCommand::system("python3"),
64        }
65    }
66
67    fn runtime_args(
68        &self,
69        path: &Path,
70        write_to: Option<&Path>,
71        mut args: Vec<String>,
72    ) -> Vec<String> {
73        match self.version {
74            LanguagePythonVersion::Autodetect => args,
75            _ => {
76                // will run for example: python3 program.py args...
77                args.insert(
78                    0,
79                    self.executable_name(path, write_to)
80                        .to_string_lossy()
81                        .to_string(),
82                );
83                args
84            }
85        }
86    }
87
88    fn runtime_dependencies(&self, path: &Path) -> Vec<Dependency> {
89        find_python_deps(path)
90    }
91}

93/// Extract all the dependencies of a python file recursively.
94fn find_python_deps(path: &Path) -> Vec<Dependency> {
95    find_dependencies(path, extract_imports)
96}

98/// Extracts all the imports in the file. The supported imports are the ones in
99/// the form:
100/// * `import __file__`
101/// * `from __file__ import stuff`
102/// * `import __file1__, __file2__`
103///
104/// The returned values are in the form (local_path, sandbox_path). Those paths
105/// are equal, and are just the import followed by `.py`, because of that the
106/// imports should not contain a dot (i.e. python modules are not supported).
107fn extract_imports(path: &Path) -> Vec<(PathBuf, PathBuf)> {
108    lazy_static! {
109        static ref RE: Regex =
110            Regex::new("import +(.+)|from +(.+) +import").expect("Invalid regex");
111    }
112    let content = match std::fs::read_to_string(path) {
113        Ok(content) => content,
114        _ => return vec![],
115    };
116    let mut res = Vec::new();
117    for cap in RE.captures_iter(&content) {
118        if let Some(type1) = cap.get(1) {
119            for piece in type1.as_str().split(',') {
120                let path = PathBuf::from(format!("{}.py", piece.trim()));
121                res.push((path.clone(), path));
122            }
123        } else if let Some(type2) = cap.get(2) {
124            let path = PathBuf::from(format!("{}.py", type2.as_str().trim()));
125            res.push((path.clone(), path));
126        }
127    }
128    res
129}

139    #[test]
140    fn test_runtime_args_autodetect() {
141        let lang = LanguagePython::new(LanguagePythonVersion::Autodetect);
142        let path = Path::new("script.py");
143        let write_to = Path::new("script.boh");
144        assert_that(&lang.runtime_command(path, Some(write_to)))
145            .is_equal_to(ExecutionCommand::local(write_to));
146        let args = lang.runtime_args(path, None, vec!["arg".to_string()]);
147        assert_that(&args).is_equal_to(vec!["arg".to_string()]);
148    }
149
150    #[test]
151    fn test_runtime_args_py3() {
152        let lang = LanguagePython::new(LanguagePythonVersion::Python3);
153        let path = Path::new("script.py");
154        let write_to = Path::new("script.boh");
155        assert_that(&lang.runtime_command(path, Some(write_to)))
156            .is_equal_to(ExecutionCommand::system("python3"));
157        let args = lang.runtime_args(path, Some(write_to), vec!["arg".to_string()]);
158        assert_that(&args).is_equal_to(vec!["script.boh".to_string(), "arg".to_string()]);
159    }

161    #[test]
162    fn test_extract_imports() {
163        let tmpdir = tempfile::TempDir::new().unwrap();
164        let path = tmpdir.path().join("script.py");
165        write(
166            &path,
167            "random stuff\nimport foo\nfrom bar import xxx\nimport baz, biz",
168        )
169        .unwrap();
170        let imports = extract_imports(&path);
171        for (i, import) in ["foo", "bar", "baz", "biz"].iter().enumerate() {
172            let import = PathBuf::from(format!("{}.py", import));
173            assert_that(&imports[i]).is_equal_to((import.clone(), import));
174        }
175    }

177    #[test]
178    fn test_find_python_deps() {
179        let tmpdir = tempfile::TempDir::new().unwrap();
180        let path = tmpdir.path().join("script.py");
181        let foo_path = tmpdir.path().join("foo.py");
182        let bar_path = tmpdir.path().join("bar.py");
183        write(&path, "import foo").unwrap();
184        write(&foo_path, "import bar").unwrap();
185        write(&bar_path, "import not_found").unwrap();
186        let deps = find_python_deps(&path);
187        assert_that(&deps).has_length(2);
188        assert_that(&deps[0].local_path).is_equal_to(foo_path);
189        assert_that(&deps[0].sandbox_path).is_equal_to(PathBuf::from("foo.py"));
190        assert_that(&deps[1].local_path).is_equal_to(bar_path);
191        assert_that(&deps[1].sandbox_path).is_equal_to(PathBuf::from("bar.py"));
192    }

194    #[test]
195    fn test_find_python_deps_loop() {
196        let tmpdir = tempfile::TempDir::new().unwrap();
197        let script_path = tmpdir.path().join("script.py");
198        let foo_path = tmpdir.path().join("foo.py");
199        // script imports itself and foo and script import each other

201        write(&foo_path, "import script").unwrap();
202        let deps = find_python_deps(&script_path);
203        assert_that(&deps).has_length(1);
204        assert_that(&deps[0].local_path).is_equal_to(foo_path);

206    }
