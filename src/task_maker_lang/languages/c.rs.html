11/// Configuration of the C language to use.
12#[derive(Clone, Debug)]
13pub struct LanguageCConfiguration {
14    /// Compiler to use (e.g. ExecutionCommand::system("gcc") ).
15    pub compiler: ExecutionCommand,
16    /// Version of the C standard library to use (e.g. c11).
17    pub std_version: String,
18    /// Extra flags to pass to the compiler.
19    pub extra_flags: Vec<String>,
20}
21
22/// The C language.
23#[derive(Debug)]
24pub struct LanguageC {
25    pub config: LanguageCConfiguration,
26}
27
28impl LanguageC {
29    /// Make a new LanguageC using the specified version.
30    pub fn new(config: LanguageCConfiguration) -> LanguageC {
31        LanguageC { config }
32    }
33}

35impl LanguageCConfiguration {
36    /// Get the configuration of C from the environment variables.
37    pub fn from_env() -> LanguageCConfiguration {
38        let compiler = std::env::var_os("TM_CC")
39            .or_else(|| std::env::var_os("CC"))
40            .unwrap_or_else(|| "gcc".into());
41        let std_version = std::env::var("TM_CC_STD_VERSION").unwrap_or_else(|_| "c11".into());
42        let extra_flags = std::env::var("TM_CFLAGS")
43            .or_else(|_| std::env::var("CFLAGS"))
44            .unwrap_or_else(|_| String::new());
45        let extra_flags = shell_words::split(&extra_flags).expect("Invalid $TM_CFLAGS");
46        LanguageCConfiguration {
47            compiler: ExecutionCommand::System(compiler.into()),
48            std_version,
49            extra_flags,
50        }
51    }
52}

54impl Language for LanguageC {
55    fn name(&self) -> &'static str {
56        "C"
57    }
58
59    fn extensions(&self) -> Vec<&'static str> {
60        vec!["c"]
61    }
62
63    fn need_compilation(&self) -> bool {
64        true
65    }
66
67    fn inline_comment_prefix(&self) -> Option<&'static str> {
68        Some("//")
69    }
70
71    fn compilation_builder(
72        &self,
73        source: &Path,
74        settings: CompilationSettings,
75    ) -> Option<Box<dyn CompiledLanguageBuilder + '_>> {
76        let mut metadata = SimpleCompiledLanguageBuilder::new(
77            self,
78            source,
79            settings,
80            self.config.compiler.clone(),
81        );
82        let binary_name = metadata.binary_name.clone();
83        metadata
84            .add_arg("-O2")
85            .add_arg("-Wall")
86            .add_arg("-ggdb3")
87            .add_arg("-DEVAL")
88            .add_arg("-fdiagnostics-color=always")
89            .add_arg(format!("-std={}", self.config.std_version))
90            .add_arg("-o")
91            .add_arg(binary_name)
92            .add_arg("-lm");
93        for arg in &self.config.extra_flags {
94            metadata.add_arg(arg);
95        }
96        if metadata.settings.list_static {
97            metadata.add_arg("-static");
98        }
99
100        find_cpp_deps(source)
101            .into_iter()
102            .for_each(|d| metadata.add_dependency(d));
103        Some(Box::new(metadata))
104    }
105}

116    fn setup() -> TempDir {

122
123    #[test]
124    fn test_compilation_args() {
125        let tmp = setup();
126
127        let lang = LanguageC::new(LanguageCConfiguration {
128            compiler: ExecutionCommand::System("gcc".into()),
129            std_version: "c11".to_string(),
130            extra_flags: vec!["-lfoobar".into()],
131        });
132        let mut builder = lang
133            .compilation_builder(&tmp.path().join("foo.c"), CompilationSettings::default())
134            .unwrap();
135        let (comp, _exec) = builder.finalize(&mut ExecutionDAG::new()).unwrap();
136
137        let args = comp.args;
138        assert_that(&args).contains("foo.c".to_string());
139        assert_that(&args).contains("-std=c11".to_string());
140        assert_that(&args).contains("-lfoobar".to_string());
141        assert_that(&args).does_not_contain("-static".to_string());
142    }
