9/// Trait that defines the properties of the supported languages. Most of the methods have a safe
10/// blanket implementation, note that not all of them are _really_ optional: based on the value
11/// returned by `need_compilation` some of the methods become required.
12///
13/// A language can be either compiler or not-compiler.
14///
15/// When a language is compiled the extra required implementations are:
16/// - `compilation_builder`
17pub trait Language: std::fmt::Debug + Send + Sync {
18    /// Full name of the language. This must be unique between all the other languages.
19    fn name(&self) -> &'static str;
20
21    /// List of valid extensions for this language. A file is considered in this language if its
22    /// extension is inside this list.
23    fn extensions(&self) -> Vec<&'static str>;
24
25    /// Whether this language needs a compilation step. Returning `true` here triggers many changes
26    /// in the behaviour of the execution. Of course the compilation step will be added, because of
27    /// that there is the need to know how to compile the source file, forcing the implementation of
28    /// some extra methods.
29    fn need_compilation(&self) -> bool;
30
31    /// The prefix to put at the start of a line to mark the whole line as a comment.
32    ///
33    /// The return value should include a space character only if required by the language.
34    ///
35    /// If the language does not support inline comments, return `None`.
36    fn inline_comment_prefix(&self) -> Option<&'static str>;
37
38    /// Return the `CompiledLanguageBuilder` for compiling a source file with this language.
39    ///
40    /// This method must return `Some` if and only if `need_compilation` returns `true`.
41    fn compilation_builder(
42        &self,
43        _source: &Path,
44        _settings: CompilationSettings,
45    ) -> Option<Box<dyn CompiledLanguageBuilder + '_>> {
46        None
47    }
48
49    /// Command to use to run the program. It defaults to the executable name of the program.
50    /// Languages that need to run a separate program (e.g. a system-wise interpreter) may change
51    /// the return value of this method.
52    fn runtime_command(&self, path: &Path, write_to: Option<&Path>) -> ExecutionCommand {
53        ExecutionCommand::local(self.executable_name(path, write_to))
54    }
55
56    /// Arguments to pass to the executable to start the evaluation.
57    fn runtime_args(
58        &self,
59        _path: &Path,
60        _write_to: Option<&Path>,
61        args: Vec<String>,
62    ) -> Vec<String> {
63        args
64    }
65
66    /// Add a file to the runtime command if the language requires that. That file can be any run
67    /// time dependency and it's relative to the sandbox.
68    ///
69    /// The new runtime arguments should be returned.
70    fn runtime_add_file(&self, args: Vec<String>, _file: &Path) -> Vec<String> {
71        args
72    }
73
74    /// The dependencies to put inside the execution sandbox. This does not include the executable.
75    fn runtime_dependencies(&self, _path: &Path) -> Vec<Dependency> {
76        vec![]
77    }
78
79    /// Update the limits for some language-specific requirements. For example the executable may
80    /// need to fork (hence use more processes).
81    fn custom_limits(&self, _limits: &mut ExecutionLimits) {}
82
83    /// The name of the executable inside the sandbox. If this binary will be written elsewhere in
84    /// the system, use the same name. Otherwise fallback to the original file name, without
85    /// extension.
86    fn executable_name(&self, path: &Path, write_to: Option<&Path>) -> PathBuf {
87        if let Some(write_to) = write_to {
88            PathBuf::from(write_to.file_name().expect("Invalid file name"))
89        } else {
90            PathBuf::from(path.file_stem().expect("Invalid file name"))
91        }
92    }
93}

95/// The generic settings that are common between all the compiled languages.
96#[derive(Clone, Debug, Default)]
97pub struct CompilationSettings {
98    /// Where to write the compiled binary file.
99    pub write_to: Option<PathBuf>,
100    /// Whether to write to `write_to` the compiled binary.
101    pub copy_exe: bool,
102    /// Whether to try to link statically the binary.
103    pub list_static: bool,
104}

106/// This trait describes the API of a "compiled language builder", a component that builds the DAG
107/// execution for the compilation.
108pub trait CompiledLanguageBuilder {
109    /// If a grader map is present, provide it with this method.
110    fn use_grader(&mut self, grader_map: &GraderMap);
111    /// Build the execution to be added to the DAG for compiling the source file.
112    ///
113    /// This returns the execution to add and the file reference to the compiled binary file.
114    ///
115    /// After calling this method, the builder cannot be used anymore. This cannot be enforced by
116    /// taking self because otherwise this trait is no longer object-safe.
117    fn finalize(&mut self, dag: &mut ExecutionDAG) -> Result<(Execution, File), Error>;
118}

120/// A simple `CompiledLanguageBuilder` that is able to compile file in most of the languages.
121///
122/// It supports customizing the compiler, the command line arguments, a grader, a custom list of
123/// additional dependencies, and also customizing the produced execution with a callback.
124///
125/// Some notes about the compilation:
126///
127/// - `language` must return true from `need_compilation`.
128/// - `args` should not include the path to the source files (`finalize()` will add them)
129/// - The compiler must write the binary file in `binary_name`, you can change this after
130///   constructing the struct and before calling `finalize()`.
131/// - The callback is called just before `finalize()` returns, so all the compilation properties
132///   are already set.
133/// - The `grader` will be set by `use_grader`, so your changes may be overwritten.
134/// - After the first call to `finalize()` this struct cannot be used (see note on the definition
135///   of `finalize` in the trait).
136pub struct SimpleCompiledLanguageBuilder<'l, 'c> {
137    /// A reference to the language that produced this builder.
138    ///
139    /// This is used to select which grader to use.
140    pub language: &'l dyn Language,
141    /// The settings for this compilation.
142    pub settings: CompilationSettings,
143    /// The local path to the source file to compile.
144    pub source_path: PathBuf,
145    /// The name of source file to compile.
146    pub source_name: String,
147    /// The name of the compiled file in the sandbox.
148    pub binary_name: String,
149    /// The compiler to use.
150    pub compiler: ExecutionCommand,
151    /// The list of arguments to pass to the compiler.
152    ///
153    /// To this list the paths to the source files are added automatically and should not be
154    /// present.
155    pub args: Vec<String>,
156    /// The grader to use.
157    pub grader: Option<Dependency>,
158    /// The list of additional compilation dependencies.
159    pub dependencies: Vec<Dependency>,
160    /// Whether the compiler wants only the path to the grader file, or the paths to all the source
161    /// files to compile together.
162    pub grader_only: bool,
163    /// The callback to call with the built `Execution` for additional customizations.
164    #[allow(clippy::type_complexity)]
165    pub callback: Option<Box<dyn FnOnce(&mut Execution) + 'c>>,
166}

168impl<'l, 'c> SimpleCompiledLanguageBuilder<'l, 'c> {
169    /// Build a new `SimpleCompiledLanguageBuilder` for a source file in the given language.
170    pub fn new<P: Into<PathBuf>>(
171        language: &'l dyn Language,
172        source_path: P,
173        settings: CompilationSettings,
174        compiler: ExecutionCommand,
175    ) -> Self {
176        let source_path = source_path.into();
177        let mut source_name = source_path
178            .file_name()
179            .expect("Invalid file name")
180            .to_string_lossy()
181            .to_string();
182        // names starting with - can be interpreted as command line options
183        if source_name.starts_with('-') {
184            source_name = format!("./{}", source_name);
185        }
186        Self {
187            language,
188            settings,
189            source_path,
190            source_name,
191            binary_name: "__compiled".into(),
192            compiler,
193            args: Default::default(),
194            grader: None,
195            dependencies: Default::default(),
196            grader_only: false,
197            callback: None,
198        }
199    }
200
201    /// Add a new argument to the list.
202    pub fn add_arg<S: Into<String>>(&mut self, arg: S) -> &mut Self {
203        self.args.push(arg.into());
204        self
205    }
206
207    /// Add an additional dependency.
208    pub fn add_dependency(&mut self, dependency: Dependency) {
209        self.dependencies.push(dependency);
210    }
211
212    /// Pass only the path to the grader source file to the compiler, if the grader is present. If
213    /// no grader is present, pass just the path to the actual source file.
214    pub fn grader_only(&mut self) {
215        self.grader_only = true;
216    }
217
218    /// Set the callback to call with the built execution.
219    pub fn callback<F: FnOnce(&mut Execution) + 'c>(&mut self, callback: F) {
220        self.callback = Some(Box::new(callback));
221    }
222}

224impl CompiledLanguageBuilder for SimpleCompiledLanguageBuilder<'_, '_> {
225    fn use_grader(&mut self, grader_map: &GraderMap) {
226        if let Some(grader) = grader_map.get_compilation_deps(self.language) {
227            self.grader = Some(grader);
228        }
229    }
230
231    fn finalize(&mut self, dag: &mut ExecutionDAG) -> Result<(Execution, File), Error> {
232        let name = self.source_path.file_name().unwrap().to_string_lossy();
233        let mut comp = Execution::new(format!("Compilation of {}", name), self.compiler.clone());
234        comp.args.clone_from(&self.args);
235
236        // compilation dependencies
237        for dep in self.dependencies.drain(..) {
238            comp.input(&dep.file, &dep.sandbox_path, dep.executable);
239            dag.provide_file(dep.file, &dep.local_path)
240                .context("Failed to provide compilation dependency")?;
241        }
242
243        // main source file
244        let source = File::new(format!("Source file of {:?}", self.source_path));
245        comp.input(&source, &self.source_name, false);
246        dag.provide_file(source, &self.source_path)
247            .context("Failed to provide source file")?;
248
249        // grader and sources in args
250        if let Some(grader) = self.grader.take() {
251            comp.args
252                .push(grader.sandbox_path.to_string_lossy().to_string());
253            comp.input(&grader.file, &grader.sandbox_path, grader.executable);
254            dag.provide_file(grader.file, &grader.local_path)
255                .context("Failed to provide grader dependency")?;
256
257            if !self.grader_only {
258                comp.args.push(self.source_name.clone());
259            }
260        } else {
261            comp.args.push(self.source_name.clone());
262        }
263
264        // compiled binary
265        let exec = comp.output(&self.binary_name);
266        if self.settings.copy_exe {
267            if let Some(write_to) = &self.settings.write_to {
268                dag.write_file_to(&exec, write_to, true);
269            }
270        }
271
272        if let Some(callback) = self.callback.take() {
273            callback(&mut comp);
274        }
275
276        Ok((comp, exec))
277    }
