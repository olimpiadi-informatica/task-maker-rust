16#[derive(Parser, Debug, Clone)]
17pub struct SandboxOpt {
18    /// Working directory of the sandbox.
19    ///
20    /// Will be mounted in /box inside the sandbox. Defaults to current working directory.
21    #[clap(long, short)]
22    pub workdir: Option<PathBuf>,
23
24    /// Memory limit for the sandbox, in KiB.
25    #[clap(long, short)]
26    pub memory_limit: Option<u64>,
27
28    /// Stack limit for the sandbox, in KiB.
29    #[clap(long, short)]
30    pub stack_limit: Option<u64>,
31
32    /// Prevent forking.
33    #[clap(long)]
34    pub single_process: bool,
35
36    /// List of additional directory mounted read-only inside the sandbox.
37    #[clap(long, short)]
38    pub readable_dirs: Vec<PathBuf>,
39
40    /// Mount /tmp and /dev/null inside the sandbox
41    #[clap(long)]
42    pub mount_tmpfs: bool,
43
44    /// Mount /proc inside the sandbox
45    #[clap(long)]
46    pub mount_proc: bool,
47
48    /// User id.
49    #[clap(long, default_value = "1000")]
50    pub uid: usize,
51
52    /// User id.
53    #[clap(long, default_value = "1000")]
54    pub gid: usize,
55
56    /// Command to execute inside the sandbox. If not specified, bash is executed.
57    pub command: Vec<String>,
58}
59
60pub fn main_sandbox(opt: SandboxOpt) -> Result<(), Error> {
61    let mut config = SandboxConfiguration::default();
62
63    #[cfg(not(target_os = "macos"))]
64    let _tempdir = {
65        let tempdir = TempDir::new()?;
66        let etcdir = tempdir.path();
67        config.working_directory("/box");
68
69        let workdir = opt
70            .workdir
71            .or_else(|| std::env::current_dir().ok())
72            .unwrap_or_else(|| "/".into());
73        config.mount(&workdir, "/box", true);
74
75        config.mount(etcdir, "/etc", true);
76        std::fs::write(
77            etcdir.join("passwd"),
78            format!(
79                "root::0:0::/:/bin/sh\n\
80                nobody::{uid}:{gid}::/:/bin/sh\n",
81                uid = opt.uid,
82                gid = opt.gid,
83            ),
84        )
85        .with_context(|| format!("Failed to write /etc/etcdir in {}", etcdir.display()))?;
86        std::fs::write(
87            etcdir.join("group"),
88            format!(
89                "root:x:0:root\n\
90                nobody:x:{gid}:nobody\n",
91                gid = opt.gid,
92            ),
93        )
94        .with_context(|| format!("Failed to write /etc/group in {}", etcdir.display()))?;
95
96        for dir in READABLE_DIRS {
97            if Path::new(dir).is_dir() {
98                config.mount(dir, dir, false);
99            }
100        }
101        for dir in &opt.readable_dirs {
102            if dir.is_dir() {
103                config.mount(dir, dir, false);
104            } else {
105                warn!("Cannot mount directory {}", dir.display());
106            }
107        }
108
109        if opt.mount_tmpfs {
110            config.mount_tmpfs(true);
111        }
112        if opt.mount_proc {
113            config.mount_proc(true);
114        }
115
116        tempdir
117    };
118
119    config.env("PATH", std::env::var("PATH").unwrap_or_default());
120    config.env("LANG", std::env::var("LANG").unwrap_or_default());
121    config.env("LC_ALL", std::env::var("LC_ALL").unwrap_or_default());
122
123    if let Some(memory_limit) = opt.memory_limit {
124        config.memory_limit(memory_limit * 1024);
125    }
126    if let Some(stack_limit) = opt.stack_limit {
127        config.stack_limit(stack_limit * 1024);
128    }
129
130    let multiproc = !opt.single_process;
131    config.syscall_filter(SyscallFilter::build(multiproc, true));
132
133    if opt.command.is_empty() {
134        config.executable("/bin/bash");
135    } else {
136        config.executable(&opt.command[0]);
137        for arg in &opt.command[1..] {
138            config.arg(arg);
139        }
140    }
141
142    config.uid(opt.uid);
143    config.gid(opt.gid);
144
145    debug!("Config: {:#?}", config);
146
147    let sandbox = SandboxImplementation::run(config).context("Failed to create sandbox")?;
148    let res = sandbox.wait().context("Failed to wait sandbox")?;
149
150    debug!("Result: {:#?}", res);
151
152    Ok(())
153}
