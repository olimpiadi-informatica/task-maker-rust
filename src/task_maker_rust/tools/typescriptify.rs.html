27/// Print to stdout the type definition for the specified type.
28macro_rules! export_ts {
29    ($t : ty) => {
30        println!("{}", fix_bug_1(<$t>::type_script_ify()))
31    };
32}

34/// Apply the fix from PR#1 fixing some invalid types.
35///
36/// https://github.com/arabidopsis/typescript-definitions/pull/1
37fn fix_bug_1<S: AsRef<str>>(def: S) -> String {
38    let regex = Regex::new(r"(?m)\[\s*(\w+)\s*:\s*(\w+)\s*\]").unwrap();
39    let substitution = "[$1 in $2]";
40    regex.replace_all(def.as_ref(), substitution).to_string()
41}
