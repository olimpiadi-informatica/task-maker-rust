14/// This state is owned by the UI.
15///
16/// It contains a shared state that can be accessed by the outside world.
17#[derive(Debug, Clone)]
18pub struct UIState {
19    /// A callback that can be used by the UI to stop the current executor.
20    pub stop_evaluation: StopEvaluation,
21
22    /// The path to the solution to evaluate.
23    pub solution: PathBuf,
24    /// The template arguments passed to the generator.
25    pub generator_args: Vec<String>,
26    /// The size of the batch.
27    pub batch_size: usize,
28
29    /// The current status of the executor, if any.
30    pub executor_status: Option<ExecutorStatus<SystemTime>>,
31    /// The current progress of the evaluation.
32    pub progress: Progress,
33    /// The set of batches that have been run.
34    pub batches: Vec<CurrentBatch>,
35
36    /// The part of the state shared with the outside world (i.e. by non-UI code).
37    pub shared: Arc<RwLock<SharedUIState>>,
38}

44/// The current progress of the evaluation.
45#[derive(Debug, Clone, Default)]
46pub struct Progress {
47    /// The number of input files that have been generated.
48    pub inputs_generated: usize,
49    /// The number of input cases that have been solved.
50    pub inputs_solved: usize,
51    /// The sum of the times of the generators. This can be used to compute the average execution
52    /// time.
53    pub generator_time_sum: f64,
54    /// The sum of the times of the solution. This can be used to compute the average execution
55    /// time.
56    pub solution_time_sum: f64,
57}

59/// Information about the status of the current batch.
60#[derive(Debug, Clone)]
61pub struct CurrentBatch {
62    /// The information about each testcase in the batch.
63    pub testcase_status: Vec<TestcaseStatus>,
64}
65
66impl CurrentBatch {
67    fn new(batch_size: usize) -> Self {
68        Self {
69            testcase_status: (0..batch_size).map(|_| TestcaseStatus::Pending).collect(),
70        }
71    }
72}

74/// The status of the evaluation of each testcase.
75#[derive(Debug, Clone)]
76pub enum TestcaseStatus {
77    /// The testcase has not been generated yey.
78    Pending,
79    /// The generator is running.
80    Generating,
81    /// The generator has run, waiting for the validator.
82    Generated,
83    /// The validator is running.
84    Validating,
85    /// The validator has run, waiting for the solution.
86    Validated,
87    /// The solution is running.
88    Solving,
89    /// The solution has run, waiting for the checker.
90    Solved,
91    /// The checker is running.
92    Checking,
93    /// The checker has run, and the solution has solved the testcase correctly.
94    Success,
95    /// The solution failed to solve the testcase.
96    Failed(String),
97    /// An error occurred while producing the testcase.
98    Error,
99}

101/// This is the state shared between the UI and the non-UI code.
102#[derive(Debug, Clone, Default)]
103pub struct SharedUIState {
104    /// The index of the current batch.
105    pub batch_index: usize,
106    /// Whether the UI and the execution should stop and no further batch should be tried.
107    pub should_stop: bool,
108    /// The last batch being evaluated.
109    pub last_batch: Option<Batch>,
110    /// A testcase that made the solution fail, together with a failing message.
111    pub failing_testcase: Option<(TestcaseData, String)>,
112    /// A testcase that failed to generate, together with a message and the result of the execution.
113    pub errored_testcase: Option<(TestcaseData, String, ExecutionResult)>,
114}

116impl UIState {
117    pub fn new(opt: &FindBadCaseOpt, stop_evaluation: impl Fn() + Send + Sync + 'static) -> Self {
118        Self {
119            stop_evaluation: StopEvaluation::new(stop_evaluation),

121            generator_args: opt.generator_args.clone(),
122            batch_size: opt.batch_size,
123            executor_status: None,

131impl UIStateT for UIState {
132    fn apply(&mut self, message: UIMessage) {
133        let mut set = |testcase: TestcaseId, state: TestcaseStatus| {
134            let testcase = &mut self.batches.last_mut().unwrap().testcase_status
135                [testcase as usize % self.batch_size];
136            *testcase = state;
137        };
138        match message {
139            UIMessage::IOITask { .. } => {
140                self.batches.push(CurrentBatch::new(self.batch_size));
141            }
142            UIMessage::ServerStatus { status } => self.executor_status = Some(status),
143            UIMessage::IOIGeneration {
144                testcase, status, ..
145            } => match status {
146                UIExecutionStatus::Started { .. } => set(testcase, TestcaseStatus::Generating),
147                UIExecutionStatus::Done { result } => {
148                    self.progress.inputs_generated += 1;
149                    self.progress.generator_time_sum += result.resources.cpu_time;
150                    if result.status.is_success() {
151                        set(testcase, TestcaseStatus::Generated);
152                    } else {
153                        set(testcase, TestcaseStatus::Error);
154                        let mut shared = self.shared.write().unwrap();
155                        let testcase = shared
156                            .last_batch
157                            .as_ref()
158                            .unwrap()
159                            .testcases
160                            .get(&testcase)
161                            .unwrap();
162                        shared.errored_testcase =
163                            Some((testcase.clone(), "Generator failed".into(), result));
164                    }
165                }
166                _ => {}
167            },
168            UIMessage::IOIValidation {
169                testcase, status, ..
170            } => match status {
171                UIExecutionStatus::Started { .. } => set(testcase, TestcaseStatus::Validating),
172                UIExecutionStatus::Done { result } => {
173                    if result.status.is_success() {
174                        set(testcase, TestcaseStatus::Validated);
175                    } else {
176                        set(testcase, TestcaseStatus::Error);
177                        let mut shared = self.shared.write().unwrap();
178                        let testcase = shared
179                            .last_batch
180                            .as_ref()
181                            .unwrap()
182                            .testcases
183                            .get(&testcase)
184                            .unwrap();
185                        shared.errored_testcase =
186                            Some((testcase.clone(), "Validator failed".into(), result));
187                    }
188                }
189                _ => {}
190            },
191            UIMessage::IOIEvaluation {
192                testcase, status, ..
193            } => match status {
194                UIExecutionStatus::Started { .. } => set(testcase, TestcaseStatus::Solving),
195                UIExecutionStatus::Done { result } => {
196                    if result.status.is_success() {
197                        set(testcase, TestcaseStatus::Solved);
198                    }
199                    self.progress.inputs_solved += 1;
200                    self.progress.solution_time_sum += result.resources.cpu_time;
201                }
202                _ => {}
203            },
204            UIMessage::IOIChecker {
205                testcase,
206                status: UIExecutionStatus::Started { .. },
207                ..
208            } => {
209                set(testcase, TestcaseStatus::Checking);
210            }
211            UIMessage::IOITestcaseScore {
212                testcase,
213                score,
214                message,
215                ..
216            } => {
217                if score == 1.0 {
218                    set(testcase, TestcaseStatus::Success);
219                } else {
220                    set(testcase, TestcaseStatus::Failed(message.clone()));
221                    let mut shared = self.shared.write().unwrap();
222                    let testcase = shared.last_batch.as_ref().unwrap().testcases.get(&testcase);
223                    shared.failing_testcase = testcase.map(|tc| (tc.clone(), message));
224                    shared.should_stop = true;
225                    self.stop_evaluation.stop();
226                }
227            }
228            _ => {}
229        }
230    }
231
232    fn finish(&mut self) {}
233}

235impl StopEvaluation {
236    fn new(stop_evaluation: impl Fn() + Send + Sync + 'static) -> Self {
237        Self(Arc::new(stop_evaluation))
238    }
239
240    fn stop(&self) {
241        (self.0)()
242    }
243}
