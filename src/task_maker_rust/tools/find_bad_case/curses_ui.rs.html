14impl CursesDrawer<UIState> for CursesUI {
15    fn draw(state: &UIState, frame: &mut Frame, loading: char, frame_index: usize) {
16        CursesUI::draw_frame(state, frame, loading, frame_index);
17    }

20impl CursesUI {
21    fn draw_frame(state: &UIState, f: &mut Frame, loading: char, frame_index: usize) {
22        let header_len = 10; // Number of lines of the header.
23        let workers_len = state
24            .executor_status
25            .as_ref()
26            .map(|s| s.connected_workers.len())
27            .unwrap_or(0) as u16
28            + 2;
29        // FIXME: shrink workers_len if needed
30        let chunks = Layout::default()
31            .direction(Direction::Vertical)
32            .constraints(
33                [
34                    Constraint::Length(header_len),
35                    Constraint::Min(0),
36                    Constraint::Length(workers_len),
37                ]
38                .as_ref(),
39            )
40            .split(f.area());
41
42        let shared = state.shared.read().unwrap();
43        Self::render_header(state, &shared, f, chunks[0]);
44        Self::render_generation_status(state, f, chunks[1]);
45        render_server_status(
46            f,
47            chunks[2],
48            state.executor_status.as_ref(),
49            loading,
50            frame_index,
51        );
52    }
53
54    fn render_header(state: &UIState, shared: &SharedUIState, f: &mut Frame, rect: Rect) {
55        let errors = state
56            .batches
57            .iter()
58            .flat_map(|b| b.testcase_status.iter())
59            .filter(|tc| matches!(tc, TestcaseStatus::Error))
60            .count();
61
62        let text = vec![
63            Line::from(vec![
64                Span::styled("Solution:        ", *BOLD),
65                Span::raw(state.solution.to_string_lossy().to_string()),
66            ]),
67            Line::from(vec![
68                Span::styled("Generator args:  ", *BOLD),
69                Span::raw(state.generator_args.iter().join(" ")),
70            ]),
71            Line::from(vec![
72                Span::styled("Batch size:      ", *BOLD),
73                Span::raw(state.batch_size.to_string()),
74            ]),
75            Line::from(vec![
76                Span::styled("Batch index:     ", *BOLD),
77                Span::raw(shared.batch_index.to_string()),
78            ]),
79            Line::from(vec![Span::styled("Progress:", *BLUE)]),
80            Line::from(vec![
81                Span::styled("    Generated:   ", *BOLD),
82                Span::raw(state.progress.inputs_generated.to_string()),
83            ]),
84            Line::from(vec![
85                Span::styled("    Solved:      ", *BOLD),
86                Span::raw(state.progress.inputs_solved.to_string()),
87            ]),
88            Line::from(vec![
89                Span::styled("    Average gen: ", *BOLD),
90                Span::raw(format!(
91                    "{:.3}s\n",
92                    state.progress.generator_time_sum
93                        / (state.progress.inputs_generated.max(1) as f64)
94                )),
95            ]),
96            Line::from(vec![
97                Span::styled("    Average sol: ", *BOLD),
98                Span::raw(format!(
99                    "{:.3}s",
100                    state.progress.solution_time_sum / (state.progress.inputs_solved.max(1) as f64)
101                )),
102            ]),
103            Line::from(vec![
104                Span::styled("    Errors:      ", *BOLD),
105                Span::raw(errors.to_string()),
106            ]),
107        ];
108
109        let paragraph = Paragraph::new(text);
110        f.render_widget(paragraph, rect);
111    }
112
113    fn render_generation_status(state: &UIState, f: &mut Frame, rect: Rect) {
114        let mut text = vec![];
115        for i in 0..state.batches.len().min(10) {
116            let mut line = Vec::new();
117            let batch_index = state.batches.len() - 1 - i;
118            let batch = &state.batches[state.batches.len() - 1 - i];
119            line.push(Span::raw(format!("Batch {:>3}: ", batch_index)));
120            line.extend(
121                batch
122                    .testcase_status
123                    .iter()
124                    .map(Self::testcase_status_to_text),
125            );
126            text.push(line.into());
127        }
128        render_block(f, rect, "Progress");
129        let paragraph = Paragraph::new(text).wrap(Wrap { trim: true });
130        f.render_widget(paragraph, inner_block(rect));
131    }
132
133    fn testcase_status_to_text(status: &TestcaseStatus) -> Span {
134        match status {
135            TestcaseStatus::Pending => Span::raw("."),
136            TestcaseStatus::Generating => Span::raw("g"),
137            TestcaseStatus::Generated => Span::raw("G"),
138            TestcaseStatus::Validating => Span::raw("v"),
139            TestcaseStatus::Validated => Span::raw("V"),
140            TestcaseStatus::Solving => Span::raw("s"),
141            TestcaseStatus::Solved => Span::raw("S"),
142            TestcaseStatus::Checking => Span::raw("c"),
143            TestcaseStatus::Success => Span::styled("✓", *GREEN),
144            TestcaseStatus::Failed(_) => Span::styled("✕", *RED),
145            TestcaseStatus::Error => Span::styled("!", *RED),
146        }
147    }
148}
