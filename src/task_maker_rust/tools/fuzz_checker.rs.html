18#[derive(Parser, Debug, Clone)]
19pub struct FuzzCheckerOpt {
20    #[clap(flatten, next_help_heading = Some("TASK SEARCH"))]
21    pub find_task: FindTaskOpt,
22
23    /// Where to store fuzzing data.
24    ///
25    /// The path is relative to the task directory.
26    #[clap(long, default_value = "fuzz")]
27    pub fuzz_dir: PathBuf,
28
29    /// Additional sanitizers to use.
30    ///
31    /// Comma separated list of sanitizers to use.
32    #[clap(long, default_value = "address,undefined")]
33    pub sanitizers: String,
34
35    /// List of additional arguments to pass to the compiler.
36    ///
37    /// If nothing is listed here, -O2 and -g are passed.
38    pub extra_args: Vec<String>,
39
40    /// Number of fuzzing process to spawn.
41    ///
42    /// Defaults to the number of cores.
43    #[clap(long, short)]
44    pub jobs: Option<usize>,
45
46    /// Maximum number of seconds the checker can run.
47    ///
48    /// If the checker takes longer than this, the fuzzer fails and the corresponding file is
49    /// emitted.
50    #[clap(long, default_value = "2")]
51    pub checker_timeout: usize,
52
53    /// Maximum fuzzing time in seconds.
54    ///
55    /// Halt after fuzzing for this amount of time. Zero should not be used.
56    #[clap(long, default_value = "60")]
57    pub max_time: usize,
58
59    /// Don't print the fuzzer output to the console, but redirect it to a file.
60    #[clap(long)]
61    pub quiet: bool,
62
63    /// Don't run the evaluation for building the output files.
64    #[clap(long)]
65    pub no_build: bool,
66
67    #[clap(flatten, next_help_heading = Some("EXECUTION"))]
68    pub execution: ExecutionOpt,
69
70    #[clap(flatten, next_help_heading = Some("STORAGE"))]
71    pub storage: StorageOpt,
72}

90pub fn main_fuzz_checker(opt: FuzzCheckerOpt) -> Result<(), Error> {
91    let task_format = opt
92        .find_task
93        .find_task(&Default::default())
94        .context("Failed to locate the task")?;
95
96    let task = if let TaskFormat::IOI(task) = &task_format {
97        task
98    } else {
99        bail!("The fuzz-checker tool only supports IOI-tasks for now");
100    };
101
102    let task_type = if let TaskType::Batch(task_type) = &task.task_type {
103        task_type
104    } else {
105        bail!("Only Batch tasks are supported");
106    };
107
108    let checker = if let Checker::Custom(checker) = &task_type.checker {
109        checker
110    } else {
111        bail!("Only tasks with a checker are supported");
112    };
113    let checker_bin_path = checker
114        .write_bin_to
115        .clone()
116        .expect("Missing checker write_bin_to");
117
118    let language = checker.language();
119    if language.name() != "C++" {
120        bail!("Only C++ checkers are supported");
121    }
122
123    let num_testcases: usize = task.subtasks.values().map(|st| st.testcases.len()).sum();
124
125    let task_dir = task.path.clone();
126    let fuzz_data = FuzzData {
127        checker_source: checker.path.clone(),
128        initial_output_files: (0..num_testcases)
129            .map(|i| task_dir.join(format!("output/output{}.txt", i)))
130            .collect(),
131        task_dir,
132        opt,
133    };
134
135    trace!("Fuzz data: {:#?}", fuzz_data);
136
137    if fuzz_data.opt.no_build {
138        for output in &fuzz_data.initial_output_files {
139            if !output.exists() {
140                bail!("The output files haven't been generated, please run task-maker");
141            }
142        }
143    } else {
144        info!("Running task-maker for building the output files");
145
146        let eval_config = EvaluationConfig {
147            solution_filter: vec!["do not evaluate the solutions!!".into()],
148            no_statement: true,
149            ..Default::default()
150        };
151
152        // setup the configuration and the evaluation metadata
153        let context = RuntimeContext::new(task_format, &fuzz_data.opt.execution, |task, eval| {
154            // build the DAG for the task
155            task.build_dag(eval, &eval_config)
156                .context("Cannot build the task DAG")
157        })?;
158        // start the execution
159        let executor =
160            context.connect_executor(&fuzz_data.opt.execution, &fuzz_data.opt.storage)?;
161        let executor = executor.start_ui(&UIType::Silent, |_, _| {})?;
162        executor.execute()?;
163    }
164
165    let fuzz_dir = fuzz_data.task_dir.join(&fuzz_data.opt.fuzz_dir);
166    if !fuzz_dir.exists() {
167        info!("Creating fuzz directory at {}", fuzz_dir.display());
168        std::fs::create_dir_all(&fuzz_dir).with_context(|| {
169            anyhow!("Failed to created fuzz directory at {}", fuzz_dir.display())
170        })?;
171    }
172
173    write_initial_corpus(&fuzz_dir, &fuzz_data)?;
174    let (checker_source, fuzzer_source) = write_checker_source(&fuzz_dir, &fuzz_data)?;
175    let fuzz_binary = compile_fuzzer(&fuzz_dir, &fuzz_data, &checker_source, &fuzzer_source)?;
176    let artifacts = run_fuzzer(&fuzz_dir, &fuzz_data, &fuzz_binary)?;
177    organize_failures(&fuzz_dir, &fuzz_data, &artifacts, &checker_bin_path)?;
178
179    Ok(())
180}

182fn write_initial_corpus(fuzz_dir: &Path, data: &FuzzData) -> Result<(), Error> {
183    info!("Creating initial corpus");
184    let initial_corpus_dir = fuzz_dir.join("initial_corpus");
185    if !initial_corpus_dir.exists() {
186        std::fs::create_dir(&initial_corpus_dir).with_context(|| {
187            anyhow!(
188                "Failed to create initial_corpus directory at {}",
189                initial_corpus_dir.display()
190            )
191        })?;
192    }
193
194    for (index, output) in data.initial_output_files.iter().enumerate() {
195        let path = initial_corpus_dir.join(format!("{}.txt", index));
196        if path.exists() {
197            debug!("Removing old corpus at {}", path.display());
198            std::fs::remove_file(&path)
199                .with_context(|| anyhow!("Failed to remove {}", path.display()))?;
200        }
201
202        let mut file = std::fs::File::create(&path)
203            .with_context(|| anyhow!("Failed to create {}", path.display()))?;
204
205        // write the testcase number as the first 4 bytes of the output file
206        let index_bytes = (index as u32).to_le_bytes();
207        file.write_all(&index_bytes)
208            .with_context(|| anyhow!("Failed to write index to {}", path.display()))?;
209
210        let output_content = std::fs::read(output)
211            .with_context(|| anyhow!("Failed to read {}", output.display()))?;
212
213        // write the rest of the output file
214        file.write_all(&output_content).with_context(|| {
215            anyhow!(
216                "Failed to write {} bytes of output at {}",
217                output_content.len(),
218                path.display()
219            )
220        })?;
221    }
222
223    Ok(())
224}

226fn write_checker_source(fuzz_dir: &Path, data: &FuzzData) -> Result<(PathBuf, PathBuf), Error> {
227    let sources_dir = fuzz_dir.join("sources");
228    std::fs::create_dir_all(&sources_dir)
229        .with_context(|| anyhow!("Failed to create sources dir at {}", sources_dir.display()))?;
230    let path = sources_dir.join("checker-patched.cpp");
231    info!("Writing patched checker source at {}", path.display());
232    let mut checker_file = std::fs::File::create(&path)
233        .with_context(|| anyhow!("Failed to create {}", path.display()))?;
234
235    checker_file
236        .write_all(CHECKER_HEADER)
237        .with_context(|| anyhow!("Failed to write checker header at {}", path.display()))?;
238
239    let mut checker_content = std::fs::read_to_string(&data.checker_source).with_context(|| {
240        anyhow!(
241            "Failed to read checker source at {}",
242            data.checker_source.display()
243        )
244    })?;
245
246    checker_sanity_check(data, &checker_content)?;
247    patch_checker(&mut checker_content).with_context(|| anyhow!("Failed to patch checker"))?;
248
249    checker_file
250        .write_all(checker_content.as_bytes())
251        .with_context(|| {
252            anyhow!(
253                "Failed to write {} bytes of checker at {}",
254                checker_content.len(),
255                path.display()
256            )
257        })?;
258
259    let fuzzer = sources_dir.join("fuzzer.cpp");
260    info!("Writing fuzzer source at {}", fuzzer.display());
261    std::fs::File::create(&fuzzer)
262        .with_context(|| anyhow!("Failed to create {}", fuzzer.display()))?
263        .write_all(FUZZER)
264        .with_context(|| {
265            anyhow!(
266                "Failed to write {} bytes of fuzzer at {}",
267                FUZZER.len(),
268                fuzzer.display()
269            )
270        })?;
271
272    Ok((path, fuzzer))
273}

275fn checker_sanity_check(data: &FuzzData, checker_content: &str) -> Result<(), Error> {
276    let mut command = std::process::Command::new("ctags");
277    command.arg("--output-format=json");
278    command.arg("--c-kinds=v");
279    command.arg("--extras=-F");
280    command.arg(&data.checker_source);
281    command.stdout(Stdio::piped());
282    let output = match command.output() {
283        Ok(output) => output,
284        Err(e) => {
285            warn!(
286                "Failed to execute ctags, you may need to installed it: {:?}",
287                e
288            );
289            return Ok(());
290        }
291    };
292    if !output.status.success() {
293        warn!(
294            "ctags failed to analyze the checker source (exit code {:?})",
295            output.status.code()
296        );
297    }
298
299    // Example output line:
300    //     {"_type": "tag", "name": "H", "path": "cor/correttore.cpp", "pattern": "/^int H[MAXN], rep[MAXN], lep[MAXN], nxt[MAXN], prv[MAXN];$/", "typeref": "typename:int[]", "kind": "variable"}
301    //     {"_type": "tag", "name": "MAXN", "path": "cor/correttore.cpp", "pattern": "/^const int MAXN = 2000005;$/", "typeref": "typename:const int", "kind": "variable"}
302    #[derive(Deserialize)]
303    struct OutputLine {
304        name: String,
305        typeref: String,
306        kind: String,
307    }
308
309    let output = String::from_utf8_lossy(&output.stdout);
310    for line in output.as_ref().lines() {
311        let line = match serde_json::from_str::<OutputLine>(line) {
312            Ok(line) => line,
313            Err(e) => {
314                warn!(
315                    "Failed to deserialize line from ctags: {:?} (line was {})",
316                    e, line
317                );
318                continue;
319            }
320        };
321        // we are only interested in global variables
322        if line.kind != "variable" {
323            continue;
324        }
325        // constants are ok to be global
326        if line.typeref.starts_with("typename:const ") {
327            continue;
328        }
329        // we found a global variable!
330        error!(
331            "Global variable found! '{}' looks like a global variable and it will probably interfere with the fuzzing process", line.name
332        )
333    }
334
335    let re = Regex::new(r".*\bstatic\s[^=;]*=.*").expect("bad regex");
336    for cap in re.captures_iter(checker_content) {
337        let definition = cap[0].trim();
338        if definition.contains("const") {
339            continue;
340        }
341        error!("Static variable found! '{}' looks like a static variable and it will probably interfere with the fuzzing process", definition);
342    }
343
344    Ok(())
345}

347fn patch_checker(source: &mut String) -> Result<(), Error> {
348    info!("Patching checker source file");
349    *source = source.replace("std::exit", "exit");
350    Ok(())

353fn compile_fuzzer(
354    fuzz_dir: &Path,
355    data: &FuzzData,
356    checker_source_path: &Path,
357    fuzzer_source_path: &Path,
358) -> Result<PathBuf, Error> {
359    let fuzzer_dir = fuzz_dir.join("fuzzer");
360    std::fs::create_dir_all(&fuzzer_dir)
361        .with_context(|| anyhow!("Failed to create fuzzer dir at {}", fuzzer_dir.display()))?;
362    let checker_shared_object_path = fuzzer_dir.join("checker.so");
363    info!(
364        "Compiling {} with clang++",
365        checker_shared_object_path.display()
366    );
367
368    let mut command = std::process::Command::new("clang++");
369    command.arg(checker_source_path);
370    command.arg("-shared");
371    command.arg("-fPIC");
372    command.arg("-o");
373    command.arg(&checker_shared_object_path);
374
375    let mut sanitizers = "-fsanitize=fuzzer".to_string();
376    if !data.opt.sanitizers.is_empty() {
377        sanitizers += ",";
378        sanitizers += &data.opt.sanitizers;
379    }
380    command.arg(sanitizers);
381
382    let std_version = std::env::var("TM_CXX_STD_VERSION").unwrap_or_else(|_| "c++17".into());
383    command.arg(format!("-std={}", std_version));
384
385    if data.opt.extra_args.is_empty() {
386        debug!("Adding -O2 and -g since no extra argument has been specified");
387        command.arg("-O2");
388        command.arg("-g");
389    } else {
390        for arg in &data.opt.extra_args {
391            command.arg(arg);
392        }
393    }
394
395    info!("Compiling with: {:?}", command);
396    let status = command
397        .status()
398        .with_context(|| anyhow!("Failed to start the checker compilation with {:?}", command))?;
399    if !status.success() {
400        bail!("Checker compilation failed (exit code {:?})", status.code());
401    }
402
403    let target = fuzzer_dir.join("fuzzer");
404    info!("Compiling {} with clang++", target.display());
405    let mut command = std::process::Command::new("clang++");
406    command.arg(fuzzer_source_path);
407    command.arg("-std=c++17");
408    command.arg("-o");
409    command.arg(&target);
410
411    command.arg(format!("-DNUM_INPUTS={}", data.initial_output_files.len()));
412    command.arg(format!("-DFUZZ_DIRECTORY=\"{}\"", fuzz_dir.display()));
413    command.arg(format!("-DTASK_DIRECTORY=\"{}\"", data.task_dir.display()));
414
415    let mut sanitizers = "-fsanitize=fuzzer".to_string();
416    if !data.opt.sanitizers.is_empty() {
417        sanitizers += ",";
418        sanitizers += &data.opt.sanitizers;
419    }
420    command.arg(sanitizers);
421
422    if data.opt.extra_args.is_empty() {
423        debug!("Adding -O2 and -g since no extra argument has been specified");
424        command.arg("-O2");
425        command.arg("-g");
426    } else {
427        for arg in &data.opt.extra_args {
428            command.arg(arg);
429        }
430    }
431
432    info!("Compiling with: {:?}", command);
433    let status = command
434        .status()
435        .with_context(|| anyhow!("Failed to start the checker compilation with {:?}", command))?;
436    if !status.success() {
437        bail!("Checker compilation failed (exit code {:?})", status.code());
438    }
439
440    Ok(target)
441}

443fn run_fuzzer(fuzz_dir: &Path, data: &FuzzData, fuzzer: &Path) -> Result<Vec<PathBuf>, Error> {
444    let artifacts = fuzz_dir.join("artifacts");
445    if artifacts.exists() {
446        warn!(
447            "Removing existing artifacts directory at {}",
448            artifacts.display()
449        );
450        std::fs::remove_dir_all(&artifacts).with_context(|| {
451            anyhow!("Failed to remove artifacts dir at {}", artifacts.display())
452        })?;
453    }
454    std::fs::create_dir(&artifacts)
455        .with_context(|| anyhow!("Failed to create artifacts dir at {}", artifacts.display()))?;
456
457    // Give unlimited stack to the checker.
458    if let Err(e) = rlimit::Resource::STACK.set(rlimit::INFINITY, rlimit::INFINITY) {
459        warn!("Failed to set ulimit -s unlimited: {:?}", e);
460    }
461
462    let mut command = std::process::Command::new(fuzzer);
463    let jobs = if let Some(jobs) = data.opt.jobs {
464        jobs
465    } else {
466        num_cpus::get()
467    };
468    command.arg(format!("-fork={}", jobs));
469    command.arg(format!("-timeout={}", data.opt.checker_timeout));
470    command.arg(format!("-max_total_time={}", data.opt.max_time));
471    command.arg(fuzz_dir.join("initial_corpus"));
472    command.arg(format!("-artifact_prefix={}/", artifacts.display()));
473    if data.opt.quiet {
474        let stdout = fuzz_dir.join("fuzzer/stdout.txt");
475        debug!("Redirecting stdout to {}", stdout.display());
476        let stdout_file = std::fs::File::create(&stdout)
477            .with_context(|| anyhow!("Failed to create stdout at {}", stdout.display()))?;
478        // SAFETY: the file is constructed and dropped here, it is not shared anywhere
479        let stdout = unsafe { Stdio::from_raw_fd(stdout_file.into_raw_fd()) };
480        command.stdout(stdout);
481
482        let stderr = fuzz_dir.join("fuzzer/stderr.txt");
483        debug!("Redirecting stderr to {}", stderr.display());
484        let stderr_file = std::fs::File::create(&stderr)
485            .with_context(|| anyhow!("Failed to create stderr at {}", stderr.display()))?;
486        let stderr = unsafe { Stdio::from_raw_fd(stderr_file.into_raw_fd()) };
487        command.stderr(stderr);
488    };
489
490    info!("Running fuzzer with: {:?}", command);
491    let status = command
492        .status()
493        .with_context(|| anyhow!("Failed to start the fuzzer with {:?}", command))?;
494    if !status.success() {
495        warn!("Fuzzer failed (exit code {:?})", status.code());
496    }
497
498    let paths = std::fs::read_dir(&artifacts)
499        .with_context(|| {
500            anyhow!(
501                "Failed to list artifacts directory content at {}",
502                artifacts.display()
503            )
504        })?
505        .filter_map(|path| path.ok().map(|p| p.path()))
506        .collect();
507    Ok(paths)
508}

510fn organize_failures(
511    fuzz_dir: &Path,
512    data: &FuzzData,
513    artifacts: &[PathBuf],
514    checker_bin_path: &Path,
515) -> Result<(), Error> {
516    if artifacts.is_empty() {
517        info!("No failure found!");
518        return Ok(());
519    }
520    info!("Reorganizing failure files");
521    let failures = fuzz_dir.join("failures");
522    if failures.exists() {
523        warn!(
524            "Removing existing failures directory at {}",
525            failures.display()
526        );
527        std::fs::remove_dir_all(&failures)
528            .with_context(|| anyhow!("Failed to remove failure dir at {}", failures.display()))?;
529    }
530    std::fs::create_dir(&failures)
531        .with_context(|| anyhow!("Failed to create failures dir at {}", failures.display()))?;
532
533    let mut printer = StdoutPrinter::default();
534    let checker_bin_path = checker_bin_path.to_string_lossy();
535    let readme = README.replace("@@CHECKER@@", &checker_bin_path);
536
537    for (artifact_id, artifact) in artifacts.iter().enumerate() {
538        let mut file = std::fs::File::open(artifact)
539            .with_context(|| anyhow!("Failed to open artifact at {}", artifact.display()))?;
540
541        // read the input id from the first 4 bytes of the output file
542        let mut id = [0u8; 4];
543        file.read_exact(&mut id)
544            .with_context(|| anyhow!("Failed to read testcase id from {}", artifact.display()))?;
545        let id = u32::from_le_bytes(id) % data.initial_output_files.len() as u32;
546
547        // read the actual output file
548        let mut output = vec![];
549        file.read_to_end(&mut output).with_context(|| {
550            anyhow!(
551                "Failed to read the output file from the artifact at {}",
552                artifact.display()
553            )
554        })?;
555
556        // obtain the failure type from the artifact file name (crash-... timeout-... ecc)
557        let artifact_name = artifact.file_name().unwrap();
558        let fail_type = if let Some(fail_type) = artifact_name.to_str().unwrap().split('-').next() {
559            fail_type
560        } else {
561            bail!("Invalid artifact name {}", artifact.display());
562        };
563
564        let target_dir = failures.join(format!("fail-{}", artifact_id));
565        std::fs::create_dir(&target_dir).with_context(|| {
566            anyhow!(
567                "Failed to create artifact output directory at {}",
568                target_dir.display()
569            )
570        })?;
571
572        let failure_path = target_dir.join(format!("output-{}.txt", fail_type));
573        std::fs::write(&failure_path, &output).with_context(|| {
574            anyhow!(
575                "Failed to write {} bytes of output file at {}",
576                output.len(),
577                failure_path.display()
578            )
579        })?;
580
581        let source_input_path = data.task_dir.join(format!("input/input{}.txt", id));
582        let source_correct_path = data.task_dir.join(format!("output/output{}.txt", id));
583        let target_input_path = target_dir.join("input.txt");
584        let target_output_path = target_dir.join("output.txt");
585        let target_artifact_path = target_dir.join("artifact.bin");
586        std::os::unix::fs::symlink(&source_input_path, &target_input_path).with_context(|| {
587            anyhow!(
588                "Failed to create symlink: {} -> {}",
589                target_input_path.display(),
590                source_input_path.display()
591            )
592        })?;
593        std::os::unix::fs::symlink(&source_correct_path, &target_output_path).with_context(
594            || {
595                anyhow!(
596                    "Failed to create symlink: {} -> {}",
597                    target_output_path.display(),
598                    source_correct_path.display()
599                )
600            },
601        )?;
602        std::os::unix::fs::symlink(artifact, &target_artifact_path).with_context(|| {
603            anyhow!(
604                "Failed to create symlink: {} -> {}",
605                target_artifact_path.display(),
606                artifact.display()
607            )
608        })?;
609        #[allow(clippy::needless_borrow)] // https://github.com/rust-lang/rust-clippy/issues/9778
610        std::fs::write(target_dir.join("README"), &readme).context("Failed to write README")?;
611        cwrite!(printer, RED, "[FAIL] {:<8}", fail_type);
612        println!(" {}", target_dir.display());
613    }
614
615    warn!(
616        "{} checker failure written to {}",
