186/// Returns the long-help for the "skip sanity checks" option.
187fn skip_sanity_checks_long_help() -> &'static str {
188    lazy_static! {
189        pub static ref DOC: String = {
190            let mut message = "List of sanity checks to skip.\n\n".to_string();
191            message += "The available checks are:\n";
192
193            let mut by_category: HashMap<_, Vec<_>> = Default::default();
194            for (name, category) in get_sanity_check_list() {
195                by_category.entry(category).or_default().push(name);
196            }
197
198            for (category, mut checks) in by_category.into_iter().sorted() {
199                checks.sort();
200                message += &format!("{} ({})\n", category.as_str(), category.purpose());
201                message += &format!("- {}\n\n", checks.join(", "));
202            }
203
204            message.trim().into()
205        };
206    }
207    &DOC
208}

210/// Returns the long-help for the --no-cache option.
211fn no_cache_long_help() -> &'static str {
212    lazy_static! {
213        pub static ref DOC: String = format!(
214            "Disable the cache for this comma separated list of tags\n\nProviding an empty list will disable all the caches. The supported tags are: {}.",
215            VALID_TAGS.iter().join(", ")
216        );
217    }
218    &DOC
219}

241impl UIOpt {
242    /// Disable the Curses UI and fallback to PrintUI if verbose output is enabled.
243    pub fn disable_if_needed(&mut self, logger: &LoggerOpt) {
244        let mut show_warning = false;
245        if logger.should_diable_curses() {
246            if let task_maker_format::ui::UIType::Curses = self.ui {
247                // warning deferred to after the logger has been initialized
248                show_warning = true;
249                self.ui = task_maker_format::ui::UIType::Print;
250            }
251        }
252        if show_warning {
253            warn!("Do not combine -v with curses ui, bad things will happen! Fallback to print ui");
254        }
255    }
256}

258impl StorageOpt {
259    /// Get the store directory of this configuration. If nothing is specified a cache directory is
260    /// used if available, otherwise a temporary directory.
261    pub fn store_dir(&self) -> PathBuf {
262        match &self.store_dir {
263            Some(dir) => dir.clone(),
264            None => {
265                let project = directories::ProjectDirs::from("", "", "task-maker");
266                if let Some(project) = project {
267                    project.cache_dir().to_owned()
268                } else {
269                    std::env::temp_dir().join("task-maker")
270                }
271            }
272        }
273    }
274}

276impl LoggerOpt {
277    /// Enable the logs according to the specified configuration.
278    pub fn enable_log(&self) {
279        if self.verbose > 0 {
280            std::env::set_var("RUST_BACKTRACE", "1");
281        }
282        match self.verbose {
283            0 => std::env::set_var("RUST_LOG", "warn,tabox=warn"),
284            1 => std::env::set_var("RUST_LOG", "info,tabox=info"),
285            2 => std::env::set_var("RUST_LOG", "debug,tabox=debug"),
286            _ => std::env::set_var("RUST_LOG", "trace,tabox=trace"),
287        }
288
289        env_logger::Builder::from_default_env()
290            .format_timestamp_nanos()
291            .init();
292        better_panic::install();
293    }
294
295    pub fn should_diable_curses(&self) -> bool {
296        self.verbose > 0
297    }
298}
