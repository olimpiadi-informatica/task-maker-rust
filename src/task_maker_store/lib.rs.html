1//! This crate manages the file store on disk, a folder with many files indexed by their hash.
2//!
3//! The files are stored in a read-only manner (removing the write bit permission) and their access
4//! is granted via their hash. The size of the store folder is limited to a specific amount and the
5//! least-recently-used files are removed automatically.
6//!
7//! The access to the store directory via this crate is exclusive even between processes.
8//!
9//! # Example
10//!
11//! Storing a file into the store and getting it back later.
12//!
13//! ```
14//! use task_maker_store::{FileStore, FileStoreKey, ReadFileIterator};
15//!
16//! # use anyhow::Error;
17//! # use std::fs;
18//! # use tempfile::TempDir;
19//! # fn main() -> Result<(), Error> {
20//! # let tmp = TempDir::new().unwrap();
21//! # let store_dir = tmp.path().join("store");
22//! # let path = tmp.path().join("file.txt");
23//! # fs::write(&path, "hello world")?;
24//! // make a new store based on a directory, this will lock if the store is already in use
25//! let store = FileStore::new(store_dir, 1000, 1000)?;
26//! // compute the key of a file and make an iterator over its content
27//! let key = FileStoreKey::from_file(&path)?;
28//! let iter = ReadFileIterator::new(&path)?;
29//! // store the file inside the file store. The file will be kept at least until handle is alive
30//! let handle = store.store(&key, iter)?;
31//! // store.get(&key) will return an handle to the file on disk if present inside the store
32//! assert!(store.get(&key).is_some());
33//! # Ok(())
34//! # }
35//! ```

62/// Whether to check the file integrity on the store before getting it.
63const INTEGRITY_CHECKS_ENABLED: bool = false;
64/// The name of the lock of the file store.
65const STORE_LOCK_FILE: &str = "exclusive.lock";
66/// The name of the index of the file store.
67const STORE_INDEX_FILE: &str = "index.bin";

69/// Container with the ref counts of all the handles still alive.
70#[derive(Debug)]
71struct LockedFiles {
72    /// Map from a `FileStoreKey` to the number of handles alive.
73    ref_counts: HashMap<FileStoreKey, usize>,
74}

76/// A file store will manage all the files in the store directory.
77///
78/// This will manage a file storage directory with the ability of:
79/// * remove files not needed anymore that takes too much space.
80/// * locking so no other instances of `FileStorage` can access the storage while
81///   this is still running, even in other processes.
82/// * do not remove files useful for the current computations.
83#[derive(Debug)]
84pub struct FileStore {
85    /// Base directory of the `FileStore`.
86    base_path: PathBuf,
87    /// Lock to the `FileStore` directory.
88    _lock: LockFile,
89    /// The files locked because there are some handles still alive.
90    locked_files: Arc<Mutex<LockedFiles>>,
91    /// The index with the files known to the store. This is used when flushing the old files.
92    index: Arc<Mutex<FileStoreIndex>>,
93    /// Maximum size of the file store.
94    max_store_size: u64,
95    /// Target size of the file store after the flush.
96    min_store_size: u64,
97}
98
99/// Handle of a file in the `FileStore`, this must be computable given the content of the file, i.e.
100/// an hash of the content.
101#[derive(Clone, PartialEq, Eq, Hash)]
102pub struct FileStoreKey {
103    /// An hash of the content of the file.
104    hash: Hash,
105}

107/// An handle to a specific file inside the store, until this handle is dropped the `FileStore` will
108/// not flush away the file for clearing space. You can clone the handle extending the life of the
109/// file.
110#[derive(Debug)]
111pub struct FileStoreHandle {
112    /// The key of the file inside the store.
113    key: FileStoreKey,
114    /// The path to the file on disk.
115    path: PathBuf,
116    /// A reference to the locked files. Will be used to remove self from the ref counts.
117    locked_files: Arc<Mutex<LockedFiles>>,
118}

120impl FileStore {
121    /// Make a new `FileStore` in the specified base directory, will lock if another instance of a
122    /// `FileStore` is locking the directory. The locking is implemented via platform-specific file
123    /// locking. Having two instances of the file store running concurrently is not safe.
124    ///
125    /// ```
126    /// use task_maker_store::FileStore;
127    ///
128    /// # use anyhow::Error;
129    /// # use std::fs;
130    /// # use tempfile::TempDir;
131    /// # fn main() -> Result<(), Error> {
132    /// # let dir = TempDir::new()?;
133    /// # let store_dir = dir.path();
134    /// // make a new store based on a directory, this will lock if the store is already in use
135    /// // somewhere
136    /// let store = FileStore::new(store_dir, 1000, 1000)?;
137    /// // let store2 = FileStore::new(store_dir) // this will lock!!
138    /// # Ok(())
139    /// # }
140    /// ```
141    pub fn new<P: Into<PathBuf>>(
142        base_path: P,
143        max_store_size: u64,
144        min_store_size: u64,
145    ) -> Result<FileStore, Error> {
146        let base_path = base_path.into();
147        debug!("Opening file store at {}", base_path.display());
148        std::fs::create_dir_all(&base_path).with_context(|| {
149            format!(
150                "Failed to create storage directory at {}",
151                base_path.display()
152            )
153        })?;
154        let lock_path = base_path.join(STORE_LOCK_FILE);
155        let mut lock = LockFile::open(&lock_path)
156            .with_context(|| format!("Failed to create lock file at {}", lock_path.display()))?;
157
158        if !lock
159            .try_lock()
160            .context("Failed to obtain exclusive lock on storage")?
161        {
162            warn!("Store locked... waiting");
163            lock.lock()
164                .context("Failed to obtain exclusive lock on storage")?;
165        }
166
167        let index = FileStoreIndex::load(base_path.join(STORE_INDEX_FILE))
168            .context("Failed to load storage index")?;
169        Ok(FileStore {
170            base_path,
171            _lock: lock,
172            locked_files: Arc::new(Mutex::new(LockedFiles::new())),
173            index: Arc::new(Mutex::new(index)),
174            max_store_size,
175            min_store_size,
176        })
177    }
178
179    /// Given an iterator of `Vec<u8>` consume all of it writing the content to the disk if the file
180    /// is not already present on disk. The file is stored inside the base directory and `chmod -w`.
181    ///
182    /// If the file is already present it is not overwritten but the iterator is consumed
183    /// nevertheless.
184    ///
185    /// Will return an handle to that file, keeping the file alive.
186    ///
187    /// ```
188    /// use task_maker_store::{FileStore, FileStoreKey, ReadFileIterator};
189    ///
190    /// # use anyhow::Error;
191    /// # use std::fs;
192    /// # use tempfile::TempDir;
193    /// # fn main() -> Result<(), Error> {
194    /// # let tmp = TempDir::new().unwrap();
195    /// # let store_dir = tmp.path().join("store");
196    /// # let path = tmp.path().join("file.txt");
197    /// # fs::write(&path, "hello world")?;
198    /// let store = FileStore::new(store_dir, 1000, 1000)?;
199    /// // compute the key of a file and make an iterator over its content
200    /// let key = FileStoreKey::from_file(&path)?;
201    /// let iter = ReadFileIterator::new(&path)?;
202    /// // store the file inside the file store. The file will be kept at least until handle is alive
203    /// let handle = store.store(&key, iter)?;
204    /// // here it's guaranteed that the file won't be flushed away
205    /// assert!(handle.path().exists());
206    /// drop(handle);
207    /// // here it's not guaranteed
208    /// # Ok(())
209    /// # }
210    /// ```
211    pub fn store<I>(&self, key: &FileStoreKey, content: I) -> Result<FileStoreHandle, Error>
212    where
213        I: IntoIterator<Item = Vec<u8>>,
214    {
215        let path = self.key_to_path(key);
216        trace!("Storing {:?}", path);
217        // make the key to avoid racing while writing
218        let handle = FileStoreHandle::new(self, key);
219        if path.exists() {
220            trace!("File {:?} already exists", path);
221            content.into_iter().last(); // consume all the iterator
222        } else {
223            // assuming moving files is atomic this should be MT-safe
224            let dir = path.parent().unwrap();
225            std::fs::create_dir_all(dir)
226                .with_context(|| format!("Cannot create directory at {}", dir.display()))?;
227            let tmpdir = tempfile::TempDir::new_in(path.parent().unwrap())
228                .context("Failed to create temporary directory for storing the file")?;
229            let tmpfile_path = tmpdir.path().join("file");
230            let mut tmpfile =
231                std::fs::File::create(&tmpfile_path).context("Failed to create temporary file")?;
232            if !content
233                .into_iter()
234                .map(|data| tmpfile.write_all(&data))
235                .all(|r| r.is_ok())
236            {
237                bail!("Failed to store file");
238            }
239            std::fs::rename(&tmpfile_path, &path).with_context(|| {
240                format!(
241                    "Failed to rename {} -> {}",
242                    tmpfile_path.display(),
243                    path.display()
244                )
245            })?;
246            FileStore::mark_readonly(&path).context("Failed to mark file as readonly")?;
247            {
248                let mut index = self.index.lock().unwrap();
249                index
250                    .add(key.clone(), path)
251                    .context("Failed to add file to index")?;
252                self.maybe_flush(&mut index)?;
253                // FIXME: maybe this can be done less frequently
254                index
255                    .store(self.base_path.join(STORE_INDEX_FILE))
256                    .context("Failed to store the index to file")?;
257            }
258        }
259        Ok(handle)
260    }

262    /// Returns an handle to the file with that key or `None` if it's not in the
263    /// [`FileStore`](struct.FileStore.html).
264    ///
265    /// This requires mutability because it will actively fix any corrupted or missing files in the
266    /// store.
267    ///
268    /// The file is guaranteed to not be flushed until all the handles to it get dropped.
269    ///
270    /// ```
271    /// use task_maker_store::{FileStore, FileStoreKey, ReadFileIterator};
272    ///
273    /// # use anyhow::Error;
274    /// # use std::fs;
275    /// # use tempfile::TempDir;
276    /// # fn main() -> Result<(), Error> {
277    /// # let tmp = TempDir::new().unwrap();
278    /// # let store_dir = tmp.path().join("store");
279    /// # let path = tmp.path().join("file.txt");
280    /// # fs::write(&path, "hello world")?;
281    /// let store = FileStore::new(store_dir, 1000, 1000)?;
282    /// let key = FileStoreKey::from_file(&path)?;
283    /// # let iter = ReadFileIterator::new(&path)?;
284    /// # let handle = store.store(&key, iter)?;
285    /// let handle = store.get(&key);
286    /// match handle {
287    ///     None => panic!("The file is gone!"),
288    ///     Some(handle) => assert!(handle.path().exists())
289    /// }
290    /// # Ok(())
291    /// # }
292    /// ```
293    pub fn get(&self, key: &FileStoreKey) -> Option<FileStoreHandle> {
294        let path = self.key_to_path(key);
295        if !path.exists() {
296            return None;
297        }
298        if INTEGRITY_CHECKS_ENABLED && !self.check_integrity(key) {
299            warn!("File {:?} failed the integrity check", path);
300            if let Err(e) = FileStore::remove_file(&path) {
301                warn!("Cannot remove corrupted file: {:?}", e);
302            }
303            return None;
304        }
305        {
306            let mut index = self.index.lock().unwrap();
307            index.touch(key);
308        }
309        Some(FileStoreHandle::new(self, key))
310    }

312    /// Path of the file to disk.
313    fn key_to_path(&self, key: &FileStoreKey) -> PathBuf {
314        self.base_path.join(key.suffix())
315    }

317    /// Mark a file as readonly.
318    fn mark_readonly(path: &Path) -> Result<(), Error> {
319        let mut perms = std::fs::metadata(path)
320            .with_context(|| format!("Failed to get file metadata of {}", path.display()))?
321            .permissions();
322        perms.set_readonly(true);
323        std::fs::set_permissions(path, perms)
324            .with_context(|| format!("Failed to set permission of {}", path.display()))?;
325        Ok(())
326    }
327
328    /// Remove a file from disk.
329    fn remove_file(path: &Path) -> Result<(), Error> {
330        let mut perms = std::fs::metadata(path)
331            .with_context(|| format!("Failed to get file metadata of {}", path.display()))?
332            .permissions();
333        perms.set_mode(0o600);
334        std::fs::set_permissions(path, perms)
335            .with_context(|| format!("Failed to set permission of {}", path.display()))?;
336        std::fs::remove_file(path)
337            .with_context(|| format!("Failed to remove {}", path.display()))?;
338        Ok(())
339    }

341    /// Check if the file is not corrupted.
342    fn check_integrity(&self, key: &FileStoreKey) -> bool {
343        let path = self.key_to_path(key);
344        let metadata = std::fs::metadata(&path);
345        // if the last modified time is the same of creation time assume it's
346        // not corrupted
347        if let Ok(metadata) = metadata {
348            let created = metadata.created();
349            let modified = metadata.modified();
350            if let (Ok(created), Ok(modified)) = (created, modified) {
351                if created == modified {
352                    return true;
353                }
354            }
355        }
356        match FileStoreKey::from_file(&path) {
357            Ok(key2) => key2.hash == key.hash,
358            Err(_) => false,
359        }
360    }
361
362    /// Check if the file store needs flushing, and do so if needed.
363    fn maybe_flush(&self, index: &mut FileStoreIndex) -> Result<(), Error> {
364        if index.need_flush(self.max_store_size) {
365            let locked = self.locked_files.lock().unwrap();
366            index
367                .flush(self, &locked, self.min_store_size)
368                .context("Failed to flush index")?;
369        }
370        Ok(())
371    }
372}

374impl Drop for FileStore {
375    fn drop(&mut self) {
376        match self.index.lock() {
377            Ok(mut index) => {
378                let locked = match self.locked_files.lock() {
379                    Ok(l) => l,
380                    Err(_) => {
381                        warn!("Cannot lock locked_files due to poison");
382                        return;
383                    }
384                };
385                if index.need_flush(self.max_store_size) {
386                    if let Err(e) = index.flush(self, &locked, self.min_store_size) {
387                        warn!("Cannot flush the index: {}", e.to_string());
388                    }
389                }
390                if let Err(e) = index.store(self.base_path.join(STORE_INDEX_FILE)) {
391                    warn!("Cannot store the index: {}", e.to_string());
392                }
393            }
394            Err(_) => {
395                warn!("Cannot store the index due to poisoned lock");
396            }
397        }
398    }
399}

401impl FileStoreKey {
402    /// Get the suffix of the path of this `FileStoreKey`. For example, if the key is
403    /// `aabbccddeeff...` this method will return `aa/bb/aabbccddeeff...`
404    fn suffix(&self) -> PathBuf {
405        let hash_hex = self.hash.to_hex();
406        let full = hash_hex.as_str();
407        #[allow(clippy::string_slice)]
408        let first = &full[0..2];
409        #[allow(clippy::string_slice)]
410        let second = &full[2..4];
411        PathBuf::from(first).join(second).join(full)
412    }
413
414    /// Make a new `FileStoreKey` from a file on disk. The file must exist and be readable.
415    pub fn from_file<P: AsRef<Path>>(path: P) -> Result<FileStoreKey, Error> {
416        let path = path.as_ref();
417        let mut hasher = Hasher::new();
418        if !path.exists() {
419            bail!("Cannot read {}, maybe broken symlink?", path.display())
420        }
421        let file_reader = ReadFileIterator::new(path)
422            .with_context(|| format!("Cannot make file iterator of {}", path.display()))?;
423        file_reader
424            .map(|buf| {
425                hasher.update(&buf);
426            })
427            .last();
428        Ok(FileStoreKey {
429            hash: hasher.finalize(),
430        })
431    }
432
433    /// Make a new `FileStoreKey` from an in-memory file.
434    pub fn from_content(content: &[u8]) -> FileStoreKey {
435        FileStoreKey {
436            hash: hash(content),
437        }
438    }
439}

486impl FileStoreHandle {
487    /// Make a new handle to a file on disk.
488    fn new(store: &FileStore, key: &FileStoreKey) -> FileStoreHandle {
489        let path = store.key_to_path(key);
490        let mut locked_files = store.locked_files.lock().unwrap();
491        *locked_files.ref_counts.entry(key.clone()).or_default() += 1;
492        FileStoreHandle {
493            path,
494            locked_files: store.locked_files.clone(),
495            key: key.clone(),
496        }
497    }
498
499    /// The path to the file pointed by this handle.
500    pub fn path(&self) -> &Path {
501        &self.path
502    }
503
504    /// The key of the file pointed by this handle.
505    pub fn key(&self) -> &FileStoreKey {
506        &self.key
507    }
508}

535impl Drop for FileStoreHandle {
536    fn drop(&mut self) {
537        let mut locked_files = match self.locked_files.lock() {
538            Ok(guard) => guard,
539            Err(_) => return, // may happen if the thread panicked
540        };
541        *locked_files
542            .ref_counts
543            .get_mut(&self.key)
544            .expect("Ref counts are broken") -= 1;
545        if locked_files.ref_counts[&self.key] == 0 {
546            locked_files.ref_counts.remove(&self.key);
547        }
548    }
549}
