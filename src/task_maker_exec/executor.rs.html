28/// List of the _interesting_ files and executions, only the callbacks listed here will be called by
29/// the server. Every other callback is not sent to the client for performance reasons.
30#[derive(Debug, Clone, Serialize, Deserialize, Default)]
31pub struct ExecutionDAGWatchSet {
32    /// Set of the handles of the executions that have at least a callback bound.
33    pub executions: HashSet<ExecutionUuid>,
34    /// Set of the handles of the files that have at least a callback bound.
35    pub files: HashSet<FileUuid>,
36    /// Set of the handles of the files that should be sent to the client as soon as possible. The
37    /// others will be sent at the end of the evaluation. Note that sending big files during the
38    /// evaluation can cause performance degradations.
39    pub urgent_files: HashSet<FileUuid>,
40}

100/// Message telling the executor that a new client connected or a new worker connected. The handling
101/// of the new peer is done by this executor.
102pub enum ExecutorInMessage {
103    /// A new client has connected, the executor starts listening for the messages and will directly
104    /// interact with it.
105    ClientConnected {
106        /// The information about the new client.
107        client: ClientInfo,
108        /// A channel for sending messages to the client.
109        sender: ChannelSender<ExecutorServerMessage>,
110        /// A channel for received the messages from the client.
111        receiver: ChannelReceiver<ExecutorClientMessage>,
112    },
113    /// A new worker has connected, the executor starts listening for the messages and will directly
114    /// interact with it.
115    WorkerConnected {
116        /// The information and connection details of the worker.
117        worker: WorkerConn,
118    },
119}

121/// The `Executor` is the main component of the server, this will listen for client and worker
122/// connections, handing them by listening to their messages. The clients will send the DAGs to the
123/// `Executor`, which will use its scheduler for executing the jobs. The workers will be attached
124/// to the `WorkerManager` which is being used by the `Scheduler` for assigning the jobs.
125pub(crate) struct Executor {
126    /// The file store used by the Scheduler and the WorkerManager for the local keeping of the
127    /// files.
128    file_store: Arc<FileStore>,
129    /// The `Cache` the Scheduler will use.
130    cache: Cache,
131    /// The receiver of the messages for the `Executor`. The actual `LocalExecutor`/`RemoteExecutor`
132    /// use this channel for the communication.
133    receiver: Receiver<ExecutorInMessage>,
134    /// Whether this executor is running for more than a single client (aka not locally). When this
135    /// flag is set to false, after the first client is done the Scheduler, the WorkerManager and
136    /// this Executor will exit.
137    long_running: bool,
138}

140impl Executor {
141    /// Create a new `Executor` using the specified `FileStore` for the Scheduler and WorkerManager,
142    /// the receiver for communicating with this Executor and if it should be "long running".
143    /// When this flag is set to false, after the first client is done the Scheduler, the
144    /// WorkerManager and this Executor will exit.
145    pub fn new(
146        file_store: Arc<FileStore>,
147        cache: Cache,
148        receiver: Receiver<ExecutorInMessage>,
149        long_running: bool,
150    ) -> Executor {
151        Executor {
152            file_store,
153            cache,
154            receiver,
155            long_running,
156        }
157    }

159    /// Run the `Executor`, listening for client and worker connections. This will block until the
160    /// first client is done (if `long_running` is false) or until the scheduler is stopped.
161    pub fn run(self) -> Result<(), Error> {
162        let (scheduler_tx, scheduler_rx) = channel();
163        let (worker_manager_tx, worker_manager_rx) = channel();
164        let (sched_executor_tx, sched_executor_rx) = channel();
165
166        let clients = Arc::new(Mutex::new(HashMap::new()));
167
168        let scheduler = Scheduler::new(
169            self.file_store.clone(),
170            self.cache,
171            scheduler_rx,
172            sched_executor_tx,
173            worker_manager_tx.clone(),
174        );
175        let worker_manager = WorkerManager::new(
176            self.file_store.clone(),
177            scheduler_tx.clone(),
178            worker_manager_tx.clone(),
179            worker_manager_rx,
180        );
181        let scheduler_thread = thread::Builder::new()
182            .name("Scheduler thread".to_string())
183            .spawn(move || scheduler.run())
184            .expect("Failed to spawn scheduler");
185        let worker_manager_thread = thread::Builder::new()
186            .name("Worker Manager thread".to_string())
187            .spawn(move || worker_manager.run())
188            .expect("Failed to spawn worker manager");
189        let clients2 = clients.clone();
190        let scheduler_binder_thread = thread::Builder::new()
191            .name("Scheduler binder".to_string())
192            .spawn(move || Executor::handle_scheduler_messages(sched_executor_rx, clients2))
193            .expect("Failed to spawn scheduler binder");
194
195        while let Ok(message) = self.receiver.recv() {
196            match message {
197                ExecutorInMessage::ClientConnected {
198                    client,
199                    sender,
200                    receiver,
201                } => {
202                    {
203                        let mut clients = clients.lock().unwrap();
204                        clients.insert(client.uuid, sender.clone());
205                    }
206                    let scheduler = scheduler_tx.clone();
207                    let file_store = self.file_store.clone();
208                    let long_running = self.long_running;
209                    // handle the new client in a new thread called "Client Manager"
210                    // FIXME: this thread is leaked, maybe we can join it as well
211                    thread::Builder::new()
212                        .name(format!(
213                            "Client manager for {} ({})",
214                            client.name, client.uuid
215                        ))
216                        .spawn(move || -> Result<(), Error> {
217                            Executor::handle_client_messages(
218                                file_store,
219                                client,
220                                sender,
221                                receiver,
222                                scheduler.clone(),
223                            )
224                            .unwrap();
225                            // if not in long running mode, the first client should tear down the
226                            // executor. To do so it's just required to tell the scheduler to exit,
227                            // it will bring down the WorkerManager and all should exit.
228                            if !long_running {
229                                scheduler
230                                    .send(SchedulerInMessage::Exit)
231                                    .map_err(|e| anyhow!("Cannot stop the scheduler: {:?}", e))
232                                    .unwrap();
233                            }
234                            Ok(())
235                        })
236                        .expect("Failed to spawn client manager");
237                }
238                ExecutorInMessage::WorkerConnected { worker } => {
239                    worker_manager_tx
240                        .send(WorkerManagerInMessage::WorkerConnected { worker })
241                        .map_err(|e| anyhow!("Cannot send WorkerConnected: {:?}", e))?;
242                }
243            }
244        }
245        debug!("Executor no longer waits for clients/workers");
246
247        scheduler_thread
248            .join()
249            .map_err(|e| anyhow!("Scheduler thread panicked: {:?}", e))?
250            .context("Scheduler thread failed")?;
251        worker_manager_thread
252            .join()
253            .map_err(|e| anyhow!("Worker manager panicked: {:?}", e))?
254            .context("Worker manager thread failed")?;
255        scheduler_binder_thread
256            .join()
257            .map_err(|e| anyhow!("Scheduler binder panicked: {:?}", e))?
258            .context("Scheduler binder thread failed")?;
259        Ok(())
260    }

262    /// Handle the messages from the scheduler, sending the notifications to the client involved.
263    #[allow(clippy::unnecessary_wraps)]
264    fn handle_scheduler_messages(
265        receiver: Receiver<SchedulerExecutorMessage>,
266        clients: Arc<Mutex<HashMap<ClientUuid, ChannelSender<ExecutorServerMessage>>>>,
267    ) -> Result<(), Error> {
268        let mut ready_files: HashMap<ClientUuid, Vec<(FileUuid, FileStoreHandle, bool)>> =
269            HashMap::new();
270        while let Ok((client_uuid, message)) = receiver.recv() {
271            let clients = clients.lock().unwrap();
272            let client = if let Some(client) = clients.get(&client_uuid) {
273                client
274            } else {
275                // ignore messages for a disconnected client
276                continue;
277            };
278            let message = match message {
279                SchedulerExecutorMessageData::ExecutionStarted { execution, worker } => {
280                    ExecutorServerMessage::NotifyStart(execution, worker)
281                }
282                SchedulerExecutorMessageData::ExecutionSkipped { execution } => {
283                    ExecutorServerMessage::NotifySkip(execution)
284                }
285                SchedulerExecutorMessageData::ExecutionDone { execution, result } => {
286                    ExecutorServerMessage::NotifyDone(execution, result)
287                }
288                SchedulerExecutorMessageData::FileReady {
289                    file,
290                    handle,
291                    successful,
292                    urgent,
293                } => {
294                    if urgent {
295                        if let Err(e) =
296                            client.send(ExecutorServerMessage::ProvideFile(file, successful))
297                        {
298                            warn!("Failed to send urgent file: {:?}", e);
299                        } else if let Err(e) = ChannelFileSender::send(handle.path(), client) {
300                            warn!("Failed to send urgent file content: {:?}", e);
301                        }
302                    } else {
303                        ready_files
304                            .entry(client_uuid)
305                            .or_default()
306                            .push((file, handle, successful));
307                    }
308                    continue;
309                }
310                SchedulerExecutorMessageData::Status { status } => {
311                    ExecutorServerMessage::Status(status)
312                }
313                SchedulerExecutorMessageData::EvaluationDone => {
314                    let files = ready_files
315                        .remove(&client_uuid)
316                        .unwrap_or_default()
317                        .into_iter()
318                        .map(|(f, h, s)| (f, h.key().clone(), s))
319                        .collect();
320                    ExecutorServerMessage::Done(files)
321                }
322            };
323            if let Err(e) = client.send(message) {
324                warn!("Failed to send message to the client: {:?}", e);
325            }
326        }
327        debug!("Scheduler binder exiting");
328        Ok(())
329    }

331    /// Handle the messages from a client.
332    fn handle_client_messages(
333        file_store: Arc<FileStore>,
334        client: ClientInfo,
335        sender: ChannelSender<ExecutorServerMessage>,
336        receiver: ChannelReceiver<ExecutorClientMessage>,
337        scheduler: Sender<SchedulerInMessage>,
338    ) -> Result<(), Error> {
339        let mut scheduler = Some(scheduler);
340        while let Ok(message) = receiver.recv() {
341            match message {
342                ExecutorClientMessage::Evaluate { dag, callbacks } => {
343                    if let Err(e) = check_dag(&dag, &callbacks) {
344                        warn!("Invalid DAG: {:?}", e);
345                        sender
346                            .send(ExecutorServerMessage::Error(e.to_string()))
347                            .context("Failed to send Error message to client")?;
348                        break;
349                    } else {
350                        trace!("DAG looks valid!");
351                    }
352                    // for each file marked as provided check if a local copy is present, otherwise
353                    // ask the client to send it.
354                    let mut ready_files = Vec::new();
355                    for (uuid, file) in dag.provided_files.iter() {
356                        let key = match file {
357                            ProvidedFile::Content { key, .. } => key,
358                            ProvidedFile::LocalFile { key, .. } => key,
359                        };
360                        let handle = file_store.get(key);
361                        if let Some(handle) = handle {
362                            ready_files.push((*uuid, handle));
363                        } else {
364                            sender
365                                .send(ExecutorServerMessage::AskFile(*uuid))
366                                .with_context(|| format!("Failed to send AskFile({})", uuid))?;
367                        }
368                    }
369                    // tell the scheduler that a new DAG is ready to be executed.
370                    scheduler
371                        .as_ref()
372                        .ok_or_else(|| anyhow!("Stopped execution"))?
373                        .send(SchedulerInMessage::EvaluateDAG {
374                            client: client.clone(),
375                            dag,
376                            callbacks,
377                        })
378                        .context("Failed to send EvaluateDAG to the scheduler")?;
379                    // tell the scheduler the files that are already locally ready. The others will
380                    // be ready when the client will send them.
381                    for (uuid, handle) in ready_files.into_iter() {
382                        scheduler
383                            .as_ref()
384                            .ok_or_else(|| anyhow!("Stopped execution"))?
385                            .send(SchedulerInMessage::FileReady {
386                                client: client.uuid,
387                                uuid,
388                                handle,
389                            })
390                            .context("Failed to send FileReady to the scheduler")?;
391                    }
392                }
393                ExecutorClientMessage::ProvideFile(uuid, key) => {
394                    info!("Client provided file {}", uuid);
395                    // the client provided a file that was not present locally, store it and tell
396                    // the scheduler that it's now ready.
397                    let handle = file_store
398                        .store(&key, ChannelFileIterator::new(&receiver))
399                        .with_context(|| {
400                            format!("Failed to store client provided file {} ({})", uuid, key)
401                        })?;
402                    scheduler
403                        .as_ref()
404                        .ok_or_else(|| anyhow!("Stopped execution"))?
405                        .send(SchedulerInMessage::FileReady {
406                            client: client.uuid,
407                            uuid,
408                            handle,
409                        })
410                        .context("Failed to send FileReady to the scheduler")?;
411                }
412                ExecutorClientMessage::AskFile(uuid, key, success) => {
413                    info!("Client asking file {:?}", key);
414                    // the client wants to know a file that was produced by the computation, send it
415                    // if it exists.
416                    if let Some(handle) = file_store.get(&key) {
417                        sender
418                            .send(ExecutorServerMessage::ProvideFile(uuid, success))
419                            .context("Failed to send ProvideFile to the client")?;
420                        ChannelFileSender::send(handle.path(), &sender).with_context(|| {
421                            format!("Failed to send file {} to the client", handle)
422                        })?;
423                    } else {
424                        sender
425                            .send(ExecutorServerMessage::Error(format!(
426                                "Unknown file {:?}",
427                                key
428                            )))
429                            .context("Failed to send Error to the client")?;
430                    }
431                }
432                ExecutorClientMessage::Status => {
433                    info!("Client asking for the status");
434                    // This may fail is the scheduler is gone.
435                    if let Some(scheduler) = scheduler.as_ref() {
436                        let _ = scheduler.send(SchedulerInMessage::Status {
437                            client: client.uuid,
438                        });
439                    }
440                }
441                ExecutorClientMessage::Stop => {
442                    info!("Client asking to stop");
443                    if let Some(scheduler) = scheduler.take() {
444                        scheduler
445                            .send(SchedulerInMessage::ClientDisconnected {
446                                client: client.uuid,
447                            })
448                            .context("Failed to send ClientDisconnected to the scheduler")?
449                    }
450                }
451            }
452        }
453        if let Some(scheduler) = scheduler.take() {
454            scheduler
455                .send(SchedulerInMessage::ClientDisconnected {
456                    client: client.uuid,
457                })
458                .context("Failed to send ClientDisconnected to the scheduler")?;
459        }
460        Ok(())
461    }
