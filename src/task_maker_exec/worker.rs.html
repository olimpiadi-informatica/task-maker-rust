24/// The information about the current job the worker is doing.
25struct WorkerCurrentJob {
26    /// Job currently waiting for, when there is a job running this should be `None`
27    current_job: Option<(Box<WorkerJob>, HashMap<FileUuid, FileStoreHandle>)>,
28    /// The currently running sandbox.
29    current_sandboxes: Option<Vec<Sandbox>>,
30    /// The dependencies that are missing and required for the execution start.

32    /// Send to the sandbox_manager the list of files the server is missing.

36/// The worker is the component that receives the work from the server and sends the results back.
37/// It computes the results by executing a process inside a sandbox, limiting the available
38/// resources and measuring the used ones.
39pub struct Worker {
40    /// The identifier of this worker.
41    uuid: WorkerUuid,
42    /// The name of this worker.
43    name: String,
44    /// The channel that sends messages to the server.
45    sender: ChannelSender<WorkerClientMessage>,
46    /// The channel that receives messages from the server.
47    receiver: ChannelReceiver<WorkerServerMessage>,
48    /// A reference to the [`FileStore`](../task_maker_store/struct.FileStore.html).
49    file_store: Arc<FileStore>,
50    /// Job the worker is currently working on.
51    current_job: Arc<Mutex<WorkerCurrentJob>>,
52    /// Where to put the sandboxes.
53    sandbox_path: PathBuf,
54    /// The function that spawns an actual sandbox.
55    sandbox_runner: Arc<dyn SandboxRunner>,
56    /// The join handle of the currently running sandbox, if any.
57    current_sandbox_thread: Option<JoinHandle<()>>,
58}

166    /// Wait for the sandbox thread to exit.
167    fn wait_sandbox(&mut self) -> Result<(), Error> {
168        if let Some(join_handle) = self.current_sandbox_thread.take() {
169            join_handle
170                .join()
171                .map_err(|e| anyhow!("Sandbox thread panicked: {:?}", e))
172                .context("Sandbox thread failed")?;
173        }
174        Ok(())
175    }

177    /// The worker body, this function will block until the worker disconnects.
178    #[allow(clippy::cognitive_complexity)]
179    pub fn work(mut self) -> Result<(), Error> {
180        trace!("Worker {} ready, asking for work", self);
181        self.sender
182            .send(WorkerClientMessage::GetWork)
183            .context("Failed to send GetWork")?;
184
185        loop {
186            match self.receiver.recv() {
187                Ok(WorkerServerMessage::Work(job)) => {
188                    trace!("Worker {} got job: {:?}", self, job);
189                    assert!(self.current_job.lock().unwrap().current_job.is_none());
190                    self.wait_sandbox()?;
191                    let mut missing_deps: HashMap<FileStoreKey, Vec<FileUuid>> = HashMap::new();
192                    let mut handles = HashMap::new();
193                    for exec in &job.group.executions {
194                        for input in exec.dependencies().iter() {
195                            let key = job
196                                .dep_keys
197                                .get(input)
198                                .ok_or(WorkerError::MissingDependencyKey { uuid: *input })?;
199                            match self.file_store.get(key) {
200                                None => {
201                                    // ask the file only once
202                                    if !missing_deps.contains_key(key) {
203                                        self.sender
204                                            .send(WorkerClientMessage::AskFile(key.clone()))
205                                            .context("Failed to send AskFile to server")?;
206                                    }
207                                    missing_deps.entry(key.clone()).or_default().push(*input);
208                                }
209                                Some(handle) => {
210                                    handles.insert(*input, handle);
211                                }
212                            }
213                        }
214                    }
215                    let job_ready = missing_deps.is_empty();
216                    {
217                        let mut current_job = self.current_job.lock().unwrap();
218                        current_job.missing_deps = missing_deps;
219                        current_job.current_job = Some((job, handles));
220                    }
221                    if job_ready {
222                        self.start_job()?;
223                    }
224                }
225                Ok(WorkerServerMessage::ProvideFile(key)) => {
226                    info!("Server sent file {:?}", key);
227                    let reader = ChannelFileIterator::new(&self.receiver);
228                    let handle = self
229                        .file_store
230                        .store(&key, reader)
231                        .with_context(|| format!("Failed to store server-provided file {}", key))?;
232                    let should_start = {
233                        let mut job = self.current_job.lock().unwrap();
234                        let uuids = job
235                            .missing_deps
236                            .remove(&key)
237                            .ok_or_else(|| anyhow!("Server sent a not required dependency"))?;
238                        for uuid in uuids {
239                            job.current_job
240                                .as_mut()
241                                .ok_or_else(|| anyhow!("Received file while doing nothing"))?
242                                .1
243                                .insert(uuid, handle.clone());
244                        }
245                        job.missing_deps.is_empty()
246                    };
247                    if should_start {
248                        self.start_job()?;
249                    }
250                }
251                Ok(WorkerServerMessage::Exit) => {
252                    info!("Worker {} ({}) is asked to exit", self.name, self.uuid);
253                    break;
254                }
255                Ok(WorkerServerMessage::KillJob(job)) => {
256                    let current_job = self.current_job.lock().unwrap();
257                    if let Some((worker_job, _)) = current_job.current_job.as_ref() {
258                        // check that the job is the same
259                        if worker_job.group.uuid == job {
260                            if let Some(sandboxes) = current_job.current_sandboxes.as_ref() {
261                                // ask the sandbox to kill the process
262                                for sandbox in sandboxes {
263                                    sandbox.kill();
264                                }
265                            }
266                        }
267                    }
268                }
269                Ok(WorkerServerMessage::AskFiles(files)) => {
270                    let mut current_job = self.current_job.lock().unwrap();
271                    if let Some(sender) = current_job.server_asked_files.take() {
272                        if let Err(e) = sender.send(files) {
273                            error!("Cannot send the list of files from the server to the worker manager: {:?}", e);
274                        }
275                    } else {
276                        error!("Unexpected WorkerServerMessage::AskFiles");
277                    }
278                }
279                Err(e) => {
280                    let cause = e.root_cause().to_string();
281                    if cause == "receiving on an empty and disconnected channel" {
282                        trace!("Connection closed: {}", cause);
283                    } else {
284                        error!("Connection error: {}", cause);
285                    }
286                    if let Some(sandboxes) =
287                        self.current_job.lock().unwrap().current_sandboxes.as_ref()
288                    {
289                        for sandbox in sandboxes {
290                            sandbox.kill();
291                        }
292                    }
293                    break;
294                }
295            }
296        }
297        {
298            // make sure that the worker doesn't wait for the server's response
299            let mut current_job = self.current_job.lock().unwrap();
300            current_job.server_asked_files.take();
301        }
302        self.wait_sandbox()?;
303        Ok(())
304    }
305}

307/// Spawn a new thread that will start the sandbox and will send the results back to the server.
308fn execute_job(
309    current_job: Arc<Mutex<WorkerCurrentJob>>,
310    sender: &ChannelSender<WorkerClientMessage>,
311    sandbox_path: &Path,
312    runner: Arc<dyn SandboxRunner>,
313) -> Result<JoinHandle<()>, Error> {
314    let (job, sandboxes, fifo_dir, server_asked_files) = {
315        let mut current_job = current_job.lock().unwrap();
316        let job = current_job
317            .current_job
318            .as_ref()
319            .ok_or_else(|| anyhow!("Worker job is gone"))?;
320        let mut boxes = Vec::new();
321        let group = &job.0.group;
322        let fifo_dir = if group.fifo.is_empty() {
323            None
324        } else {
325            let fifo_dir = TempDir::new_in(sandbox_path).with_context(|| {
326                format!(
327                    "Failed to create temporary directory in {}",
328                    sandbox_path.display()
329                )
330            })?;
331            for fifo in &group.fifo {
332                let path = fifo_dir
333                    .path()
334                    .join(fifo.sandbox_path().file_name().unwrap());
335                nix::unistd::mkfifo(&path, nix::sys::stat::Mode::S_IRWXU)
336                    .with_context(|| format!("Failed to create FIFO at {}", path.display()))?;
337            }
338            Some(fifo_dir)
339        };
340        let keep_sandboxes = group.config().keep_sandboxes;
341        for exec in &group.executions {
342            let mut sandbox = Sandbox::new(
343                sandbox_path,
344                exec,
345                &job.1,
346                fifo_dir.as_ref().map(|d| d.path().to_owned()),
347            )?;
348            if keep_sandboxes {
349                sandbox.keep()?;
350            }
351            boxes.push(sandbox);
352        }
353        let job = job.0.clone();
354        current_job.current_sandboxes = Some(boxes.clone());
355        let (sender, receiver) = channel();
356        current_job.server_asked_files = Some(sender);
357        (job, boxes, fifo_dir, receiver)
358    };
359    let sender = sender.clone();
360    let description = job.group.description.clone();
361    let join_handle = std::thread::Builder::new()
362        .name(format!("Sandbox group manager for {}", description))
363        .spawn(move || {
364            sandbox_group_manager(
365                current_job,
366                *job,
367                sender,
368                server_asked_files,
369                sandboxes,
370                runner,
371                fifo_dir,
372            )
373            .with_context(|| format!("Sandbox group for {} failed", description))
374            // FIXME: find a better way to propagate the error to the server
375            .unwrap();
376        })?;
377    Ok(join_handle)
378}

380/// The sandbox group manager spawns the threads of the sandbox of all the executions in the group.
381/// Then waits for their outcome and eventually stops the sandboxes if a process fails. When all the
382/// sandboxes complete, this manager collects their results and send them back to the server.
383///
384/// Note that this function owns `fifo_dir`, the `TempDir` where the FIFOs are stored, it has not to
385/// be dropped before all the sandboxes end.
386fn sandbox_group_manager(
387    current_job: Arc<Mutex<WorkerCurrentJob>>,
388    job: WorkerJob,
389    sender: ChannelSender<WorkerClientMessage>,
390    server_asked_files_receiver: Receiver<Vec<FileUuid>>,
391    mut sandboxes: Vec<Sandbox>,
392    runner: Arc<dyn SandboxRunner>,
393    fifo_dir: Option<TempDir>,
394) -> Result<(), Error> {
395    assert_eq!(sandboxes.len(), job.group.executions.len());
396    let mut results = vec![None; job.group.executions.len()];
397    let mut outputs = HashMap::new();
398    let mut output_paths = HashMap::new();
399
400    // in case of simple executions there's no need to spawn the sandbox in a different thread and
401    // then join from here
402    if job.group.executions.len() == 1 {
403        let sandbox = sandboxes.pop().unwrap();
404        let result = match sandbox.run(runner.as_ref()) {
405            Ok(res) => res,
406            Err(e) => SandboxResult::Failed {
407                error: e.to_string(),
408            },
409        };
410        let exec = &job.group.executions[0];
411        let mut result = compute_execution_result(exec, result, &sandbox);
412        get_result_outputs(
413            exec,
414            &sandbox,
415            &mut outputs,
416            &mut output_paths,
417            &mut result.status,
418        );
419
420        results[0] = Some(result);
421    // this is the complex case: more than an execution (therefore more than a sandbox)
422    // All the sandboxes will run in a separate thread and this thread will wait all of them. When
423    // a sandbox is done, it signals to this thread the completion which simply computes the result.
424    // When all of them have finished this thread sends the result and exits.
425    } else {
426        let mut missing = job.group.executions.len();
427        let mut handles = Vec::new();
428        let (group_sender, receiver) = channel();
429        for (index, sandbox) in sandboxes.clone().into_iter().enumerate() {
430            handles.push(
431                spawn_sandbox(
432                    &job.group.description,
433                    sandbox,
434                    runner.clone(),
435                    index,
436                    group_sender.clone(),
437                )
438                .context("Failed to spawn sandbox thread")?,
439            );
440        }
441
442        while missing > 0 {
443            match receiver.recv() {
444                Ok((index, result)) => {
445                    assert!(results[index].is_none());
446
447                    let exec = &job.group.executions[index];
448                    let sandbox = &sandboxes[index];
449
450                    let mut result = compute_execution_result(exec, result, sandbox);
451                    // if the process didn't exit successfully, kill the remaining sandboxes
452                    if !result.status.is_success() {
453                        for (i, (res, sandbox)) in results.iter().zip(sandboxes.iter()).enumerate()
454                        {
455                            // do not kill the current process
456                            if i != index && res.is_none() {
457                                sandbox.kill();
458                            }
459                        }
460                    }
461
462                    get_result_outputs(
463                        exec,
464                        sandbox,
465                        &mut outputs,
466                        &mut output_paths,
467                        &mut result.status,
468                    );
469
470                    results[index] = Some(result);
471                    missing -= 1;
472                }
473                _ => bail!("The sandboxes didn't exit well"),
474            }
475        }
476        for handle in handles {
477            handle
478                .join()
479                .map_err(|e| anyhow!("Sandbox thread panicked: {:?}", e))?
480                .context("Sandbox thread failed")?;
481        }
482    }
483    // tell the server the results and the list of produced files
484    sender
485        .send(WorkerClientMessage::WorkerDone(
486            results.into_iter().map(Option::unwrap).collect(),
487            outputs.clone(),
488        ))
489        .context("Failed to send WorkerDone")?;
490    // wait for the list of files to send
491    match server_asked_files_receiver.recv() {
492        Ok(missing_files) => {
493            for uuid in missing_files {
494                if let Some(key) = outputs.get(&uuid) {
495                    sender
496                        .send(WorkerClientMessage::ProvideFile(uuid, key.clone()))
497                        .context("Failed to send ProvideFile")?;
498                    ChannelFileSender::send(&output_paths[&uuid], &sender)
499                        .context("Failed to send missing file")?;
500                } else {
501                    error!(
502                        "Server asked for file {}, which is not known to the worker",
503                        uuid
504                    );
505                }
506            }
507        }
508        Err(e) => {
509            // not receiving the list from the server means that the server is going down and does
510            // not bother of responding, letting the worker crash will crash the local executor.
511            // So just cleanup and exit without asking for more jobs.
512            warn!(
513                "List of missing files not received from the server: {:?}",
514                e
515            );
516            let mut job = current_job.lock().unwrap();
517            job.current_job = None;
518            job.current_sandboxes = None;
519            return Ok(());
520        }
521    }
522    // this job is completed, reset the worker and ask for more work
523    let mut job = current_job.lock().unwrap();
524    job.current_job = None;
525    job.current_sandboxes = None;
526    let _ = sender.send(WorkerClientMessage::GetWork);
527    // The sandbox may chmod -r the directory, revert it to allow deletion on drop
528    if let Some(fifo_dir) = fifo_dir {
529        let _ = std::fs::set_permissions(fifo_dir.path(), Permissions::from_mode(0o755));
530    }
531    Ok(())
532}

534/// Spawn the sandbox of an execution in a different thread and send to the group manager the
535/// results.
536fn spawn_sandbox(
537    description: &str,
538    sandbox: Sandbox,
539    runner: Arc<dyn SandboxRunner>,
540    index: usize,
541    group_sender: Sender<(usize, SandboxResult)>,
542) -> Result<JoinHandle<Result<(), Error>>, Error> {
543    Ok(thread::Builder::new()
544        .name(format!("Sandbox of {}", description))
545        .spawn(move || {
546            let res = match sandbox.run(runner.as_ref()) {
547                Ok(res) => res,
548                Err(e) => SandboxResult::Failed {
549                    error: e.to_string(),
550                },
551            };
552            group_sender.send((index, res))?;
553            Ok(())
554        })?)
555}

557/// Compute the [`ExecutionResult`](../task_maker_dag/struct.ExecutionResult.html) based on the
558/// result of the sandbox.
559fn compute_execution_result(
560    execution: &Execution,
561    result: SandboxResult,
562    sandbox: &Sandbox,
563) -> ExecutionResult {
564    match result {
565        SandboxResult::Success {
566            exit_status,
567            signal,
568            resources,
569            was_killed,
570        } => {
571            let stdout = capture_stream(&sandbox.stdout_path(), execution.capture_stdout);
572            let stderr = capture_stream(&sandbox.stderr_path(), execution.capture_stderr);
573            let status = match (&stdout, &stderr) {
574                (Ok(_), Ok(_)) => execution.status(exit_status, signal, &resources),
575                (Err(err), _) => ExecutionStatus::internal_error(format!(
576                    "Failed to read stdout file: {:?}",
577                    err
578                )),
579                (_, Err(err)) => ExecutionStatus::internal_error(format!(
580                    "Failed to read stderr file: {:?}",
581                    err
582                )),
583            };
584            ExecutionResult {
585                status,
586                resources,
587                stdout: stdout.ok().unwrap_or_default(),
588                was_killed,
589                was_cached: false,
590                stderr: stderr.ok().unwrap_or_default(),
591            }
592        }
593        SandboxResult::Failed { error } => ExecutionResult {
594            status: ExecutionStatus::InternalError(error),
595            resources: ExecutionResourcesUsage::default(),
596            stdout: None,
597            was_killed: false,
598            was_cached: false,
599            stderr: None,
600        },
601    }
602}

604/// Extract the output files from the result of the sandbox and store them in the provided HashMaps.
605fn get_result_outputs(
606    exec: &Execution,
607    sandbox: &Sandbox,
608    outputs: &mut HashMap<FileUuid, FileStoreKey>,
609    output_paths: &mut HashMap<FileUuid, PathBuf>,
610    status: &mut ExecutionStatus,
611) {
612    let mut add_file = |file: FileUuid, path: PathBuf| {
613        if path.exists() {
614            let key = FileStoreKey::from_file(&path);
615            match key {
616                Ok(key) => {
617                    outputs.insert(file, key);
618                    output_paths.insert(file, path);
619                }
620                Err(e) => {
621                    *status = ExecutionStatus::internal_error(format!(
622                        "Failed to get store key for {} at {}: {:?}",
623                        file,
624                        path.display(),
625                        e
626                    ));
627                }
628            }
629        } else {
630            outputs.insert(file, FileStoreKey::from_content(&[]));
631            output_paths.insert(file, "/dev/null".into());
632        }
633    };
634
635    if let Some(stdout) = &exec.stdout {
636        add_file(stdout.uuid, sandbox.stdout_path());
637    }
638    if let Some(stderr) = &exec.stderr {
639        add_file(stderr.uuid, sandbox.stderr_path());
640    }
641    for (path, file) in exec.outputs.iter() {
642        add_file(file.uuid, sandbox.output_path(path));
643    }
644}

646/// If `count` is `None` do not read anything, otherwise read at most that number of bytes from the
647/// `path`.
648fn capture_stream(path: &Path, count: Option<usize>) -> Result<Option<Vec<u8>>, Error> {
649    if let Some(count) = count {
650        let mut file = std::fs::File::open(path)?;
651        let mut result = Vec::new();
652        let mut buffer = vec![0; 1024];
653        let mut read = 0;
654        while read < count {
655            let n = file.read(&mut buffer)?;
656            // EOF
657            if n == 0 {
658                break;
659            } else {
660                result.extend_from_slice(&buffer[0..n]);
661                read += n;
662            }
663        }
664        Ok(Some(result))
665    } else {
666        Ok(None)
667    }
668}

682fn check_sandbox_is_supported(
683    sandbox_path: &Path,
684    runner: Arc<dyn SandboxRunner>,
685) -> Result<(), Error> {
686    let execution = Execution::new(
687        "Execution to check if sandbox is supported",
688        ExecutionCommand::system("true"),
689    );
690    let sandbox = Sandbox::new(sandbox_path, &execution, &Default::default(), None)?;
691    let result = sandbox.run(runner.as_ref())?;
692    match result {
693        SandboxResult::Failed { error } => bail!("Sandbox failed: {}", error),
694        SandboxResult::Success {
695            exit_status,
696            signal,
697            ..
698        } => {
699            if exit_status != 0 {
700                bail!("Sandbox failed: exited with {}", exit_status);
701            }
702            if let Some((signal, name)) = signal {
703                bail!("Sandbox failed: exited with signal {} ({})", signal, name);
704            }
705        }
706    }
707    Ok(())
708}
