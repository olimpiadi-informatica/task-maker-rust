1//! The protocol related structs and enums.
2//!
3//! # Protocol Specification
4//!
5//! The communication between the services is done using the abstraction of
6//! `std::sync::mpsc::channel`. The messages are serialized and then sent inside a `channel`. The
7//! way the messages are serialized is unspecified and
8//! [`serialize_into`](../../task_maker_exec/fn.serialize_into.html)/[`deserialize_from`](../../task_maker_exec/fn.deserialize_from.html)
9//! should be used instead of reading from / writing to the channels.
10//!
11//! There are 3 basic actors in the protocol:
12//!
13//! - The Client: the one who makes the DAG and is interested in the results;
14//! - The Server: the one who manages the execution of the DAGs;
15//! - The Worker: the one who is able to run an `Execution` inside a `Sandbox`.
16//!
17//! The clients are able to communicate to the server, the workers are able to communicate to the
18//! server and the server is able to communicate with both. The clients and the workers should not
19//! communicate directly.
20//!
21//! The 4 valid communication directions are:
22//! - `Client` — [`ExecutorClientMessage`](enum.ExecutorClientMessage.html) → `Server`
23//! - `Client` ← [`ExecutorServerMessage`](enum.ExecutorServerMessage.html) — `Server`
24//! - `Worker` — [`WorkerClientMessage`](enum.WorkerClientMessage.html) → `Server`
25//! - `Worker` ← [`WorkerServerMessage`](enum.WorkerServerMessage.html) — `Server`
26//!
27//! When an actor needs a file a particular series of messages is sent. Let's assume `A` wants a
28//! file from `B`:
29//! - `A` sends a `AskFile` to `B`
30//! - `B` answers with `ProvideFile` which triggers a protocol switch for sending the file
31//! - `B` sends raw data (`send_raw`) zero or more times
32//! - `B` sends empty raw data which triggers a protocol switch, back into normal mode
33

46/// Messages that the client sends to the server.
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub enum ExecutorClientMessage {
49    /// The client is asking to evaluate a DAG.
50    Evaluate {
51        /// The DAG to evaluate.
52        dag: Box<ExecutionDAGData>,
53        /// The list of the executions and files to keep track of.
54        callbacks: Box<ExecutionDAGWatchSet>,
55    },
56    /// The client is providing a file. After this message there is a protocol switch for the file
57    /// transmission.
58    ProvideFile(FileUuid, FileStoreKey),
59    /// The client is asking the server to send a file. After this message there is a protocol
60    /// switch for the file transmission.
61    AskFile(FileUuid, FileStoreKey, bool),
62    /// The client is asking to stop the evaluation. All the running executions will be killed and
63    /// no more execution will be run. All the callbacks will be called as usual.
64    Stop,
65    /// The client is asking for the server status. After this message the client should expect a
66    /// [`Status`](enum.ExecutorServerMessage.html#variant.Status) message back.
67    Status,
68}

70/// Messages that the server sends to the client.
71#[derive(Debug, Clone, Serialize, Deserialize)]
72pub enum ExecutorServerMessage {
73    /// The server needs the file with that Uuid. The client must send back that file in order to
74    /// proceed with the execution.
75    AskFile(FileUuid),
76    /// The server is sending a file. After this message there is a protocol switch for the file
77    /// transmission protocol. The second entry is true if the generation of the file was
78    /// successful.
79    ProvideFile(FileUuid, bool),
80    /// The execution has started on a worker.
81    NotifyStart(ExecutionUuid, WorkerUuid),
82    /// The execution has completed with that result.
83    NotifyDone(ExecutionUuid, ExecutionResult),
84    /// The execution has been skipped.
85    NotifySkip(ExecutionUuid),
86    /// There was an error during the evaluation.
87    Error(String),
88    /// The server status as asked by the client.
89    Status(ExecutorStatus<Duration>),
90    /// The evaluation of the DAG is complete, this message will close the connection.
91    Done(Vec<(FileUuid, FileStoreKey, bool)>),
92}

94/// Messages sent by the workers to the server.
95#[derive(Debug, Clone, Serialize, Deserialize)]
96pub enum WorkerClientMessage {
97    /// The worker is ready for some job. The worker will wait for a
98    /// [`Work`](enum.WorkerServerMessage.html#variant.Work) message.
99    GetWork,
100    /// The worker completed the job with this result producing those files. The actual files will
101    /// be sent immediately after using `ProvideFile` messages.
102    /// The list of `ExecutionResult` contains the results of all the executions inside the group,
103    /// in the same order.
104    WorkerDone(Vec<ExecutionResult>, HashMap<FileUuid, FileStoreKey>),
105    /// The worker is sending a file to the server. After this message there is a protocol switch
106    /// for the file transmission.
107    ProvideFile(FileUuid, FileStoreKey),
108    /// The worker needs a file from the server. The server should send back that file in order to
109    /// run the execution on the worker.
110    AskFile(FileStoreKey),
111}
112
113/// Messages sent by the server to the worker.
114#[derive(Debug, Clone, Serialize, Deserialize)]
115pub enum WorkerServerMessage {
116    /// The job the worker should do. Boxed to reduce the enum size.
117    Work(Box<WorkerJob>),
118    /// Stop the current worker sandbox if currently running the specified execution.
119    KillJob(ExecutionGroupUuid),
120    /// The file the workers as asked. After this message there is a protocol switch for the file
121    /// transmission.
122    ProvideFile(FileStoreKey),
123    /// The worker completed the execution and produced some files, the server asks the ones that
124    /// are missing using this message.
125    AskFiles(Vec<FileUuid>),
126    /// Ask the worker to exit.
127    Exit,
128}

130/// An iterator over the byte chunks sent during the file transfer mode in a channel.
131pub struct ChannelFileIterator<'a, T>
132where
133    T: Send + Sync + DeserializeOwned,
134{
135    /// Reference to the channel from where to read
136    reader: &'a ChannelReceiver<T>,
137}
138
139impl<'a, T> ChannelFileIterator<'a, T>
140where
141    T: 'static + Send + Sync + DeserializeOwned,
142{
143    /// Create a new iterator over a receiver channel.
144    pub fn new(reader: &'a ChannelReceiver<T>) -> ChannelFileIterator<'a, T> {
145        ChannelFileIterator { reader }
146    }
147}
148
149impl<T> Iterator for ChannelFileIterator<'_, T>
150where
151    T: 'static + Send + Sync + DeserializeOwned,
152{
153    type Item = Vec<u8>;
154    fn next(&mut self) -> Option<Self::Item> {
155        // errors cannot be handled in this iterator yet
156        let data = self.reader.recv_raw().expect("deserialize error");
157        if data.is_empty() {
158            None
159        } else {
160            Some(data)
161        }
162    }
163}

165/// Utility function to send a file to a channel using [`send_raw`](https://docs.rs/ductile/0.1.0/ductile/struct.ChannelSender.html#method.send_raw).
166pub struct ChannelFileSender;
167
168impl ChannelFileSender {
169    /// Send a local file to a channel using `send_raw`.
170    pub fn send<P: AsRef<Path>, T>(path: P, sender: &ChannelSender<T>) -> Result<(), Error>
171    where
172        T: 'static + Send + Sync + Serialize,
173    {
174        let path = path.as_ref();
175        let iterator = ReadFileIterator::new(path)
176            .with_context(|| format!("Failed to read file to send: {}", path.display()))?;
177        for buf in iterator {
178            sender.send_raw(&buf).context("Failed to send file chunk")?;
179        }
180        sender
181            .send_raw(&[])
182            .context("Failed to send file terminator")?;
183        Ok(())
184    }
185
186    /// Send the file content to a channel using `send_raw`.
187    pub fn send_data<T>(data: Vec<u8>, sender: &ChannelSender<T>) -> Result<(), Error>
188    where
189        T: 'static + Send + Sync + Serialize,
190    {
191        sender
192            .send_raw(&data)
193            .context("Failed to send file chunk")?;
194        // Send the EOF chunk only if the buffer is not empty (otherwise we would send EOF twice
195        // breaking the protocol).
196        if !data.is_empty() {
197            sender
198                .send_raw(&[])
199                .context("Failed to send file terminator")?;
200        }
201        Ok(())
202    }
203}

205#[cfg(test)]
206mod tests {
207    use super::*;
208
209    #[test]
210    fn test_send_file() {
211        let tmpdir = tempfile::TempDir::new().unwrap();
212        std::fs::write(tmpdir.path().join("file.txt"), "hello world").unwrap();
213
214        let (sender, receiver) = new_local_channel::<()>();
215        let receiver = ChannelFileIterator::new(&receiver);
216        ChannelFileSender::send(tmpdir.path().join("file.txt"), &sender).unwrap();
217        let data: Vec<u8> = receiver.flat_map(|d| d.into_iter()).collect();
218        assert_eq!(String::from_utf8(data).unwrap(), "hello world");
219    }
220
221    #[test]
222    fn test_send_content() {
223        let (sender, receiver) = new_local_channel::<()>();
224        let receiver = ChannelFileIterator::new(&receiver);
225        ChannelFileSender::send_data(b"hello world".to_vec(), &sender).unwrap();
226        let data: Vec<u8> = receiver.flat_map(|d| d.into_iter()).collect();
227        assert_eq!(String::from_utf8(data).unwrap(), "hello world");
228    }
