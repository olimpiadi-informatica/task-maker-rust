6/// An error in the DAG structure.
7#[derive(Debug, Error)]
8pub enum DAGError {
9    /// A file is used as input in an execution but it's missing, or a callback is registered on a
10    /// file but it's missing.
11    #[error("missing file {description} ({uuid})")]
12    MissingFile {
13        /// The UUID of the missing file.
14        uuid: FileUuid,
15        /// The description of the missing file.
16        description: String,
17    },
18    /// Stdout/Stderr capture is requested, but a UUID for them is missing.
19    #[error("missing UUID for captured {stream} on execution {uuid} ({description})")]
20    InvalidCapture {
21        /// Either "stdout" or "stderr".
22        stream: String,
23        /// The UUID of the missing file.
24        uuid: ExecutionUuid,
25        /// The description of the missing file.
26        description: String,
27    },
28    /// A callback is registered on an execution but it's missing.
29    #[error("missing execution {uuid}")]
30    MissingExecution {
31        /// The UUID of the missing execution.
32        uuid: ExecutionUuid,
33    },
34    /// There is a dependency cycle in the DAG.
35    #[error("detected dependency cycle, '{description}' is in the cycle")]
36    CycleDetected {
37        /// The description of an execution inside the cycle.
38        description: String,
39    },
40    /// There is a duplicate file UUID.
41    #[error("duplicate file UUID {uuid}")]
42    DuplicateFileUUID {
43        /// The duplicated UUID.
44        uuid: FileUuid,
45    },
46    /// There is a duplicate Fifo UUID.
47    #[error("duplicate FIFO UUID {uuid}")]
48    DuplicateFifoUUID {
49        /// The duplicated UUID.
50        uuid: FifoUuid,
51    },
52    /// There is a duplicate execution UUID.
53    #[error("duplicate execution UUID {uuid}")]
54    DuplicateExecutionUUID {
55        /// The duplicated UUID.
56        uuid: FileUuid,
57    },
58    /// There is an invalid execution group.
59    #[error("empty execution group {uuid}")]
60    EmptyGroup {
61        /// The UUID of the execution group.
62        uuid: ExecutionGroupUuid,
63    },
64}

66/// Validate the DAG checking if all the required pieces are present and they actually make a DAG.
67/// It's checked that no duplicated UUID are present, no files are missing, all the executions are
68/// reachable and no cycles are present.
69pub fn check_dag(dag: &ExecutionDAGData, callbacks: &ExecutionDAGWatchSet) -> Result<(), DAGError> {
70    let mut dependencies: HashMap<FileUuid, Vec<ExecutionGroupUuid>> = HashMap::new();
71    let mut num_dependencies: HashMap<ExecutionGroupUuid, usize> = HashMap::new();
72    let mut known_files: HashSet<FileUuid> = HashSet::new();
73    let mut known_execs: HashSet<ExecutionUuid> = HashSet::new();
74    let mut ready_groups: VecDeque<ExecutionGroupUuid> = VecDeque::new();
75    let mut ready_files: VecDeque<FileUuid> = VecDeque::new();
76
77    let mut add_dependency = |file: FileUuid, group: ExecutionGroupUuid| {
78        dependencies.entry(file).or_default().push(group);
79    };
80
81    // add the executions and check for duplicated UUIDs
82    for (group_uuid, group) in dag.execution_groups.iter() {
83        if group.executions.is_empty() {
84            return Err(DAGError::EmptyGroup { uuid: *group_uuid });
85        }
86        let mut fifo_uuids = HashSet::new();
87        for fifo in group.fifo.iter() {
88            if !fifo_uuids.insert(fifo.uuid) {
89                return Err(DAGError::DuplicateFifoUUID { uuid: fifo.uuid });
90            }
91        }
92        let mut count = 0;
93        for exec in &group.executions {
94            let deps = exec.dependencies();
95            if !known_execs.insert(exec.uuid) {
96                return Err(DAGError::DuplicateExecutionUUID { uuid: exec.uuid });
97            }
98            count += deps.len();
99            for dep in deps.into_iter() {
100                add_dependency(dep, *group_uuid);
101            }
102            if exec.capture_stdout.is_some() && exec.stdout.is_none() {
103                return Err(DAGError::InvalidCapture {
104                    stream: "stdout".to_string(),
105                    uuid: exec.uuid,
106                    description: exec.description.clone(),
107                });
108            }
109            if exec.capture_stderr.is_some() && exec.stderr.is_none() {
110                return Err(DAGError::InvalidCapture {
111                    stream: "stderr".to_string(),
112                    uuid: exec.uuid,
113                    description: exec.description.clone(),
114                });
115            }
116            for out in exec.outputs().into_iter() {
117                if !known_files.insert(out) {
118                    return Err(DAGError::DuplicateFileUUID { uuid: out });
119                }
120            }
121        }
122        num_dependencies.insert(*group_uuid, count);
123        if count == 0 {
124            ready_groups.push_back(*group_uuid);
125        }
126    }
127    // add the provided files
128    for uuid in dag.provided_files.keys() {
129        ready_files.push_back(*uuid);
130        if !known_files.insert(*uuid) {
131            return Err(DAGError::DuplicateFileUUID { uuid: *uuid });
132        }
133    }
134    // visit the DAG for finding the unreachable executions / cycles
135    while !ready_groups.is_empty() || !ready_files.is_empty() {
136        for file in ready_files.drain(..) {
137            if !dependencies.contains_key(&file) {
138                continue;
139            }
140            for group_uuid in dependencies[&file].iter() {
141                let num_deps = num_dependencies
142                    .get_mut(group_uuid)
143                    .expect("num_dependencies of an unknown execution group");
144                assert_ne!(
145                    *num_deps, 0,
146                    "num_dependencies is going to be negative for {}",
147                    group_uuid
148                );
149                *num_deps -= 1;
150                if *num_deps == 0 {
151                    ready_groups.push_back(*group_uuid);
152                }
153            }
154        }
155        for group_uuid in ready_groups.drain(..) {
156            let group = dag
157                .execution_groups
158                .get(&group_uuid)
159                .expect("No such exec group");
160            for exec in &group.executions {
161                for file in exec.outputs().into_iter() {
162                    ready_files.push_back(file);
163                }
164            }
165        }
166    }
167    // search for unreachable execution / cycles
168    for (group_uuid, count) in num_dependencies.iter() {
169        if *count == 0 {
170            continue;
171        }
172        let group = &dag.execution_groups[group_uuid];
173        for exec in &group.executions {
174            for dep in exec.dependencies().iter() {
175                if !known_files.contains(dep) {
176                    return Err(DAGError::MissingFile {
177                        uuid: *dep,
178                        description: format!("Dependency of '{}'", exec.description),
179                    });
180                }
181            }
182        }
183        return Err(DAGError::CycleDetected {
184            description: dag.execution_groups[group_uuid].description.clone(),
185        });
186    }
187    // check the file callbacks
188    for file in callbacks.files.iter() {
189        if !known_files.contains(file) {
190            return Err(DAGError::MissingFile {
191                uuid: *file,
192                description: "File required by a callback".to_owned(),
193            });
194        }
195    }
196    // check the execution callbacks
197    for exec in callbacks.executions.iter() {
198        if !known_execs.contains(exec) {
199            return Err(DAGError::MissingExecution { uuid: *exec });
200        }
201    }
202    Ok(())
203}

210    #[test]
211    fn test_missing_file() {
212        let mut dag = ExecutionDAG::new();
213        let mut exec = Execution::new("exec", ExecutionCommand::local("foo"));
214        let file = File::new("file");
215        exec.stdin(file);
216        dag.add_execution(exec);
217        assert!(check_dag(&dag.data, &ExecutionDAGWatchSet::default()).is_err());
218    }
