19/// Message coming from the Scheduler or the Executor for the WorkerManager
20#[allow(clippy::large_enum_variant)]
21pub(crate) enum WorkerManagerInMessage {
22    /// A new worker has connected. The WorkerManager will take care of it.
23    WorkerConnected { worker: WorkerConn },
24    /// A worker has disconnected. This message is sent by the WorkerManager itself, from a
25    /// different thread.
26    WorkerDisconnected { worker: WorkerUuid },
27    /// The scheduler sent a new job for a worker. The WorkerManager will forward the job to the
28    /// actual worker.
29    WorkerJob { worker: WorkerUuid, job: WorkerJob },
30    /// The scheduler is asking a worker to stop doing a job, if the worker is still working on
31    /// that.
32    StopWorkerJob {
33        worker: WorkerUuid,
34        job: ExecutionGroupUuid,
35    },
36    /// The WorkerManager is asked to exit and tell all the connected worker to exit too.
37    Exit,
38}
39
40/// The entity that manages the connections with the workers, eventually writing files to disk and
41/// telling to the `Scheduler` the connection and disconnection of the workers.
42pub(crate) struct WorkerManager {
43    /// A reference to the file store.
44    file_store: Arc<FileStore>,
45    /// A channel for sending the messages to the scheduler.
46    scheduler: Sender<SchedulerInMessage>,
47    /// A channel for sending the messages to the WorkerManager itself. It is used by the threads
48    /// that manage the actual workers for sending back the notification of disconnection.
49    sender: Sender<WorkerManagerInMessage>,
50    /// The receiver of the messages for the worker manager.
51    receiver: Receiver<WorkerManagerInMessage>,
52}
53
54impl WorkerManager {
55    /// Make a new `WorkerManager` based on the specified file store, talking to the specified
56    /// scheduler. `sender` is just a sender that sends messages to the `receiver`, this is needed
57    /// internally for sending back the disconnection notification from other threads.
58    pub fn new(
59        file_store: Arc<FileStore>,
60        scheduler: Sender<SchedulerInMessage>,
61        sender: Sender<WorkerManagerInMessage>,
62        receiver: Receiver<WorkerManagerInMessage>,
63    ) -> WorkerManager {
64        WorkerManager {
65            file_store,
66            scheduler,
67            sender,
68            receiver,
69        }
70    }
71
72    /// Run the worker manager blocking until an exit message is received. On exiting the connected
73    /// workers will stop.
74    pub fn run(self) -> Result<(), Error> {
75        let mut connected_workers: HashMap<WorkerUuid, ChannelSender<WorkerServerMessage>> =
76            HashMap::new();
77        while let Ok(message) = self.receiver.recv() {
78            match message {
79                WorkerManagerInMessage::WorkerConnected { worker } => {
80                    if connected_workers.contains_key(&worker.uuid) {
81                        warn!("Duplicate worker uuid");
82                        continue;
83                    }
84                    connected_workers.insert(worker.uuid, worker.sender.clone());
85                    info!("Worker {} ({}) connected", worker.name, worker.uuid);
86                    let scheduler = self.scheduler.clone();
87                    let file_store = self.file_store.clone();
88                    let sender = self.sender.clone();
89                    thread::Builder::new()
90                        .name(format!(
91                            "Manager of worker {} ({})",
92                            worker.name, worker.uuid
93                        ))
94                        .spawn(move || {
95                            if let Err(e) =
96                                WorkerManager::worker_thread(worker, scheduler, sender, file_store)
97                            {
98                                warn!("The manager of a worker failed: {:?}", e);
99                            }
100                        })
101                        .context("Failed to spawn manager for a worker")?;
102                }
103                WorkerManagerInMessage::WorkerDisconnected { worker } => {
104                    connected_workers
105                        .remove(&worker)
106                        .context("Unknown worker disconnected")?;
107                }
108                WorkerManagerInMessage::WorkerJob { worker, job } => {
109                    // if the worker is not present, it means it has just disconnected. The
110                    // scheduler should be already informed and should have resheduled the job.
111                    if let Some(sender) = connected_workers.get(&worker) {
112                        sender
113                            .send(WorkerServerMessage::Work(Box::new(job)))
114                            .context("Failed to send Work to worker")?;
115                    }
116                }
117                WorkerManagerInMessage::Exit => {
118                    debug!("Worker manager asked to exit");
119                    break;
120                }
121                WorkerManagerInMessage::StopWorkerJob { worker, job } => {
122                    if let Some(sender) = connected_workers.get(&worker) {
123                        sender
124                            .send(WorkerServerMessage::KillJob(job))
125                            .context("Failed to send KillJob to worker")?;
126                    }
127                }
128            }
129        }
130        debug!("Worker manager exiting");
131        for (worker, sender) in connected_workers.iter() {
132            if sender.send(WorkerServerMessage::Exit).is_err() {
133                warn!("Cannot tell worker {} to exit", worker);
134            }
135        }
136        Ok(())
137    }

139    /// Thread body that manages the actual connection with a worker. `worker_manager` will send
140    /// messages back to the `WorkerManager` main thread for the notification about the
141    /// disconnection of this worker.
142    fn worker_thread(
143        worker: WorkerConn,
144        scheduler: Sender<SchedulerInMessage>,
145        worker_manager: Sender<WorkerManagerInMessage>,
146        file_store: Arc<FileStore>,
147    ) -> Result<(), Error> {
148        while let Ok(message) = worker.receiver.recv() {
149            match message {
150                WorkerClientMessage::GetWork => {
151                    // the worker is asking for more work to do
152                    let res = scheduler.send(SchedulerInMessage::WorkerConnected {
153                        uuid: worker.uuid,
154                        name: worker.name.clone(),
155                    });
156                    if res.is_err() {
157                        // the scheduler is gone
158                        break;
159                    }
160                }
161                WorkerClientMessage::AskFile(key) => {
162                    // the worker is asking for a file it doesn't have locally stored
163                    let handle = file_store
164                        .get(&key)
165                        .context("Worker is asking for an unknown file")?;
166                    worker
167                        .sender
168                        .send(WorkerServerMessage::ProvideFile(key))
169                        .context("Failed to send ProvideFile to worker")?;
170                    ChannelFileSender::send(handle.path(), &worker.sender)
171                        .context("Failed to send file to worker")?;
172                }
173                WorkerClientMessage::ProvideFile(_, _) => {
174                    // the worker should not provide files unless just after a WorkerDone message is
175                    // received
176                    unreachable!("Unexpected ProvideFile from worker");
177                }
178                WorkerClientMessage::WorkerDone(result, outputs) => {
179                    // the worker completed its job and will send the produced files
180                    let mut output_handlers = HashMap::new();
181                    let mut missing_files = Vec::new();
182                    for (uuid, key) in &outputs {
183                        if let Some(handle) = file_store.get(key) {
184                            output_handlers.insert(*uuid, handle);
185                        } else {
186                            missing_files.push(*uuid);
187                        }
188                    }
189                    let num_missing = missing_files.len();
190                    info!(
191                        "Asking worker {} for {} missing files",
192                        worker.uuid, num_missing
193                    );
194                    worker
195                        .sender
196                        .send(WorkerServerMessage::AskFiles(missing_files))
197                        .context("Failed to send AskFiles to worker")?;
198                    for _ in 0..num_missing {
199                        let message = worker
200                            .receiver
201                            .recv()
202                            .context("Failed to receive file from worker")?;
203                        if let WorkerClientMessage::ProvideFile(uuid, key) = message {
204                            let handle = file_store
205                                .store(&key, ChannelFileIterator::new(&worker.receiver))
206                                .context("Failed to store worker-provided file")?;
207                            output_handlers.insert(uuid, handle);
208                        } else {
209                            bail!("Unexpected message from worker: {:?}", message);
210                        }
211                    }
212                    let mex = SchedulerInMessage::WorkerResult {
213                        worker: worker.uuid,
214                        result,
215                        outputs: output_handlers,
216                    };
217                    if let Err(e) = scheduler.send(mex) {
218                        warn!("Failed to send message to scheduler: {:?}", e);
219                        break;
220                    }
221                }
222            }
223        }
224        // when the worker disconnects, tell the scheduler that the worker is no longer alive (thus
225        // rescheduling the job if needed).
226        if scheduler
227            .send(SchedulerInMessage::WorkerDisconnected { uuid: worker.uuid })
228            .is_err()
229        {
230            debug!("Cannot tell the scheduler that a worker left, maybe it's gone");
231        }
232        // send back to the WorkerManager a message, letting it know that the worker is no longer
233        // connected, thus removing it from the list.
234        if worker_manager
235            .send(WorkerManagerInMessage::WorkerDisconnected {
236                worker: worker.uuid,
237            })
238            .is_err()
239        {
240            debug!("Worker manager is gone");
241        }
242        Ok(())
243    }
